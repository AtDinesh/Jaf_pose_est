00107 /*
00108  * To conserve space, bigram info is kept in many tables.  Since the number
00109  * of distinct values << #bigrams, these table indices can be 16-bit values.
00110  * prob2 and bo_wt2 are such indices, but keeping trigram index is less easy.
00111  * It is supposed to be the index of the first trigram entry for each bigram.
00112  * But such an index cannot be represented in 16-bits, hence the following
00113  * segmentation scheme: Partition bigrams into segments of BG_SEG_SZ
00114  * consecutive entries, such that #trigrams in each segment <= 2**16 (the
00115  * corresponding trigram segment).  The bigram_t.trigrams value is then a
00116  * 16-bit relative index within the trigram segment.  A separate table--
00117  * lm_t.tseg_base--has the index of the 1st trigram for each bigram segment.
00118  */
00119 #define BG_SEG_SZ       512     /* chosen so that #trigram/segment <= 2**16 */
00120 #define LOG_BG_SEG_SZ   9
