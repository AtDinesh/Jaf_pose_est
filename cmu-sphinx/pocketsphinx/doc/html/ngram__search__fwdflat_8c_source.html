00063 static void
00064 ngram_fwdflat_expand_all(ngram_search_t *ngs)
00065 {
00066     int n_words, i;
00067 
00068     /* For all "real words" (not fillers or <s>/</s>) in the dictionary,
00069      *
00070      * 1) Add the ones which are in the LM to the fwdflat wordlist
00071      * 2) And to the expansion list (since we are expanding all)
00072      */
00073     ngs->n_expand_words = 0;
00074     n_words = ps_search_n_words(ngs);
00075     bitvec_clear_all(ngs->expand_word_flag, ps_search_n_words(ngs));
00076     for (i = 0; i < n_words; ++i) {
00077         if (!ngram_model_set_known_wid(ngs->lmset,
00078                                        dict_basewid(ps_search_dict(ngs),i)))
00079             continue;
00080         ngs->fwdflat_wordlist[ngs->n_expand_words] = i;
00081         ngs->expand_word_list[ngs->n_expand_words] = i;
00082         bitvec_set(ngs->expand_word_flag, i);
00083         ngs->n_expand_words++;
00084     }
00085     E_INFO("Utterance vocabulary contains %d words\n", ngs->n_expand_words);
00086     ngs->expand_word_list[ngs->n_expand_words] = -1;
00087     ngs->fwdflat_wordlist[ngs->n_expand_words] = -1;
00088 }

00226 static void
00227 build_fwdflat_wordlist(ngram_search_t *ngs)
00228 {
00229     int32 i, f, sf, ef, wid, nwd;
00230     bptbl_t *bp;
00231     ps_latnode_t *node, *prevnode, *nextnode;
00232 
00233     /* No tree-search, use statically allocated wordlist. */
00234     if (!ngs->fwdtree)
00235         return;
00236 
00237     memset(ngs->frm_wordlist, 0, ngs->n_frame_alloc * sizeof(*ngs->frm_wordlist));
00238 
00239     /* Scan the backpointer table for all active words and record
00240      * their exit frames. */
00241     for (i = 0, bp = ngs->bp_table; i < ngs->bpidx; i++, bp++) {
00242         sf = (bp->bp < 0) ? 0 : ngs->bp_table[bp->bp].frame + 1;
00243         ef = bp->frame;
00244         wid = bp->wid;
00245 
00246         /* Anything that can be transitioned to in the LM can go in
00247          * the word list. */
00248         if (!ngram_model_set_known_wid(ngs->lmset,
00249                                        dict_basewid(ps_search_dict(ngs), wid)))
00250             continue;
00251 
00252         /* Look for it in the wordlist. */
00253         for (node = ngs->frm_wordlist[sf]; node && (node->wid != wid);
00254              node = node->next);
00255 
00256         /* Update last end frame. */
00257         if (node)
00258             node->lef = ef;
00259         else {
00260             /* New node; link to head of list */
00261             node = listelem_malloc(ngs->latnode_alloc);
00262             node->wid = wid;
00263             node->fef = node->lef = ef;
00264 
00265             node->next = ngs->frm_wordlist[sf];
00266             ngs->frm_wordlist[sf] = node;
00267         }
00268     }
00269 
00270     /* Eliminate "unlikely" words, for which there are too few end points */
00271     for (f = 0; f < ngs->n_frame; f++) {
00272         prevnode = NULL;
00273         for (node = ngs->frm_wordlist[f]; node; node = nextnode) {
00274             nextnode = node->next;
00275             /* Word has too few endpoints */
00276             if ((node->lef - node->fef < ngs->min_ef_width) ||
00277                 /* Word is </s> and doesn't actually end in last frame */
00278                 ((node->wid == ps_search_finish_wid(ngs)) && (node->lef < ngs->n_frame - 1))) {
00279                 if (!prevnode)
00280                     ngs->frm_wordlist[f] = nextnode;
00281                 else
00282                     prevnode->next = nextnode;
00283                 listelem_free(ngs->latnode_alloc, node);
00284             }
00285             else
00286                 prevnode = node;
00287         }
00288     }
00289 
00290     /* Form overall wordlist for 2nd pass */
00291     nwd = 0;
00292     bitvec_clear_all(ngs->word_active, ps_search_n_words(ngs));
00293     for (f = 0; f < ngs->n_frame; f++) {
00294         for (node = ngs->frm_wordlist[f]; node; node = node->next) {
00295             if (!bitvec_is_set(ngs->word_active, node->wid)) {
00296                 bitvec_set(ngs->word_active, node->wid);
00297                 ngs->fwdflat_wordlist[nwd++] = node->wid;
00298             }
00299         }
00300     }
00301     ngs->fwdflat_wordlist[nwd] = -1;
00302     E_INFO("Utterance vocabulary contains %d words\n", nwd);
00303 }

00308 static void
00309 build_fwdflat_chan(ngram_search_t *ngs)
00310 {
00311     int32 i, wid, p;
00312     root_chan_t *rhmm;
00313     chan_t *hmm, *prevhmm;
00314     dict_t *dict;
00315     dict2pid_t *d2p;
00316 
00317     dict = ps_search_dict(ngs);
00318     d2p = ps_search_dict2pid(ngs);
00319 
00320     /* Build word HMMs for each word in the lattice. */
00321     for (i = 0; ngs->fwdflat_wordlist[i] >= 0; i++) {
00322         wid = ngs->fwdflat_wordlist[i];
00323 
00324         /* Single-phone words are permanently allocated */
00325         if (dict_is_single_phone(dict, wid))
00326             continue;
00327 
00328         assert(ngs->word_chan[wid] == NULL);
00329 
00330         /* Multiplex root HMM for first phone (one root per word, flat
00331          * lexicon).  diphone is irrelevant here, for the time being,
00332          * at least. */
00333         rhmm = listelem_malloc(ngs->root_chan_alloc);
00334         rhmm->ci2phone = dict_second_phone(dict, wid);
00335         rhmm->ciphone = dict_first_phone(dict, wid);
00336         rhmm->next = NULL;
00337         hmm_init(ngs->hmmctx, &rhmm->hmm, TRUE,
00338                  bin_mdef_pid2ssid(ps_search_acmod(ngs)->mdef, rhmm->ciphone),
00339                  bin_mdef_pid2tmatid(ps_search_acmod(ngs)->mdef, rhmm->ciphone));
00340 
00341         /* HMMs for word-internal phones */
00342         prevhmm = NULL;
00343         for (p = 1; p < dict_pronlen(dict, wid) - 1; p++) {
00344             hmm = listelem_malloc(ngs->chan_alloc);
00345             hmm->ciphone = dict_pron(dict, wid, p);
00346             hmm->info.rc_id = (p == dict_pronlen(dict, wid) - 1) ? 0 : -1;
00347             hmm->next = NULL;
00348             hmm_init(ngs->hmmctx, &hmm->hmm, FALSE,
00349                      dict2pid_internal(d2p,wid,p), 
00350                      bin_mdef_pid2tmatid(ps_search_acmod(ngs)->mdef, hmm->ciphone));
00351 
00352             if (prevhmm)
00353                 prevhmm->next = hmm;
00354             else
00355                 rhmm->next = hmm;
00356 
00357             prevhmm = hmm;
00358         }
00359 
00360         /* Right-context phones */
00361         ngram_search_alloc_all_rc(ngs, wid);
00362 
00363         /* Link in just allocated right-context phones */
00364         if (prevhmm)
00365             prevhmm->next = ngs->word_chan[wid];
00366         else
00367             rhmm->next = ngs->word_chan[wid];
00368         ngs->word_chan[wid] = (chan_t *) rhmm;
00369     }
00370 
00371 }

00538             }
00539             else {
00540                 assert(dict_is_single_phone(ps_search_dict(ngs), w));
00541 
00542                 /* Word exit for single-phone words (where did their
00543                  * whmms come from?) (either from
00544                  * ngram_search_fwdtree, or from
00545                  * ngram_fwdflat_allocate_1ph(), that's where) */
00546                 if (newscore BETTER_THAN wordthresh) {
00547                     ngram_search_save_bp(ngs, cf, w, newscore,
00548                                          hmm_out_history(&rhmm->hmm), 0);
00549                 }
00550             }
00551         }
00552 
00553         /* Transitions out of non-root channels. */
00554         for (hmm = rhmm->next; hmm; hmm = hmm->next) {
00555             if (hmm_frame(&hmm->hmm) >= cf) {
00556                 /* Propagate forward HMMs inside the beam. */
00557                 if (hmm_bestscore(&hmm->hmm) BETTER_THAN thresh) {
00558                     hmm_frame(&hmm->hmm) = nf;
00559                     bitvec_set(ngs->word_active, w);
00560 
00561                     newscore = hmm_out_score(&hmm->hmm);
00562                     /* Word-internal phones */
00563                     if (hmm->info.rc_id < 0) {
00564                         newscore += pip;
00565                         if (newscore BETTER_THAN thresh) {
00566                             nexthmm = hmm->next;
00567                             /* Enter all right-context phones. */
00568                             if (nexthmm->info.rc_id >= 0) {
00569                                  for (; nexthmm; nexthmm = nexthmm->next) {
00570                                     if ((hmm_frame(&nexthmm->hmm) < cf)
00571                                         || (newscore BETTER_THAN
00572                                             hmm_in_score(&nexthmm->hmm))) {
00573                                         hmm_enter(&nexthmm->hmm,
00574                                                   newscore,
00575                                                   hmm_out_history(&hmm->hmm),
00576                                                   nf);
00577                                     }
00578                                 }
00579                             }
00580                             /* Enter single word-internal phone. */
00581                             else {
00582                                 if ((hmm_frame(&nexthmm->hmm) < cf)
00583                                     || (newscore BETTER_THAN
00584                                         hmm_in_score(&nexthmm->hmm))) {
00585                                     hmm_enter(&nexthmm->hmm, newscore,
00586                                               hmm_out_history(&hmm->hmm), nf);
00587                                 }
00588                             }
00589                         }
00590                     }
00591                     /* Right-context phones - apply word beam and exit. */
00592                     else {
00593                         if (newscore BETTER_THAN wordthresh) {
00594                             ngram_search_save_bp(ngs, cf, w, newscore,
00595                                                  hmm_out_history(&hmm->hmm),
00596                                                  hmm->info.rc_id);
00597                         }
00598                     }
00599                 }
00600                 /* Zero out inactive HMMs. */
00601                 else if (hmm_frame(&hmm->hmm) != nf) {
00602                     hmm_clear_scores(&hmm->hmm);
00603                 }
00604             }
00605         }
00606     }

00609 static void
00610 get_expand_wordlist(ngram_search_t *ngs, int32 frm, int32 win)
00611 {
00612     int32 f, sf, ef;
00613     ps_latnode_t *node;
00614 
00615     if (!ngs->fwdtree) {
00616         ngs->st.n_fwdflat_word_transition += ngs->n_expand_words;
00617         return;
00618     }
00619 
00620     sf = frm - win;
00621     if (sf < 0)
00622         sf = 0;
00623     ef = frm + win;
00624     if (ef > ngs->n_frame)
00625         ef = ngs->n_frame;
00626 
00627     bitvec_clear_all(ngs->expand_word_flag, ps_search_n_words(ngs));
00628     ngs->n_expand_words = 0;
00629 
00630     for (f = sf; f < ef; f++) {
00631         for (node = ngs->frm_wordlist[f]; node; node = node->next) {
00632             if (!bitvec_is_set(ngs->expand_word_flag, node->wid)) {
00633                 ngs->expand_word_list[ngs->n_expand_words++] = node->wid;
00634                 bitvec_set(ngs->expand_word_flag, node->wid);
00635             }
00636         }
00637     }
00638     ngs->expand_word_list[ngs->n_expand_words] = -1;
00639     ngs->st.n_fwdflat_word_transition += ngs->n_expand_words;
00640 }

00642 static void
00643 fwdflat_word_transition(ngram_search_t *ngs, int frame_idx)
00644 {
00645     int32 cf, nf, b, thresh, pip, i, w, newscore;
00646     int32 best_silrc_score = 0, best_silrc_bp = 0;      /* FIXME: good defaults? */
00647     bptbl_t *bp;
00648     int32 *rcss;
00649     root_chan_t *rhmm;
00650     int32 *awl;
00651     float32 lwf;
00652     dict_t *dict = ps_search_dict(ngs);
00653     dict2pid_t *d2p = ps_search_dict2pid(ngs);
00654 
00655     cf = frame_idx;
00656     nf = cf + 1;
00657     thresh = ngs->best_score + ngs->fwdflatbeam;
00658     pip = ngs->pip;
00659     best_silrc_score = WORST_SCORE;
00660     lwf = ngs->fwdflat_fwdtree_lw_ratio;
00661 
00662     /* Search for all words starting within a window of this frame.
00663      * These are the successors for words exiting now. */
00664     get_expand_wordlist(ngs, cf, ngs->max_sf_win);
00665 
00666     /* Scan words exited in current frame */
00667     for (b = ngs->bp_table_idx[cf]; b < ngs->bpidx; b++) {
00668         xwdssid_t *rssid;
00669         int32 silscore;
00670 
00671         bp = ngs->bp_table + b;
00672         ngs->word_lat_idx[bp->wid] = NO_BP;
00673 
00674         if (bp->wid == ps_search_finish_wid(ngs))
00675             continue;
00676 
00677         /* DICT2PID location */
00678         /* Get the mapping from right context phone ID to index in the
00679          * right context table and the bscore_stack. */
00680         rcss = ngs->bscore_stack + bp->s_idx;
00681         if (bp->last2_phone == -1)
00682             rssid = NULL;
00683         else
00684             rssid = dict2pid_rssid(d2p, bp->last_phone, bp->last2_phone);
00685 
00686         /* Transition to all successor words. */
00687         for (i = 0; ngs->expand_word_list[i] >= 0; i++) {
00688             int32 n_used;
00689 
00690             w = ngs->expand_word_list[i];
00691 
00692             /* Get the exit score we recorded in save_bwd_ptr(), or
00693              * something approximating it. */
00694             if (rssid)
00695                 newscore = rcss[rssid->cimap[dict_first_phone(dict, w)]];
00696             else
00697                 newscore = bp->score;
00698             if (newscore == WORST_SCORE)
00699                 continue;
00700             /* FIXME: Floating point... */
00701             newscore += lwf
00702                 * (ngram_tg_score(ngs->lmset,
00703                                   dict_basewid(dict, w),
00704                                   bp->real_wid,
00705                                   bp->prev_real_wid,
00706                                   &n_used) >> SENSCR_SHIFT);
00707             newscore += pip;
00708 
00709             /* Enter the next word */
00710             if (newscore BETTER_THAN thresh) {
00711                 rhmm = (root_chan_t *) ngs->word_chan[w];
00712                 if ((hmm_frame(&rhmm->hmm) < cf)
00713                     || (newscore BETTER_THAN hmm_in_score(&rhmm->hmm))) {
00714                     hmm_enter(&rhmm->hmm, newscore, b, nf);
00715                     /* DICT2PID: This is where mpx ssids get introduced. */
00716                     /* Look up the ssid to use when entering this mpx triphone. */
00717                     hmm_mpx_ssid(&rhmm->hmm, 0) =
00718                         dict2pid_ldiph_lc(d2p, rhmm->ciphone, rhmm->ci2phone,
00719                                           dict_last_phone(dict, bp->wid));
00720                     assert(IS_S3SSID(hmm_mpx_ssid(&rhmm->hmm, 0)));
00721                     E_DEBUG(6,("ssid %d(%d,%d) = %d\n",
00722                                rhmm->ciphone, dict_last_phone(dict, bp->wid), rhmm->ci2phone,
00723                                hmm_mpx_ssid(&rhmm->hmm, 0)));
00724                     bitvec_set(ngs->word_active, w);
00725                 }
00726             }
00727         }
00728 
00729         /* Get the best exit into silence. */
00730         if (rssid)
00731             silscore = rcss[rssid->cimap[ps_search_acmod(ngs)->mdef->sil]];
00732         else
00733             silscore = bp->score;
00734         if (silscore BETTER_THAN best_silrc_score) {
00735             best_silrc_score = silscore;
00736             best_silrc_bp = b;
00737         }
00738     }
00739 
00740     /* Transition to <sil> */
00741     newscore = best_silrc_score + ngs->silpen + pip;
00742     if ((newscore BETTER_THAN thresh) && (newscore BETTER_THAN WORST_SCORE)) {
00743         w = ps_search_silence_wid(ngs);
00744         rhmm = (root_chan_t *) ngs->word_chan[w];
00745         if ((hmm_frame(&rhmm->hmm) < cf)
00746             || (newscore BETTER_THAN hmm_in_score(&rhmm->hmm))) {
00747             hmm_enter(&rhmm->hmm, newscore,
00748                       best_silrc_bp, nf);
00749             bitvec_set(ngs->word_active, w);
00750         }
00751     }
00752     /* Transition to noise words */
00753     newscore = best_silrc_score + ngs->fillpen + pip;
00754     if ((newscore BETTER_THAN thresh) && (newscore BETTER_THAN WORST_SCORE)) {
00755         for (w = ps_search_silence_wid(ngs) + 1; w < ps_search_n_words(ngs); w++) {
00756             rhmm = (root_chan_t *) ngs->word_chan[w];
00757             /* Noise words that aren't a single phone will have NULL here. */
00758             if (rhmm == NULL)
00759                 continue;
00760             if ((hmm_frame(&rhmm->hmm) < cf)
00761                 || (newscore BETTER_THAN hmm_in_score(&rhmm->hmm))) {
00762                 hmm_enter(&rhmm->hmm, newscore,
00763                           best_silrc_bp, nf);
00764                 bitvec_set(ngs->word_active, w);
00765             }
00766         }
00767     }
00768 
00769     /* Reset initial channels of words that have become inactive even after word trans. */
00770     i = ngs->n_active_word[cf & 0x1];
00771     awl = ngs->active_word_list[cf & 0x1];
00772     for (w = *(awl++); i > 0; --i, w = *(awl++)) {
00773         rhmm = (root_chan_t *) ngs->word_chan[w];
00774         if (hmm_frame(&rhmm->hmm) == cf) {
00775             hmm_clear_scores(&rhmm->hmm);
00776         }
00777     }
00778 }

00807 int
00808 ngram_fwdflat_search(ngram_search_t *ngs, int frame_idx)
00809 {
00810     int16 const *senscr;
00811     int32 nf, i, j;
00812     int32 *nawl;
00813 
00814     /* Activate our HMMs for the current frame if need be. */
00815     if (!ps_search_acmod(ngs)->compallsen)
00816         compute_fwdflat_sen_active(ngs, frame_idx);
00817 
00818     /* Compute GMM scores for the current frame. */
00819     senscr = acmod_score(ps_search_acmod(ngs), &frame_idx);
00820     ngs->st.n_senone_active_utt += ps_search_acmod(ngs)->n_senone_active;
00821 
00822     /* Mark backpointer table for current frame. */
00823     ngram_search_mark_bptable(ngs, frame_idx);
00824 
00825     /* If the best score is equal to or worse than WORST_SCORE,
00826      * recognition has failed, don't bother to keep trying. */
00827     if (ngs->best_score == WORST_SCORE || ngs->best_score WORSE_THAN WORST_SCORE)
00828         return 0;
00829     /* Renormalize if necessary */
00830     if (ngs->best_score + (2 * ngs->beam) WORSE_THAN WORST_SCORE) {
00831         E_INFO("Renormalizing Scores at frame %d, best score %d\n",
00832                frame_idx, ngs->best_score);
00833         fwdflat_renormalize_scores(ngs, frame_idx, ngs->best_score);
00834     }
00835 
00836     ngs->best_score = WORST_SCORE;
00837     hmm_context_set_senscore(ngs->hmmctx, senscr);
00838 
00839     /* Evaluate HMMs */
00840     fwdflat_eval_chan(ngs, frame_idx);
00841     /* Prune HMMs and do phone transitions. */
00842     fwdflat_prune_chan(ngs, frame_idx);
00843     /* Do word transitions. */
00844     fwdflat_word_transition(ngs, frame_idx);
00845 
00846     /* Create next active word list */
00847     nf = frame_idx + 1;
00848     nawl = ngs->active_word_list[nf & 0x1];
00849     for (i = 0, j = 0; ngs->fwdflat_wordlist[i] >= 0; i++) {
00850         if (bitvec_is_set(ngs->word_active, ngs->fwdflat_wordlist[i])) {
00851             *(nawl++) = ngs->fwdflat_wordlist[i];
00852             j++;
00853         }
00854     }
00855     for (i = ps_search_start_wid(ngs); i < ps_search_n_words(ngs); i++) {
00856         if (bitvec_is_set(ngs->word_active, i)) {
00857             *(nawl++) = i;
00858             j++;
00859         }
00860     }
00861     if (!ngs->fwdtree)
00862         ++ngs->n_frame;
00863     ngs->n_active_word[nf & 0x1] = j;
00864 
00865     /* Return the number of frames processed. */
00866     return 1;
00867 }
