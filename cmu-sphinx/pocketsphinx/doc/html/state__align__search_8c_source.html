00063 static int32
00064 evaluate_hmms(state_align_search_t *sas, int16 const *senscr, int frame_idx)
00065 {
00066     int32 bs = WORST_SCORE;
00067     int i;
00068 
00069     hmm_context_set_senscore(sas->hmmctx, senscr);
00070 
00071     for (i = 0; i < sas->n_phones; ++i) {
00072         hmm_t *hmm = sas->hmms + i;
00073         int32 score;
00074 
00075         if (hmm_frame(hmm) < frame_idx)
00076             continue;

00087 {
00088     int nf = frame_idx + 1;

00090

00100 static void
00101 phone_transition(state_align_search_t *sas, int frame_idx)
00102 {
00103     int nf = frame_idx + 1;
00104     int i;
00105 
00106     for (i = 0; i < sas->n_phones - 1; ++i) {
00107         hmm_t *hmm, *nhmm;
00108         int32 newphone_score;
00109 
00110         hmm = sas->hmms + i;
00111         if (hmm_frame(hmm) != nf)
00112             continue;
00113 
00114         newphone_score = hmm_out_score(hmm);
00115         /* Transition into next phone using the usual Viterbi rule. */
00116         nhmm = hmm + 1;
00117         if (hmm_frame(nhmm) < frame_idx
00118             || newphone_score BETTER_THAN hmm_in_score(nhmm)) {
00119             hmm_enter(nhmm, newphone_score, hmm_out_history(hmm), nf);
00120         }
00121     }
00122 }

00138 static void
00139 record_transitions(state_align_search_t *sas, int frame_idx)
00140 {
00141     uint16 *tokens;
00142     int i;
00143 
00144     /* Push another frame of tokens on the stack. */
00145     extend_tokenstack(sas, frame_idx);
00146     tokens = sas->tokens + frame_idx * sas->n_emit_state;
00147 
00148     /* Scan all active HMMs */
00149     for (i = 0; i < sas->n_phones; ++i) {
00150         hmm_t *hmm = sas->hmms + i;
00151         int j;
00152 
00153         if (hmm_frame(hmm) < frame_idx)
00154             continue;
00155         for (j = 0; j < sas->hmmctx->n_emit_state; ++j) {
00156             int state_idx = i * sas->hmmctx->n_emit_state + j;
00157             /* Record their backpointers on the token stack. */
00158             tokens[state_idx] = hmm_history(hmm, j);
00159             /* Update backpointer fields with state index. */
00160             hmm_history(hmm, j) = state_idx;
00161         }
00162     }
00163 }

00165 static int
00166 state_align_search_step(ps_search_t *search, int frame_idx)
00167 {
00168     state_align_search_t *sas = (state_align_search_t *)search;
00169     acmod_t *acmod = ps_search_acmod(search);
00170     int16 const *senscr;
00171     int i;
00172 
00173     /* Calculate senone scores. */
00174     for (i = 0; i < sas->n_phones; ++i)
00175         acmod_activate_hmm(acmod, sas->hmms + i);
00176     senscr = acmod_score(acmod, &frame_idx);
00177 
00178     /* Renormalize here if needed. */
00179     /* FIXME: Make sure to (unit-)test this!!! */
00180     if ((sas->best_score - 0x300000) WORSE_THAN WORST_SCORE) {
00181         E_INFO("Renormalizing Scores at frame %d, best score %d\n",
00182                frame_idx, sas->best_score);
00183         renormalize_hmms(sas, frame_idx, sas->best_score);
00184     }
00185     
00186     /* Viterbi step. */
00187     sas->best_score = evaluate_hmms(sas, senscr, frame_idx);
00188     prune_hmms(sas, frame_idx);
00189 
00190     /* Transition out of non-emitting states. */
00191     phone_transition(sas, frame_idx);
00192 
00193     /* Generate new tokens from best path results. */
00194     record_transitions(sas, frame_idx);
00195 
00196     /* Update frame counter */
00197     sas->frame = frame_idx;
00198 
00199     return 0;
00200 }

00202 static int
00203 state_align_search_finish(ps_search_t *search)
00204 {
00205     state_align_search_t *sas = (state_align_search_t *)search;
00206     hmm_t *final_phone = sas->hmms + sas->n_phones - 1;
00207     ps_alignment_iter_t *itor;
00208     ps_alignment_entry_t *ent;
00209     int next_state, next_start, state, frame;
00210 
00211     /* Best state exiting the last frame. */
00212     next_state = state = hmm_out_history(final_phone);
00213     if (state == 0xffff) {
00214         E_ERROR("Failed to reach final state in alignment\n");
00215         return -1;
00216     }
00217     itor = ps_alignment_states(sas->al);
00218     next_start = sas->frame + 1;
00219     for (frame = sas->frame - 1; frame >= 0; --frame) {
00220         state = sas->tokens[frame * sas->n_emit_state + state];
00221         /* State boundary, update alignment entry for next state. */
00222         if (state != next_state) {
00223             itor = ps_alignment_iter_goto(itor, next_state);
00224             assert(itor != NULL);
00225             ent = ps_alignment_iter_get(itor);
00226             ent->start = frame + 1;
00227             ent->duration = next_start - ent->start;
00228             E_DEBUG(1,("state %d start %d end %d\n", next_state,
00229                        ent->start, next_start));
00230             next_state = state;
00231             next_start = frame + 1;
00232         }
00233     }
00234     /* Update alignment entry for initial state. */
00235     itor = ps_alignment_iter_goto(itor, 0);
00236     assert(itor != NULL);
00237     ent = ps_alignment_iter_get(itor);
00238     ent->start = 0;
00239     ent->duration = next_start;
00240     E_DEBUG(1,("state %d start %d end %d\n", 0,
00241                ent->start, next_start));
00242     ps_alignment_iter_free(itor);
00243     ps_alignment_propagate(sas->al);
00244 
00245     return 0;
00246 }
