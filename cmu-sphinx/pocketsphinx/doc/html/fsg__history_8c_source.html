00138 void
00139 fsg_history_entry_add(fsg_history_t * h,
00140                       fsg_link_t * link,
00141                       int32 frame, int32 score, int32 pred,
00142                       int32 lc, fsg_pnode_ctxt_t rc)
00143 {
00144     fsg_hist_entry_t *entry, *new_entry;
00145     int32 s;
00146     gnode_t *gn, *prev_gn;
00147 
00148     /* Skip the optimization for the initial dummy entries; always enter them */
00149     if (frame < 0) {
00150         new_entry =
00151             (fsg_hist_entry_t *) ckd_calloc(1, sizeof(fsg_hist_entry_t));
00152         new_entry->fsglink = link;
00153         new_entry->frame = frame;
00154         new_entry->score = score;
00155         new_entry->pred = pred;
00156         new_entry->lc = lc;
00157         new_entry->rc = rc;
00158 
00159         blkarray_list_append(h->entries, (void *) new_entry);
00160         return;
00161     }
00162 
00163     s = fsg_link_to_state(link);
00164 
00165     /* Locate where this entry should be inserted in frame_entries[s][lc] */
00166     prev_gn = NULL;
00167     for (gn = h->frame_entries[s][lc]; gn; gn = gnode_next(gn)) {
00168         entry = (fsg_hist_entry_t *) gnode_ptr(gn);
00169 
00170         if (score BETTER_THAN entry->score)
00171             break;              /* Found where to insert new entry */
00172 
00173         /* Existing entry score not worse than new score */
00174         if (FSG_PNODE_CTXT_SUB(&rc, &(entry->rc)) == 0)
00175             return;             /* rc set reduced to 0; new entry can be ignored */
00176 
00177         prev_gn = gn;
00178     }
00179 
00180     /* Create new entry after prev_gn (if prev_gn is NULL, at head) */
00181     new_entry =
00182         (fsg_hist_entry_t *) ckd_calloc(1, sizeof(fsg_hist_entry_t));
00183     new_entry->fsglink = link;
00184     new_entry->frame = frame;
00185     new_entry->score = score;
00186     new_entry->pred = pred;
00187     new_entry->lc = lc;
00188     new_entry->rc = rc;         /* Note: rc set must be non-empty at this point */
00189 
00190     if (!prev_gn) {
00191         h->frame_entries[s][lc] = glist_add_ptr(h->frame_entries[s][lc],
00192                                                 (void *) new_entry);
00193         prev_gn = h->frame_entries[s][lc];
00194     }
00195     else
00196         prev_gn = glist_insert_ptr(prev_gn, (void *) new_entry);
00197 
00198     /*
00199      * Update the rc set of all the remaining entries in the list.  At this
00200      * point, gn is the entry, if any, immediately following new entry.
00201      */
00202     while (gn) {
00203         entry = (fsg_hist_entry_t *) gnode_ptr(gn);
00204 
00205         if (FSG_PNODE_CTXT_SUB(&(entry->rc), &rc) == 0) {
00206             /* rc set of entry reduced to 0; can prune this entry */
00207             ckd_free((void *) entry);
00208             gn = gnode_free(gn, prev_gn);
00209         }
00210         else {
00211             prev_gn = gn;
00212             gn = gnode_next(gn);
00213         }
00214     }
00215 }

00258     c = id - (r * blkarray_list_blksize(entries));
