00092     int32 n_ci;
00093     fsg_model_t *fsg;
00094     int32 silcipid;
00095     int32 len;
00096 
00097     silcipid = bin_mdef_silphone(lextree->mdef);
00098     assert(silcipid >= 0);
00099     n_ci = bin_mdef_n_ciphone(lextree->mdef);
00100 
00101     fsg = lextree->fsg;
00102     /*
00103      * lextree->lc[s] = set of left context CIphones for state s.  Similarly, rc[s]
00104      * for right context CIphones.
00105      */
00106     lextree->lc = ckd_calloc_2d(fsg->n_state, n_ci + 1, sizeof(**lextree->lc));
00107     lextree->rc = ckd_calloc_2d(fsg->n_state, n_ci + 1, sizeof(**lextree->rc));
00108     E_INFO("Allocated %d bytes (%d KiB) for left and right context phones\n",
00109            fsg->n_state * (n_ci + 1) * 2,
00110            fsg->n_state * (n_ci + 1) * 2 / 1024);
00111 
00112 
00113     for (s = 0; s < fsg->n_state; s++) {
00114         fsg_arciter_t *itor;
00115         for (itor = fsg_model_arcs(fsg, s); itor; itor = fsg_arciter_next(itor)) {
00116             fsg_link_t *l = fsg_arciter_get(itor);
00117             int32 dictwid; 
00119             if (fsg_link_wid(l) >= 0) {
00120                 dictwid = dict_wordid(lextree->dict,
00121                                       fsg_model_word_str(lextree->fsg, l->wid));
00122 
00123                 /*
00124                  * Add the first CIphone of l->wid to the rclist of state s, and
00125                  * the last CIphone to lclist of state d.
00126                  * (Filler phones are a pain to deal with.  There is no direct
00127                  * marking of a filler phone; but only filler words are supposed to
00128                  * use such phones, so we use that fact.  HACK!!  FRAGILE!!)
00129                  */
00130                 if (fsg_model_is_filler(fsg, fsg_link_wid(l))) {
00131                     /* Filler phone; use silence phone as context */
00132                     lextree->rc[fsg_link_from_state(l)][silcipid] = 1;
00133                     lextree->lc[fsg_link_to_state(l)][silcipid] = 1;
00134                 }
00135                 else {
00136                     len = dict_pronlen(lextree->dict, dictwid);
00137                     lextree->rc[fsg_link_from_state(l)][dict_pron(lextree->dict, dictwid, 0)] = 1;
00138                     lextree->lc[fsg_link_to_state(l)][dict_pron(lextree->dict, dictwid, len - 1)] = 1;
00139                 }
00140             }
00141         }
00142     }
00143 
00144     for (s = 0; s < fsg->n_state; s++) {
00145         /*
00146          * Add SIL phone to the lclist and rclist of each state.  Strictly
00147          * speaking, only needed at start and final states, respectively, but
00148          * all states considered since the user may change the start and final
00149          * states.  In any case, most applications would have a silence self
00150          * loop at each state, hence these would be needed anyway.
00151          */
00152         lextree->lc[s][silcipid] = 1;
00153         lextree->rc[s][silcipid] = 1;
00154     }
00155 
00156     /*
00157      * Propagate lc and rc lists past null transitions.  (Since FSG contains
00158      * null transitions closure, no need to worry about a chain of successive
00159      * null transitions.  Right??)
00160      *
00161      * This can't be joined with the previous loop because we first calculate 
00162      * contexts and only then we can propagate them.
00163      */
00164     for (s = 0; s < fsg->n_state; s++) {
00165         fsg_arciter_t *itor;
00166         for (itor = fsg_model_arcs(fsg, s); itor; itor = fsg_arciter_next(itor)) {
00167             fsg_link_t *l = fsg_arciter_get(itor);
00168             if (fsg_link_wid(l) < 0) {
00169 
00170                 /*
00171                  * lclist(d) |= lclist(s), because all the words ending up at s, can
00172                  * now also end at d, becoming the left context for words leaving d.
00173                  */
00174                 for (i = 0; i < n_ci; i++)
00175                     lextree->lc[fsg_link_to_state(l)][i] |= lextree->lc[fsg_link_from_state(l)][i];
00176                 /*
00177                  * Similarly, rclist(s) |= rclist(d), because all the words leaving d
00178                  * can equivalently leave s, becoming the right context for words
00179                  * ending up at s.
00180                  */
00181                 for (i = 0; i < n_ci; i++)
00182                     lextree->rc[fsg_link_from_state(l)][i] |= lextree->rc[fsg_link_to_state(l)][i];
00183             }
00184         }
00185     }
00186 
00187     /* Convert the bit-vector representation into a list */
00188     for (s = 0; s < fsg->n_state; s++) {
00189         j = 0;
00190         for (i = 0; i < n_ci; i++) {
00191             if (lextree->lc[s][i]) {
00192                 lextree->lc[s][j] = i;
00193                 j++;
00194             }
00195         }
00196         lextree->lc[s][j] = -1;     /* Terminate the list */
00197 
00198         j = 0;
00199         for (i = 0; i < n_ci; i++) {
00200             if (lextree->rc[s][i]) {
00201                 lextree->rc[s][j] = i;
00202                 j++;
00203             }
00204         }
00205         lextree->rc[s][j] = -1;     /* Terminate the list */
00206     }
00207 }

00698 static fsg_pnode_t *
00699 fsg_psubtree_init(fsg_lextree_t *lextree,
00700                   fsg_model_t * fsg, int32 from_state,
00701                   fsg_pnode_t ** alloc_head)
00702 {
00703     int32 dst;
00704     gnode_t *gn;
00705     fsg_link_t *fsglink;
00706     fsg_pnode_t *root;
00707     int32 n_ci, n_arc;
00708     fsg_glist_linklist_t *glist = NULL;
00709 
00710     root = NULL;
00711     assert(*alloc_head == NULL);
00712 
00713     n_ci = bin_mdef_n_ciphone(lextree->mdef);
00714     if (n_ci > (FSG_PNODE_CTXT_BVSZ * 32)) {
00715         E_FATAL
00716             ("#phones > %d; increase FSG_PNODE_CTXT_BVSZ and recompile\n",
00717              FSG_PNODE_CTXT_BVSZ * 32);
00718     }
00719     n_arc = 0;
00720     for (dst = 0; dst < fsg_model_n_state(fsg); dst++) {
00721         /* Add all links from from_state to dst */
00722         for (gn = fsg_model_trans(fsg, from_state, dst); gn;
00723              gn = gnode_next(gn)) {
00724             /* Add word emitted by this transition (fsglink) to lextree */
00725             fsglink = (fsg_link_t *) gnode_ptr(gn);
00726 
00727             assert(fsg_link_wid(fsglink) >= 0);     /* Cannot be a null trans */
00728 
00729             E_DEBUG(2,("Building lextree for arc from %d to %d: %s\n",
00730                        from_state, dst, fsg_model_word_str(fsg, fsg_link_wid(fsglink))));
00731             root = psubtree_add_trans(lextree, root, &glist, fsglink,
00732                                       lextree->lc[from_state],
00733                                       lextree->rc[dst],
00734                                       alloc_head);
00735             ++n_arc;
00736         }
00737     }
00738     E_DEBUG(2,("State %d has %d outgoing arcs\n", from_state, n_arc));
00739 
00740     fsg_glist_linklist_free(glist);
00741 
00742     return root;
00743 }

00820     }
00821
