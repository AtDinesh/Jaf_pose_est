00079 s3wid_t
00080 dict_add_word(dict_t * d, char const *word, s3cipid_t const * p, int32 np)
00081 {
00082     int32 len;
00083     dictword_t *wordp;
00084     s3wid_t newwid;
00085     char *wword;
00086 
00087     if (d->n_word >= d->max_words) {
00088         E_INFO("Reallocating to %d KiB for word entries\n",
00089                (d->max_words + S3DICT_INC_SZ) * sizeof(dictword_t) / 1024);
00090         d->word =
00091             (dictword_t *) ckd_realloc(d->word,
00092                                        (d->max_words +
00093                                         S3DICT_INC_SZ) * sizeof(dictword_t));
00094         d->max_words = d->max_words + S3DICT_INC_SZ;
00095     }
00096 
00097     wordp = d->word + d->n_word;
00098     wordp->word = (char *) ckd_salloc(word);    /* Freed in dict_free */
00099 
00100     /* Associate word string with d->n_word in hash table */
00101     if (hash_table_enter_int32(d->ht, wordp->word, d->n_word) != d->n_word) {
00102         ckd_free(wordp->word);
00103         wordp->word = NULL;
00104         return BAD_S3WID;
00105     }
00106 
00107     /* Fill in word entry, and set defaults */
00108     if (p && (np > 0)) {
00109         wordp->ciphone = (s3cipid_t *) ckd_malloc(np * sizeof(s3cipid_t));      /* Freed in dict_free */
00110         memcpy(wordp->ciphone, p, np * sizeof(s3cipid_t));
00111         wordp->pronlen = np;
00112     }
00113     else {
00114         wordp->ciphone = NULL;
00115         wordp->pronlen = 0;
00116     }
00117     wordp->alt = BAD_S3WID;
00118     wordp->basewid = d->n_word;
00119 
00120     /* Determine base/alt wids */
00121     wword = ckd_salloc(word);
00122     if ((len = dict_word2basestr(wword)) > 0) {
00123         int32 w;
00124 
00125         /* Truncated to a baseword string; find its ID */
00126         if (hash_table_lookup_int32(d->ht, wword, &w) < 0) {
00127             E_ERROR("Missing base word for: %s\n", word);
00128             ckd_free(wword);
00129             ckd_free(wordp->word);
00130             wordp->word = NULL;
00131             return BAD_S3WID;
00132         }
00133 
00134         /* Link into alt list */
00135         wordp->basewid = w;
00136         wordp->alt = d->word[w].alt;
00137         d->word[w].alt = d->n_word;
00138     }
00139     ckd_free(wword);
00140 
00141     newwid = d->n_word++;
00142 
00143     return newwid;
00144 }

00147 static int32
00148 dict_read(FILE * fp, dict_t * d)
00149 {
00150     lineiter_t *li;
00151     char **wptr;
00152     s3cipid_t *p;
00153     int32 lineno, nwd;
00154     s3wid_t w;
00155     int32 i, maxwd;
00156     size_t stralloc, phnalloc;
00157 
00158     maxwd = 512;
00159     p = (s3cipid_t *) ckd_calloc(maxwd + 4, sizeof(*p));
00160     wptr = (char **) ckd_calloc(maxwd, sizeof(char *)); /* Freed below */
00161 
00162     lineno = 0;
00163     stralloc = phnalloc = 0;
00164     for (li = lineiter_start(fp); li; li = lineiter_next(li)) {
00165         lineno++;
00166         if (0 == strncmp(li->buf, "##", 2)
00167             || 0 == strncmp(li->buf, ";;", 2))
00168             continue;
00169 
00170         if ((nwd = str2words(li->buf, wptr, maxwd)) < 0) {
00171             /* Increase size of p, wptr. */
00172             nwd = str2words(li->buf, NULL, 0);
00173             assert(nwd > maxwd); /* why else would it fail? */
00174             maxwd = nwd;
00175             p = (s3cipid_t *) ckd_realloc(p, (maxwd + 4) * sizeof(*p));
00176             wptr = (char **) ckd_realloc(wptr, maxwd * sizeof(*wptr));
00177         }
00178 
00179         if (nwd == 0)           /* Empty line */
00180             continue;
00181         /* wptr[0] is the word-string and wptr[1..nwd-1] the pronunciation sequence */
00182         if (nwd == 1) {
00183             E_ERROR("Line %d: No pronunciation for word '%s'; ignored\n",
00184                     lineno, wptr[0]);
00185             continue;
00186         }
00187 
00188 
00189         /* Convert pronunciation string to CI-phone-ids */
00190         for (i = 1; i < nwd; i++) {
00191             p[i - 1] = dict_ciphone_id(d, wptr[i]);
00192             if (NOT_S3CIPID(p[i - 1])) {
00193                 E_ERROR("Line %d: Phone '%s' is mising in the acoustic model; word '%s' ignored\n",
00194                         lineno, wptr[i], wptr[0]);
00195                 break;
00196             }
00197         }
00198 
00199         if (i == nwd) {         /* All CI-phones successfully converted to IDs */
00200             w = dict_add_word(d, wptr[0], p, nwd - 1);
00201             if (NOT_S3WID(w))
00202                 E_ERROR
00203                     ("Line %d: Failed to add the word '%s' (duplicate?); ignored\n",
00204                      lineno, wptr[0]);
00205             else {
00206                 stralloc += strlen(d->word[w].word);
00207                 phnalloc += d->word[w].pronlen * sizeof(s3cipid_t);
00208             }
00209         }
00210     }
00211     E_INFO("Allocated %d KiB for strings, %d KiB for phones\n",
00212            (int)stralloc / 1024, (int)phnalloc / 1024);
00213     ckd_free(p);
00214     ckd_free(wptr);
00215 
00216     return 0;
00217 }

00256     dict_t *d;

00259 
00260     if (config) {
00261         dictfile = cmd_ln_str_r(config, "-dict");
00262         fillerfile = cmd_ln_str_r(config, "-fdict");
00263     }
00264 
00265     /*
00266      * First obtain #words in dictionary (for hash table allocation).
00267      * Reason: The PC NT system doesn't like to grow memory gradually.  Better to allocate
00268      * all the required memory in one go.
00269      */
00270     fp = NULL;
00271     n = 0;
00272     if (dictfile) {
00273         if ((fp = fopen(dictfile, "r")) == NULL) {
00274             E_ERROR_SYSTEM("Failed to open dictionary file '%s' for reading", dictfile);
00275             return NULL;
00276         }
00277         for (li = lineiter_start(fp); li; li = lineiter_next(li)) {
00278             if (0 != strncmp(li->buf, "##", 2)
00279                 && 0 != strncmp(li->buf, ";;", 2))
00280                 n++;
00281         }
00282         rewind(fp);
00283     }
00284 
00285     fp2 = NULL;
00286     if (fillerfile) {
00287         if ((fp2 = fopen(fillerfile, "r")) == NULL) {
00288             E_ERROR_SYSTEM("Failed to open filler dictionary file '%s' for reading", fillerfile);
00289             fclose(fp);
00290             return NULL;
00291         }
00292         for (li = lineiter_start(fp2); li; li = lineiter_next(li)) {
00293             if (0 != strncmp(li->buf, "##", 2)
00294                 && 0 != strncmp(li->buf, ";;", 2))
00295                 n++;
00296         }
00297         rewind(fp2);
00298     }
00299 
00300     /*
00301      * Allocate dict entries.  HACK!!  Allow some extra entries for words not in file.
00302      * Also check for type size restrictions.
00303      */
00304     d = (dict_t *) ckd_calloc(1, sizeof(dict_t));       /* freed in dict_free() */
00305     d->refcnt = 1;
00306     d->max_words =
00307         (n + S3DICT_INC_SZ < MAX_S3WID) ? n + S3DICT_INC_SZ : MAX_S3WID;
00308     if (n >= MAX_S3WID) {
00309         E_ERROR("Number of words in dictionaries (%d) exceeds limit (%d)\n", n,
00310                 MAX_S3WID);
00311         fclose(fp);
00312         fclose(fp2);
00313         ckd_free(d);
00314         return NULL;
00315     }
00316 
00317     E_INFO("Allocating %d * %d bytes (%d KiB) for word entries\n",
00318            d->max_words, sizeof(dictword_t),
00319            d->max_words * sizeof(dictword_t) / 1024);
00320     d->word = (dictword_t *) ckd_calloc(d->max_words, sizeof(dictword_t));      /* freed in dict_free() */
00321     d->n_word = 0;
00322     if (mdef)
00323         d->mdef = bin_mdef_retain(mdef);
00324 
00325     /* Create new hash table for word strings; case-insensitive word strings */
00326     if (config && cmd_ln_exists_r(config, "-dictcase"))
00327         d->nocase = cmd_ln_boolean_r(config, "-dictcase");
00328     d->ht = hash_table_new(d->max_words, d->nocase);
00329 
00330     /* Digest main dictionary file */
00331     if (fp) {
00332         E_INFO("Reading main dictionary: %s\n", dictfile);
00333         dict_read(fp, d);
00334         fclose(fp);
00335         E_INFO("%d words read\n", d->n_word);
00336     }
00337 
00338     /* Now the filler dictionary file, if it exists */
00339     d->filler_start = d->n_word;
00340     if (fillerfile) {
00341         E_INFO("Reading filler dictionary: %s\n", fillerfile);
00342         dict_read(fp2, d);
00343         fclose(fp2);
00344         E_INFO("%d words read\n", d->n_word - d->filler_start);
00345     }
00346     if (mdef)
00347         sil = bin_mdef_silphone(mdef);
00348     else
00349         sil = 0;
00350     if (dict_wordid(d, S3_START_WORD) == BAD_S3WID) {
00351         dict_add_word(d, S3_START_WORD, &sil, 1);
00352     }
00353     if (dict_wordid(d, S3_FINISH_WORD) == BAD_S3WID) {
00354         dict_add_word(d, S3_FINISH_WORD, &sil, 1);
00355     }
00356     if (dict_wordid(d, S3_SILENCE_WORD) == BAD_S3WID) {
00357         dict_add_word(d, S3_SILENCE_WORD, &sil, 1);
00358     }
00359 
00360     d->filler_end = d->n_word - 1;
00361 
00362     /* Initialize distinguished word-ids */
00363     d->startwid = dict_wordid(d, S3_START_WORD);
00364     d->finishwid = dict_wordid(d, S3_FINISH_WORD);
00365     d->silwid = dict_wordid(d, S3_SILENCE_WORD);
00366 
00367     if ((d->filler_start > d->filler_end)
00368         || (!dict_filler_word(d, d->silwid))) {
00369         E_ERROR("Word '%s' must occur (only) in filler dictionary\n",
00370                 S3_SILENCE_WORD);
00371         dict_free(d);
00372         return NULL;
00373     }
00374 
00375     /* No check that alternative pronunciations for filler words are in filler range!! */
00376 
00377     return d;
00378 }

00426 {
00427     int32 i, len;
00428 
00429     len = strlen(word);
00430     if (word[len - 1] == ')') {
00431         for (i = len - 2; (i > 0) && (word[i] != '('); --i);
00432 
00433         if (i > 0) {
00434             /* The word is of the form <baseword>(...); strip from left-paren */
00435             word[i] = '\0';
