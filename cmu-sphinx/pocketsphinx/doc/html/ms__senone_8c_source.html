00056     int32 byteswap, chksum_present, n_gauden_present;
00057     uint32 chksum;
00058     int32 i;
00059     char eofchk;

00061     void *ptr;
00062     float32 v;
00063 
00064     E_INFO("Reading senone gauden-codebook map file: %s\n", file_name);
00065 
00066     if ((fp = fopen(file_name, "rb")) == NULL)
00067         E_FATAL_SYSTEM("Failed to open map file '%s' for reading", file_name);
00068 
00069     /* Read header, including argument-value info and 32-bit byteorder magic */
00070     if (bio_readhdr(fp, &argname, &argval, &byteswap) < 0)
00071         E_FATAL("Failed to read header from file '%s'\n", file_name);
00072 
00073     /* Parse argument-value list */
00074     chksum_present = 0;
00075     n_gauden_present = 0;
00076     for (i = 0; argname[i]; i++) {
00077         if (strcmp(argname[i], "version") == 0) {
00078             if (strcmp(argval[i], SPDEF_PARAM_VERSION) != 0) {
00079                 E_WARN("Version mismatch(%s): %s, expecting %s\n",
00080                        file_name, argval[i], SPDEF_PARAM_VERSION);
00081             }
00082 
00083             /* HACK!! Convert version# to float32 and take appropriate action */
00084             if (sscanf(argval[i], "%f", &v) != 1)
00085                 E_FATAL("%s: Bad version no. string: %s\n", file_name,
00086                         argval[i]);
00087 
00088             n_gauden_present = (v > 1.1) ? 1 : 0;
00089         }
00090         else if (strcmp(argname[i], "chksum0") == 0) {
00091             chksum_present = 1; /* Ignore the associated value */
00092         }
00093     }
00094     bio_hdrarg_free(argname, argval);
00095     argname = argval = NULL;
00096 
00097     chksum = 0;
00098 
00099     /* Read #gauden (if version matches) */
00100     if (n_gauden_present) {
00101         E_INFO("Reading number of codebooks from %s\n", file_name);
00102         if (bio_fread
00103             (&(s->n_gauden), sizeof(int32), 1, fp, byteswap, &chksum) != 1)
00104             E_FATAL("fread(%s) (#gauden) failed\n", file_name);
00105     }
00106 
00107     /* Read 1d array data */
00108     if (bio_fread_1d(&ptr, sizeof(uint32), &(s->n_sen), fp,
00109                      byteswap, &chksum) < 0) {
00110         E_FATAL("bio_fread_1d(%s) failed\n", file_name);
00111     }
00112     s->mgau = ptr;
00113     E_INFO("Mapping %d senones to %d codebooks\n", s->n_sen, s->n_gauden);
00114 
00115     /* Infer n_gauden if not present in this version */
00116     if (!n_gauden_present) {
00117         s->n_gauden = 1;
00118         for (i = 0; i < s->n_sen; i++)
00119             if (s->mgau[i] >= s->n_gauden)
00120                 s->n_gauden = s->mgau[i] + 1;
00121     }
00122 
00123     if (chksum_present)
00124         bio_verify_chksum(fp, byteswap, chksum);
00125 
00126     if (fread(&eofchk, 1, 1, fp) == 1)
00127         E_FATAL("More data than expected in %s: %d\n", file_name, eofchk);
00128 
00129     fclose(fp);
00130 
00131     E_INFO("Read %d->%d senone-codebook mappings\n", s->n_sen,
00132            s->n_gauden);
00133 
00134     return 1;
00135 }

00141     char eofchk;
00142     FILE *fp;
00143     int32 byteswap, chksum_present;
00144     uint32 chksum;
00145     float32 *pdf;
00146     int32 i, f, c, p, n_err;
00147     char **argname, **argval;
00148 
00149     E_INFO("Reading senone mixture weights: %s\n", file_name);
00150 
00151     if ((fp = fopen(file_name, "rb")) == NULL)
00152         E_FATAL_SYSTEM("Failed to open mixture weights file '%s' for reading", file_name);
00153 
00154     /* Read header, including argument-value info and 32-bit byteorder magic */
00155     if (bio_readhdr(fp, &argname, &argval, &byteswap) < 0)
00156         E_FATAL("Failed to read header from file '%s'\n", file_name);
00157 
00158     /* Parse argument-value list */
00159     chksum_present = 0;
00160     for (i = 0; argname[i]; i++) {
00161         if (strcmp(argname[i], "version") == 0) {
00162             if (strcmp(argval[i], MIXW_PARAM_VERSION) != 0)
00163                 E_WARN("Version mismatch(%s): %s, expecting %s\n",
00164                        file_name, argval[i], MIXW_PARAM_VERSION);
00165         }
00166         else if (strcmp(argname[i], "chksum0") == 0) {
00167             chksum_present = 1; /* Ignore the associated value */
00168         }
00169     }
00170     bio_hdrarg_free(argname, argval);
00171     argname = argval = NULL;
00172 
00173     chksum = 0;
00174 
00175     /* Read #senones, #features, #codewords, arraysize */
00176     if ((bio_fread(&(s->n_sen), sizeof(int32), 1, fp, byteswap, &chksum) !=
00177          1)
00178         ||
00179         (bio_fread(&(s->n_feat), sizeof(int32), 1, fp, byteswap, &chksum)
00180          != 1)
00181         || (bio_fread(&(s->n_cw), sizeof(int32), 1, fp, byteswap, &chksum)
00182             != 1)
00183         || (bio_fread(&i, sizeof(int32), 1, fp, byteswap, &chksum) != 1)) {
00184         E_FATAL("bio_fread(%s) (arraysize) failed\n", file_name);
00185     }
00186     if (i != s->n_sen * s->n_feat * s->n_cw) {
00187         E_FATAL
00188             ("%s: #float32s(%d) doesn't match dimensions: %d x %d x %d\n",
00189              file_name, i, s->n_sen, s->n_feat, s->n_cw);
00190     }
00191 
00192     /*
00193      * Compute #LSB bits to be dropped to represent mixwfloor with 8 bits.
00194      * All PDF values will be truncated (in the LSB positions) by these many bits.
00195      */
00196     if ((s->mixwfloor <= 0.0) || (s->mixwfloor >= 1.0))
00197         E_FATAL("mixwfloor (%e) not in range (0, 1)\n", s->mixwfloor);
00198 
00199     /* Use a fixed shift for compatibility with everything else. */
00200     E_INFO("Truncating senone logs3(pdf) values by %d bits\n", SENSCR_SHIFT);
00201 
00202     /*
00203      * Allocate memory for senone PDF data.  Organize normally or transposed depending on
00204      * s->n_gauden.
00205      */
00206     if (s->n_gauden > 1) {
00207         E_INFO("Not transposing mixture weights in memory\n");
00208         s->pdf =
00209             (senprob_t ***) ckd_calloc_3d(s->n_sen, s->n_feat, s->n_cw,
00210                                           sizeof(senprob_t));
00211     }
00212     else {
00213         E_INFO("Transposing mixture weights in memory\n");
00214         s->pdf =
00215             (senprob_t ***) ckd_calloc_3d(s->n_feat, s->n_cw, s->n_sen,
00216                                           sizeof(senprob_t));
00217     }
00218 
00219     /* Temporary structure to read in floats */
00220     pdf = (float32 *) ckd_calloc(s->n_cw, sizeof(float32));
00221 
00222     /* Read senone probs data, normalize, floor, convert to logs3, truncate to 8 bits */
00223     n_err = 0;
00224     for (i = 0; i < s->n_sen; i++) {
00225         for (f = 0; f < s->n_feat; f++) {
00226             if (bio_fread
00227                 ((void *) pdf, sizeof(float32), s->n_cw, fp, byteswap,
00228                  &chksum)
00229                 != s->n_cw) {
00230                 E_FATAL("bio_fread(%s) (arraydata) failed\n", file_name);
00231             }
00232 
00233             /* Normalize and floor */
00234             if (vector_sum_norm(pdf, s->n_cw) <= 0.0)
00235                 n_err++;
00236             vector_floor(pdf, s->n_cw, s->mixwfloor);
00237             vector_sum_norm(pdf, s->n_cw);
00238 
00239             /* Convert to logs3, truncate to 8 bits, and store in s->pdf */
00240             for (c = 0; c < s->n_cw; c++) {
00241                 p = -(logmath_log(lmath, pdf[c]));
00242                 p += (1 << (SENSCR_SHIFT - 1)) - 1; /* Rounding before truncation */
00243 
00244                 if (s->n_gauden > 1)
00245                     s->pdf[i][f][c] =
00246                         (p < (255 << SENSCR_SHIFT)) ? (p >> SENSCR_SHIFT) : 255;
00247                 else
00248                     s->pdf[f][c][i] =
00249                         (p < (255 << SENSCR_SHIFT)) ? (p >> SENSCR_SHIFT) : 255;
00250             }
00251         }
00252     }
00253     if (n_err > 0)
00254         E_WARN("Weight normalization failed for %d mixture weights components\n", n_err);
00255 
00256     ckd_free(pdf);
00257 
00258     if (chksum_present)
00259         bio_verify_chksum(fp, byteswap, chksum);
00260 
00261     if (fread(&eofchk, 1, 1, fp) == 1)
00262         E_FATAL("More data than expected in %s\n", file_name);
00263 
00264     fclose(fp);
00265 
00266     E_INFO
00267         ("Read mixture weights for %d senones: %d features x %d codewords\n",
00268          s->n_sen, s->n_feat, s->n_cw);
00269 
00270     return 1;
00271 }

00275 senone_init(gauden_t *g, char const *mixwfile, char const *sen2mgau_map_file,

00290             senone_mgau_map_read(s, sen2mgau_map_file);
00291             n = s->n_sen;

00294     else {
00295         if (s->n_gauden == 1)
00296             sen2mgau_map_file = ".semi.";

00309     }
00310     else if (strcmp(sen2mgau_map_file, ".ptm.") == 0) {
00311         /* All-to-ciphone-id senones-codebook mapping */
00312         E_INFO("Mapping senones to context-independent phone codebooks\n");
00313         s->mgau = (uint32 *) ckd_calloc(s->n_sen, sizeof(*s->mgau));
00314         for (i = 0; i < s->n_sen; i++)
00315             s->mgau[i] = bin_mdef_sen2cimap(mdef, i);
00316     }
00317     else if (strcmp(sen2mgau_map_file, ".cont.") == 0
00318              || strcmp(sen2mgau_map_file, ".s3cont.") == 0) {
00319         /* 1-to-1 senone-codebook mapping */
00320         E_INFO("Mapping senones to individual codebooks\n");
00321         if (s->n_sen <= 1)
00322             E_FATAL("#senone=%d; must be >1\n", s->n_sen);
00323 
00324         s->mgau = (uint32 *) ckd_calloc(s->n_sen, sizeof(*s->mgau));
00325         for (i = 0; i < s->n_sen; i++)
00326             s->mgau[i] = i;
00327         /* Not sure why this is here, it probably does nothing. */
00328         s->n_gauden = s->n_sen;
00329     }
00330     else {
00331         if (s->n_sen != n)
00332             E_FATAL("#senones inconsistent: %d in %s; %d in %s\n",
00333                     n, sen2mgau_map_file, s->n_sen, mixwfile);
00334     }
00335 
00336     s->featscr = NULL;
00337     return s;
00338 }

00356 /*
00357  * Compute senone score for one senone.
00358  * NOTE:  Remember that senone PDF tables contain SCALED, NEGATED logs3 values.
00359  * NOTE:  Remember also that PDF data may be transposed or not depending on s->n_gauden.
00360  */
00361 int32
00362 senone_eval(senone_t * s, int id, gauden_dist_t ** dist, int32 n_top)
00363 {
00364     int32 scr;                  /* total senone score */
00365     int32 fden;                 /* Gaussian density */
00366     int32 fscr;                 /* senone score for one feature */
00367     int32 fwscr;                /* senone score for one feature, one codeword */
00368     int32 f, t;
00369     gauden_dist_t *fdist;
00370 
00371     assert((id >= 0) && (id < s->n_sen));
00372     assert((n_top > 0) && (n_top <= s->n_cw));
00373 
00374     scr = 0;
00375 
00376     for (f = 0; f < s->n_feat; f++) {
00377         int top;
00378         fdist = dist[f];
00379 
00380         /* Top codeword for feature f */
00381         top = fden = ((int32)fdist[0].dist + ((1<<SENSCR_SHIFT) - 1)) >> SENSCR_SHIFT;
00382         fscr = (s->n_gauden > 1)
00383             ? (fden + -s->pdf[id][f][fdist[0].id])  /* untransposed */
00384             : (fden + -s->pdf[f][fdist[0].id][id]); /* transposed */
00385         E_DEBUG(1, ("fden[%d][%d] l+= %d + %d = %d\n",
00386                     id, f, -(fscr - fden), -(fden-top), -(fscr-top)));
00387         /* Remaining of n_top codewords for feature f */
00388         for (t = 1; t < n_top; t++) {
00389             fden = ((int32)fdist[t].dist + ((1<<SENSCR_SHIFT) - 1)) >> SENSCR_SHIFT;
00390             fwscr = (s->n_gauden > 1) ?
00391                 (fden + -s->pdf[id][f][fdist[t].id]) :
00392                 (fden + -s->pdf[f][fdist[t].id][id]);
00393             fscr = logmath_add(s->lmath, fscr, fwscr);
00394             E_DEBUG(1, ("fden[%d][%d] l+= %d + %d = %d\n",
00395                         id, f, -(fwscr - fden), -(fden-top), -(fscr-top)));
00396         }
00397         /* Senone scores are also scaled, negated logs3 values.  Hence
00398          * we have to negate the stuff we calculated above. */
00399         scr -= fscr;
00400     }
00401     /* Downscale scores. */
00402     scr /= s->aw;
00403 
00404     /* Avoid overflowing int16 */
00405     if (scr > 32767)
00406       scr = 32767;
00407     if (scr < -32768)
00408       scr = -32768;
00409     return scr;
00410 }
