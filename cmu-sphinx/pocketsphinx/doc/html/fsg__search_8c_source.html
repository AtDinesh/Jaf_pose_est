00092                 acmod_t *acmod,
00093                 dict_t *dict,
00094                 dict2pid_t *d2p)
00095 {
00096     fsg_search_t *fsgs;
00097     char const *path;
00098 
00099     fsgs = ckd_calloc(1, sizeof(*fsgs));
00100     ps_search_init(ps_search_base(fsgs), &fsg_funcs, config, acmod, dict, d2p);
00101 
00102     /* Initialize HMM context. */
00103     fsgs->hmmctx = hmm_context_init(bin_mdef_n_emit_state(acmod->mdef),
00104                                     acmod->tmat->tp, NULL, acmod->mdef->sseq);
00105     if (fsgs->hmmctx == NULL) {
00106         ps_search_free(ps_search_base(fsgs));
00107         return NULL;
00108     }
00109 
00110     /* Intialize the search history object */
00111     fsgs->history = fsg_history_init(NULL, dict);
00112     fsgs->frame = -1;
00113 
00114     /* Initialize FSG table. */
00115     fsgs->fsgs = hash_table_new(5, HASH_CASE_YES);
00116 
00117     /* Get search pruning parameters */
00118     fsgs->beam_factor = 1.0f;
00119     fsgs->beam = fsgs->beam_orig
00120         = (int32) logmath_log(acmod->lmath, cmd_ln_float64_r(config, "-beam"))
00121         >> SENSCR_SHIFT;
00122     fsgs->pbeam = fsgs->pbeam_orig
00123         = (int32) logmath_log(acmod->lmath, cmd_ln_float64_r(config, "-pbeam"))
00124         >> SENSCR_SHIFT;
00125     fsgs->wbeam = fsgs->wbeam_orig
00126         = (int32) logmath_log(acmod->lmath, cmd_ln_float64_r(config, "-wbeam"))
00127         >> SENSCR_SHIFT;
00128 
00129     /* LM related weights/penalties */
00130     fsgs->lw = cmd_ln_float32_r(config, "-lw");
00131     fsgs->pip = (int32) (logmath_log(acmod->lmath, cmd_ln_float32_r(config, "-pip"))
00132                            * fsgs->lw)
00133         >> SENSCR_SHIFT;
00134     fsgs->wip = (int32) (logmath_log(acmod->lmath, cmd_ln_float32_r(config, "-wip"))
00135                            * fsgs->lw)
00136         >> SENSCR_SHIFT;
00137 
00138     /* Best path search (and confidence annotation)? */
00139     if (cmd_ln_boolean_r(config, "-bestpath"))
00140         fsgs->bestpath = TRUE;
00141 
00142     /* Acoustic score scale for posterior probabilities. */
00143     fsgs->ascale = 1.0 / cmd_ln_float32_r(config, "-ascale");
00144 
00145     E_INFO("FSG(beam: %d, pbeam: %d, wbeam: %d; wip: %d, pip: %d)\n",
00146            fsgs->beam_orig, fsgs->pbeam_orig, fsgs->wbeam_orig,
00147            fsgs->wip, fsgs->pip);
00148 
00149     /* Load an FSG if one was specified in config */
00150     if ((path = cmd_ln_str_r(config, "-fsg"))) {
00151         fsg_model_t *fsg;
00152 
00153         if ((fsg = fsg_model_readfile(path, acmod->lmath, fsgs->lw)) == NULL)
00154             goto error_out;
00155         if (fsg_set_add(fsgs, fsg_model_name(fsg), fsg) != fsg) {
00156             fsg_model_free(fsg);
00157             goto error_out;
00158         }
00159         if (fsg_set_select(fsgs, fsg_model_name(fsg)) == NULL)
00160             goto error_out;
00161         if (fsg_search_reinit(ps_search_base(fsgs),
00162                               ps_search_dict(fsgs),
00163                               ps_search_dict2pid(fsgs)) < 0)
00164             goto error_out;
00165     }
00166     /* Or load a JSGF grammar */
00167     else if ((path = cmd_ln_str_r(config, "-jsgf"))) {
00168         fsg_model_t *fsg;
00169         jsgf_rule_t *rule;
00170         char const *toprule;
00171 
00172         if ((fsgs->jsgf = jsgf_parse_file(path, NULL)) == NULL)
00173             goto error_out;
00174 
00175         rule = NULL;
00176         /* Take the -toprule if specified. */
00177         if ((toprule = cmd_ln_str_r(config, "-toprule"))) {
00178             char *anglerule;
00179             anglerule = string_join("<", toprule, ">", NULL);
00180             rule = jsgf_get_rule(fsgs->jsgf, anglerule);
00181             ckd_free(anglerule);
00182             if (rule == NULL) {
00183                 E_ERROR("Start rule %s not found\n", toprule);
00184                 goto error_out;
00185             }
00186         }
00187         /* Otherwise, take the first public rule. */
00188         else {
00189             jsgf_rule_iter_t *itor;
00190 
00191             for (itor = jsgf_rule_iter(fsgs->jsgf); itor;
00192                  itor = jsgf_rule_iter_next(itor)) {
00193                 rule = jsgf_rule_iter_rule(itor);
00194                 if (jsgf_rule_public(rule)) {
00195                     jsgf_rule_iter_free(itor);
00196                     break;
00197                 }
00198             }
00199             if (rule == NULL) {
00200                 E_ERROR("No public rules found in %s\n", path);
00201                 goto error_out;
00202             }
00203         }
00204         fsg = jsgf_build_fsg(fsgs->jsgf, rule, acmod->lmath, fsgs->lw);
00205         if (fsg_set_add(fsgs, fsg_model_name(fsg), fsg) != fsg) {
00206             fsg_model_free(fsg);
00207             goto error_out;
00208         }
00209         if (fsg_set_select(fsgs, fsg_model_name(fsg)) == NULL)
00210             goto error_out;
00211         if (fsg_search_reinit(ps_search_base(fsgs),
00212                               ps_search_dict(fsgs),
00213                               ps_search_dict2pid(fsgs)) < 0)
00214             goto error_out;
00215     }
00216 
00217     return ps_search_base(fsgs);
00218 
00219 error_out:
00220     fsg_search_free(ps_search_base(fsgs));
00221     return NULL;
00222 }

00282 static int
00283 fsg_search_add_silences(fsg_search_t *fsgs, fsg_model_t *fsg)
00284 {
00285     dict_t *dict;
00286     int32 wid;
00287     int n_sil;
00288 
00289     dict = ps_search_dict(fsgs);
00290     /*
00291      * NOTE: Unlike N-Gram search, we do not use explicit start and
00292      * end symbols.  This is because the start and end nodes are
00293      * defined in the grammar.  We do add silence/filler self-loops to
00294      * all states in order to allow for silence between words and at
00295      * the beginning and end of utterances.
00296      *
00297      * This has some implications for word graph generation, namely,
00298      * that there can (and usually will) be multiple start and end
00299      * states in the word graph.  We therefore do add explicit start
00300      * and end nodes to the graph.
00301      */
00302     /* Add silence self-loops to all states. */
00303     fsg_model_add_silence(fsg, "<sil>", -1,
00304                           cmd_ln_float32_r(ps_search_config(fsgs), "-silprob"));
00305     n_sil = 0;
00306     /* Add self-loops for all other fillers. */
00307     for (wid = dict_filler_start(dict); wid < dict_filler_end(dict); ++wid) {
00308         char const *word = dict_wordstr(dict, wid);
00309         if (wid == dict_startwid(dict) || wid == dict_finishwid(dict))
00310             continue;
00311         fsg_model_add_silence(fsg, word, -1,
00312                               cmd_ln_float32_r(ps_search_config(fsgs), "-fillprob"));
00313         ++n_sil;
00314     }
00315 
00316     return n_sil;
00317 }

00510 /*
00511  * Evaluate all the active HMMs.
00512  * (Executed once per frame.)
00513  */
00514 static void
00515 fsg_search_hmm_eval(fsg_search_t *fsgs)
00516 {
00517     gnode_t *gn;
00518     fsg_pnode_t *pnode;
00519     hmm_t *hmm;
00520     int32 bestscore;
00521     int32 n, maxhmmpf;
00522 
00523     bestscore = WORST_SCORE;
00524 
00525     if (!fsgs->pnode_active) {
00526         E_ERROR("Frame %d: No active HMM!!\n", fsgs->frame);
00527         return;
00528     }
00529 
00530     for (n = 0, gn = fsgs->pnode_active; gn; gn = gnode_next(gn), n++) {
00531         int32 score;
00532 
00533         pnode = (fsg_pnode_t *) gnode_ptr(gn);
00534         hmm = fsg_pnode_hmmptr(pnode);
00535         assert(hmm_frame(hmm) == fsgs->frame);
00536 
00537 #if __FSG_DBG__
00538         E_INFO("pnode(%08x) active @frm %5d\n", (int32) pnode,
00539                fsgs->frame);
00540         hmm_dump(hmm, stdout);
00541 #endif
00542         score = hmm_vit_eval(hmm);
00543 #if __FSG_DBG_CHAN__
00544         E_INFO("pnode(%08x) after eval @frm %5d\n",
00545                (int32) pnode, fsgs->frame);
00546         hmm_dump(hmm, stdout);
00547 #endif
00548 
00549         if (score BETTER_THAN bestscore)
00550             bestscore = score;
00551     }
00552 
00553 #if __FSG_DBG__
00554     E_INFO("[%5d] %6d HMM; bestscr: %11d\n", fsgs->frame, n, bestscore);
00555 #endif
00556     fsgs->n_hmm_eval += n;
00557 
00558     /* Adjust beams if #active HMMs larger than absolute threshold */
00559     maxhmmpf = cmd_ln_int32_r(ps_search_config(fsgs), "-maxhmmpf");
00560     if (maxhmmpf != -1 && n > maxhmmpf) {
00561         /*
00562          * Too many HMMs active; reduce the beam factor applied to the default
00563          * beams, but not if the factor is already at a floor (0.1).
00564          */
00565         if (fsgs->beam_factor > 0.1) {        /* Hack!!  Hardwired constant 0.1 */
00566             fsgs->beam_factor *= 0.9f;        /* Hack!!  Hardwired constant 0.9 */
00567             fsgs->beam =
00568                 (int32) (fsgs->beam_orig * fsgs->beam_factor);
00569             fsgs->pbeam =
00570                 (int32) (fsgs->pbeam_orig * fsgs->beam_factor);
00571             fsgs->wbeam =
00572                 (int32) (fsgs->wbeam_orig * fsgs->beam_factor);
00573         }
00574     }
00575     else {
00576         fsgs->beam_factor = 1.0f;
00577         fsgs->beam = fsgs->beam_orig;
00578         fsgs->pbeam = fsgs->pbeam_orig;
00579         fsgs->wbeam = fsgs->wbeam_orig;
00580     }
00581 
00582     if (n > fsg_lextree_n_pnode(fsgs->lextree))
00583         E_FATAL("PANIC! Frame %d: #HMM evaluated(%d) > #PNodes(%d)\n",
00584                 fsgs->frame, n, fsg_lextree_n_pnode(fsgs->lextree));
00585 
00586     fsgs->bestscore = bestscore;

00590 static void
00591 fsg_search_pnode_trans(fsg_search_t *fsgs, fsg_pnode_t * pnode)
00592 {
00593     fsg_pnode_t *child;
00594     hmm_t *hmm;
00595     int32 newscore, thresh, nf;
00596 
00597     assert(pnode);
00598     assert(!fsg_pnode_leaf(pnode));
00599 
00600     nf = fsgs->frame + 1;
00601     thresh = fsgs->bestscore + fsgs->beam;
00602 
00603     hmm = fsg_pnode_hmmptr(pnode);
00604 
00605     for (child = fsg_pnode_succ(pnode);
00606          child; child = fsg_pnode_sibling(child)) {
00607         newscore = hmm_out_score(hmm) + child->logs2prob;
00608 
00609         if ((newscore BETTER_THAN thresh)
00610             && (newscore BETTER_THAN hmm_in_score(&child->hmm))) {
00611             /* Incoming score > pruning threshold and > target's existing score */
00612             if (hmm_frame(&child->hmm) < nf) {
00613                 /* Child node not yet activated; do so */
00614                 fsgs->pnode_active_next =
00615                     glist_add_ptr(fsgs->pnode_active_next,
00616                                   (void *) child);
00617             }
00618 
00619             hmm_enter(&child->hmm, newscore, hmm_out_history(hmm), nf);
00620         }
00621     }

00682 /*
00683  * (Beam) prune the just evaluated HMMs, determine which ones remain
00684  * active, which ones transition to successors, which ones exit and
00685  * terminate in their respective destination FSM states.
00686  * (Executed once per frame.)
00687  */
00688 static void
00689 fsg_search_hmm_prune_prop(fsg_search_t *fsgs)
00690 {
00691     gnode_t *gn;
00692     fsg_pnode_t *pnode;
00693     hmm_t *hmm;
00694     int32 thresh, word_thresh, phone_thresh;
00695 
00696     assert(fsgs->pnode_active_next == NULL);
00697 
00698     thresh = fsgs->bestscore + fsgs->beam;
00699     phone_thresh = fsgs->bestscore + fsgs->pbeam;
00700     word_thresh = fsgs->bestscore + fsgs->wbeam;
00701 
00702     for (gn = fsgs->pnode_active; gn; gn = gnode_next(gn)) {
00703         pnode = (fsg_pnode_t *) gnode_ptr(gn);
00704         hmm = fsg_pnode_hmmptr(pnode);
00705 
00706         if (hmm_bestscore(hmm) >= thresh) {
00707             /* Keep this HMM active in the next frame */
00708             if (hmm_frame(hmm) == fsgs->frame) {
00709                 hmm_frame(hmm) = fsgs->frame + 1;
00710                 fsgs->pnode_active_next =
00711                     glist_add_ptr(fsgs->pnode_active_next,
00712                                   (void *) pnode);
00713             }
00714             else {
00715                 assert(hmm_frame(hmm) == fsgs->frame + 1);
00716             }
00717 
00718             if (!fsg_pnode_leaf(pnode)) {
00719                 if (hmm_out_score(hmm) >= phone_thresh) {
00720                     /* Transition out of this phone into its children */
00721                     fsg_search_pnode_trans(fsgs, pnode);
00722                 }
00723             }
00724             else {
00725                 if (hmm_out_score(hmm) >= word_thresh) {
00726                     /* Transition out of leaf node into destination FSG state */
00727                     fsg_search_pnode_exit(fsgs, pnode);
00728                 }
00729             }
00730         }
00731     }
00732 }

00736  * Propagate newly created history entries through null transitions.
00737  */
00738 static void
00739 fsg_search_null_prop(fsg_search_t *fsgs)
00740 {
00741     int32 bpidx, n_entries, thresh, newscore;
00742     fsg_hist_entry_t *hist_entry;
00743     fsg_link_t *l;
00744     int32 s;
00745     fsg_model_t *fsg;
00746 
00747     fsg = fsgs->fsg;
00748     thresh = fsgs->bestscore + fsgs->wbeam; /* Which beam really?? */
00749 
00750     n_entries = fsg_history_n_entries(fsgs->history);
00751 
00752     for (bpidx = fsgs->bpidx_start; bpidx < n_entries; bpidx++) {
00753         fsg_arciter_t *itor;
00754         hist_entry = fsg_history_entry_get(fsgs->history, bpidx);
00755 
00756         l = fsg_hist_entry_fsglink(hist_entry);
00757 
00758         /* Destination FSG state for history entry */
00759         s = l ? fsg_link_to_state(l) : fsg_model_start_state(fsg);
00760 
00761         /*
00762          * Check null transitions from d to all other states.  (Only need to
00763          * propagate one step, since FSG contains transitive closure of null
00764          * transitions.)
00765          */
00766         /* Add all links from from_state to dst */
00767         for (itor = fsg_model_arcs(fsg, s); itor;

00770

00774             newscore =
00775                 fsg_hist_entry_score(hist_entry) +
00776                 (fsg_link_logs2prob(l) >> SENSCR_SHIFT);
00777 
00778             if (newscore >= thresh) {
00779                 fsg_history_entry_add(fsgs->history, l,
00780                                       fsg_hist_entry_frame(hist_entry),
00781                                       newscore,
00782                                       bpidx,
00783                                       fsg_hist_entry_lc(hist_entry),
00784                                       fsg_hist_entry_rc(hist_entry));
00785             }
00786         }
00787     }

00791 /*
00792  * Perform cross-word transitions; propagate each history entry created in this
00793  * frame to lextree roots attached to the target FSG state for that entry.
00794  */
00795 static void
00796 fsg_search_word_trans(fsg_search_t *fsgs)
00797 {
00798     int32 bpidx, n_entries;
00799     fsg_hist_entry_t *hist_entry;
00800     fsg_link_t *l;
00801     int32 score, newscore, thresh, nf, d;
00802     fsg_pnode_t *root;
00803     int32 lc, rc;
00804 
00805     n_entries = fsg_history_n_entries(fsgs->history);
00806 
00807     thresh = fsgs->bestscore + fsgs->beam;
00808     nf = fsgs->frame + 1;
00809 
00810     for (bpidx = fsgs->bpidx_start; bpidx < n_entries; bpidx++) {
00811         hist_entry = fsg_history_entry_get(fsgs->history, bpidx);
00812         assert(hist_entry);
00813         score = fsg_hist_entry_score(hist_entry);
00814         assert(fsgs->frame == fsg_hist_entry_frame(hist_entry));
00815 
00816         l = fsg_hist_entry_fsglink(hist_entry);
00817 
00818         /* Destination state for hist_entry */
00819         d = l ? fsg_link_to_state(l) : fsg_model_start_state(fsgs->
00820                                                                 fsg);
00821 
00822         lc = fsg_hist_entry_lc(hist_entry);
00823 
00824         /* Transition to all root nodes attached to state d */
00825         for (root = fsg_lextree_root(fsgs->lextree, d);
00826              root; root = root->sibling) {
00827             rc = root->ci_ext;
00828 
00829             if ((root->ctxt.bv[lc >> 5] & (1 << (lc & 0x001f))) &&
00830                 (hist_entry->rc.bv[rc >> 5] & (1 << (rc & 0x001f)))) {
00831                 /*
00832                  * Last CIphone of history entry is in left-context list supported by
00833                  * target root node, and
00834                  * first CIphone of target root node is in right context list supported
00835                  * by history entry;
00836                  * So the transition can go ahead (if new score is good enough).
00837                  */
00838                 newscore = score + root->logs2prob;
00839 
00840                 if ((newscore BETTER_THAN thresh)
00841                     && (newscore BETTER_THAN hmm_in_score(&root->hmm))) {
00842                     if (hmm_frame(&root->hmm) < nf) {
00843                         /* Newly activated node; add to active list */
00844                         fsgs->pnode_active_next =
00845                             glist_add_ptr(fsgs->pnode_active_next,
00846                                           (void *) root);
00847 #if __FSG_DBG__
00848                         E_INFO
00849                             ("[%5d] WordTrans bpidx[%d] -> pnode[%08x] (activated)\n",
00850                              fsgs->frame, bpidx, (int32) root);
00851 #endif
00852                     }
00853                     else {
00854 #if __FSG_DBG__
00855                         E_INFO
00856                             ("[%5d] WordTrans bpidx[%d] -> pnode[%08x]\n",
00857                              fsgs->frame, bpidx, (int32) root);
00858 #endif
00859                     }
00860 
00861                     hmm_enter(&root->hmm, newscore, bpidx, nf);
00862                 }
00863             }
00864         }
00865     }
00866 }

00869 int
00870 fsg_search_step(ps_search_t *search, int frame_idx)
00871 {
00872     fsg_search_t *fsgs = (fsg_search_t *)search;
00873     int16 const *senscr;
00874     acmod_t *acmod = search->acmod;
00875     gnode_t *gn;
00876     fsg_pnode_t *pnode;
00877     hmm_t *hmm;
00878 
00879     /* Activate our HMMs for the current frame if need be. */
00880     if (!acmod->compallsen)
00881         fsg_search_sen_active(fsgs);
00882     /* Compute GMM scores for the current frame. */
00883     senscr = acmod_score(acmod, &frame_idx);
00884     fsgs->n_sen_eval += acmod->n_senone_active;
00885     hmm_context_set_senscore(fsgs->hmmctx, senscr);
00886 
00887     /* Mark backpointer table for current frame. */
00888     fsgs->bpidx_start = fsg_history_n_entries(fsgs->history);
00889 
00890     /* Evaluate all active pnodes (HMMs) */
00891     fsg_search_hmm_eval(fsgs);
00892 
00893     /*
00894      * Prune and propagate the HMMs evaluated; create history entries for
00895      * word exits.  The words exits are tentative, and may be pruned; make
00896      * the survivors permanent via fsg_history_end_frame().
00897      */
00898     fsg_search_hmm_prune_prop(fsgs);
00899     fsg_history_end_frame(fsgs->history);
00900 
00901     /*
00902      * Propagate new history entries through any null transitions, creating
00903      * new history entries, and then make the survivors permanent.
00904      */
00905     fsg_search_null_prop(fsgs);
00906     fsg_history_end_frame(fsgs->history);
00907 
00908     /*
00909      * Perform cross-word transitions; propagate each history entry across its
00910      * terminating state to the root nodes of the lextree attached to the state.
00911      */
00912     fsg_search_word_trans(fsgs);
00913 
00914     /*
00915      * We've now come full circle, HMM and FSG states have been updated for
00916      * the next frame.
00917      * Update the active lists, deactivate any currently active HMMs that
00918      * did not survive into the next frame
00919      */
00920     for (gn = fsgs->pnode_active; gn; gn = gnode_next(gn)) {
00921         pnode = (fsg_pnode_t *) gnode_ptr(gn);
00922         hmm = fsg_pnode_hmmptr(pnode);
00923 
00924         if (hmm_frame(hmm) == fsgs->frame) {
00925             /* This HMM NOT activated for the next frame; reset it */
00926             fsg_psubtree_pnode_deactivate(pnode);
00927         }
00928         else {
00929             assert(hmm_frame(hmm) == (fsgs->frame + 1));
00930         }
00931     }
00932 
00933     /* Free the currently active list */
00934     glist_free(fsgs->pnode_active);
00935 
00936     /* Make the next-frame active list the current one */
00937     fsgs->pnode_active = fsgs->pnode_active_next;
00938     fsgs->pnode_active_next = NULL;
00939 
00940     /* End of this frame; ready for the next */
00941     ++fsgs->frame;
00942 
00943     return 1;
00944 }

01043 static int
01044 fsg_search_find_exit(fsg_search_t *fsgs, int frame_idx, int final, int32 *out_score, int32* out_is_final)
01045 {
01046     fsg_hist_entry_t *hist_entry;
01047     fsg_model_t *fsg;
01048     int bpidx, frm, last_frm, besthist;
01049     int32 bestscore;
01050 
01051     if (frame_idx == -1)
01052         frame_idx = fsgs->frame - 1;
01053     last_frm = frm = frame_idx;
01054 
01055     /* Scan backwards to find a word exit in frame_idx. */
01056     bpidx = fsg_history_n_entries(fsgs->history) - 1;
01057     while (bpidx > 0) {
01058         hist_entry = fsg_history_entry_get(fsgs->history, bpidx);
01059         if (fsg_hist_entry_frame(hist_entry) <= frame_idx) {
01060             frm = last_frm = fsg_hist_entry_frame(hist_entry);
01061             break;
01062         }
01063     }
01064 
01065     /* No hypothesis (yet). */
01066     if (bpidx <= 0) 
01067         return bpidx;
01068 
01069     /* Now find best word exit in this frame. */
01070     bestscore = INT_MIN;
01071     besthist = -1;
01072     fsg = fsgs->fsg;
01073     while (frm == last_frm) {
01074         fsg_link_t *fl;
01075         int32 score;
01076 
01077         fl = fsg_hist_entry_fsglink(hist_entry);
01078         score = fsg_hist_entry_score(hist_entry);
01079         
01080         if (fl == NULL)
01081             break;
01082 
01083         /* Prefer final hypothesis */
01084         if (score == bestscore && fsg_link_to_state(fl) == fsg_model_final_state(fsg)) {
01085             besthist = bpidx;
01086         } else if (score BETTER_THAN bestscore) {
01087             /* Only enforce the final state constraint if this is a final hypothesis. */
01088             if ((!final)
01089                 || fsg_link_to_state(fl) == fsg_model_final_state(fsg)) {
01090                 bestscore = score;
01091                 besthist = bpidx;
01092             }
01093         }
01094         
01095         --bpidx;
01096         if (bpidx < 0)
01097             break;
01098         hist_entry = fsg_history_entry_get(fsgs->history, bpidx);
01099         frm = fsg_hist_entry_frame(hist_entry);
01100     }
01101 
01102     /* Final state not reached. */
01103     if (besthist == -1) {
01104         E_ERROR("Final result does not match the grammar in frame %d\n", frame_idx);
01105         return -1;
01106     }
01107 
01108     /* This here's the one we want. */
01109     if (out_score)
01110         *out_score = bestscore;
01111     if (out_is_final) {
01112         fsg_link_t *fl;
01113         hist_entry = fsg_history_entry_get(fsgs->history, besthist);
01114         fl = fsg_hist_entry_fsglink(hist_entry);
01115         *out_is_final = (fsg_link_to_state(fl) == fsg_model_final_state(fsg));
01116     }
01117     return besthist;
01118 }

01539 static ps_lattice_t *
01540 fsg_search_lattice(ps_search_t *search)
01541 {
01542     fsg_search_t *fsgs;
01543     fsg_model_t *fsg;
01544     ps_latnode_t *node;
01545     ps_lattice_t *dag;
01546     int32 i, n;
01547 
01548     fsgs = (fsg_search_t *)search;
01549 
01550     /* Check to see if a lattice has previously been created over the
01551      * same number of frames, and reuse it if so. */
01552     if (search->dag && search->dag->n_frames == fsgs->frame)
01553         return search->dag;
01554 
01555     /* Nope, create a new one. */
01556     ps_lattice_free(search->dag);
01557     search->dag = NULL;
01558     dag = ps_lattice_init_search(search, fsgs->frame);
01559     fsg = fsgs->fsg;
01560 
01561     /*
01562      * Each history table entry represents a link in the word graph.
01563      * The set of nodes is determined by the number of unique
01564      * (word,start-frame) pairs in the history table.  So we will
01565      * first find all those nodes.
01566      */
01567     n = fsg_history_n_entries(fsgs->history);
01568     for (i = 0; i < n; ++i) {
01569         fsg_hist_entry_t *fh = fsg_history_entry_get(fsgs->history, i);
01570         int32 ascr;
01571         int sf;
01572 
01573         /* Skip null transitions. */
01574         if (fh->fsglink == NULL || fh->fsglink->wid == -1)
01575             continue;
01576 
01577         /* Find the start node of this link. */
01578         if (fh->pred) {
01579             fsg_hist_entry_t *pfh = fsg_history_entry_get(fsgs->history, fh->pred);
01580             /* FIXME: We include the transition score in the lattice
01581              * link score.  This is because of the practical
01582              * difficulty of obtaining it separately in bestpath or
01583              * forward-backward search, and because it is essentially
01584              * a unigram probability, so there is no need to treat it
01585              * separately from the acoustic score.  However, it's not
01586              * clear that this will actually yield correct results.*/
01587             ascr = fh->score - pfh->score;
01588             sf = pfh->frame + 1;
01589         }
01590         else {
01591             ascr = fh->score;
01592             sf = 0;
01593         }
01594 
01595         /*
01596          * Note that although scores are tied to links rather than
01597          * nodes, it's possible that there are no links out of the
01598          * destination node, and thus we need to preserve its score in
01599          * case it turns out to be utterance-final.
01600          */
01601         new_node(dag, fsg, sf, fh->frame, fh->fsglink->wid, fsg_link_to_state(fh->fsglink), ascr);
01602     }
01603 
01604     /*
01605      * Now, we will create links only to nodes that actually exist.
01606      */
01607     n = fsg_history_n_entries(fsgs->history);
01608     for (i = 0; i < n; ++i) {
01609         fsg_hist_entry_t *fh = fsg_history_entry_get(fsgs->history, i);
01610         fsg_arciter_t *itor;
01611         ps_latnode_t *src, *dest;
01612         int32 ascr;
01613         int sf;
01614 
01615         /* Skip null transitions. */
01616         if (fh->fsglink == NULL || fh->fsglink->wid == -1)
01617             continue;
01618 
01619         /* Find the start node of this link and calculate its link score. */
01620         if (fh->pred) {
01621             fsg_hist_entry_t *pfh = fsg_history_entry_get(fsgs->history, fh->pred);
01622             sf = pfh->frame + 1;
01623             ascr = fh->score - pfh->score;
01624         }
01625         else {
01626             ascr = fh->score;
01627             sf = 0;
01628         }
01629         src = find_node(dag, fsg, sf, fh->fsglink->wid, fsg_link_to_state(fh->fsglink));
01630         sf = fh->frame + 1;
01631 
01632         for (itor = fsg_model_arcs(fsg, fsg_link_to_state(fh->fsglink));
01633              itor; itor = fsg_arciter_next(itor)) {
01634             fsg_link_t *link = fsg_arciter_get(itor);
01635             
01636             /* FIXME: Need to figure out what to do about tag transitions. */
01637             if (link->wid >= 0) {
01638                 /*
01639                  * For each non-epsilon link following this one, look for a
01640                  * matching node in the lattice and link to it.
01641                  */
01642                 if ((dest = find_node(dag, fsg, sf, link->wid, fsg_link_to_state(link))) != NULL)
01643                     ps_lattice_link(dag, src, dest, ascr, fh->frame);
01644             }
01645             else {
01646                 /*
01647                  * Transitive closure on nulls has already been done, so we
01648                  * just need to look one link forward from them.
01649                  */
01650                 fsg_arciter_t *itor2;
01651                 
01652                 /* Add all non-null links out of j. */
01653                 for (itor2 = fsg_model_arcs(fsg, fsg_link_to_state(link));
01654                      itor2; itor2 = fsg_arciter_next(itor2)) {
01655                     fsg_link_t *link = fsg_arciter_get(itor2);
01656 
01657                     if (link->wid == -1)
01658                         continue;
01659                     
01660                     if ((dest = find_node(dag, fsg, sf, link->wid, fsg_link_to_state(link))) != NULL) {
01661                         ps_lattice_link(dag, src, dest, ascr, fh->frame);
01662                     }
01663                 }
01664             }
01665         }
01666     }
01667 
01668 
01669     /* Figure out which nodes are the start and end nodes. */
01670     if ((dag->start = find_start_node(fsgs, dag)) == NULL) {
01671         E_WARN("Failed to find the start node\n");
01672         goto error_out;
01673     }
01674     if ((dag->end = find_end_node(fsgs, dag)) == NULL) {
01675         E_WARN("Failed to find the end node\n");
01676         goto error_out;
01677     }
01678 
01679 
01680     E_INFO("lattice start node %s.%d end node %s.%d\n",
01681            fsg_model_word_str(fsg, dag->start->wid), dag->start->sf,
01682            fsg_model_word_str(fsg, dag->end->wid), dag->end->sf);
01683     /* FIXME: Need to calculate final_node_ascr here. */
01684 
01685     /*
01686      * Convert word IDs from FSG to dictionary.
01687      */
01688     for (node = dag->nodes; node; node = node->next) {
01689         node->wid = dict_wordid(dag->search->dict,
01690                                 fsg_model_word_str(fsg, node->wid));
01691         node->basewid = dict_basewid(dag->search->dict, node->wid);
01692     }
01693 
01694     /*
01695      * Now we are done, because the links in the graph are uniquely
01696      * defined by the history table.  However we should remove any
01697      * nodes which are not reachable from the end node of the FSG.
01698      * Everything is reachable from the start node by definition.
01699      */
01700     mark_reachable(dag, dag->end);
01701 
01702     ps_lattice_delete_unreachable(dag);
01703     {
01704         int32 silpen, fillpen;
01705 
01706         silpen = (int32)(logmath_log(fsg->lmath,
01707                                      cmd_ln_float32_r(ps_search_config(fsgs), "-silprob"))
01708                          * fsg->lw)
01709             >> SENSCR_SHIFT;
01710         fillpen = (int32)(logmath_log(fsg->lmath,
01711                                       cmd_ln_float32_r(ps_search_config(fsgs), "-fillprob"))
01712                           * fsg->lw)
01713             >> SENSCR_SHIFT;
01714         ps_lattice_bypass_fillers(dag, silpen, fillpen);
01715     }
01716     search->dag = dag;
01717 
01718     return dag;
01719 
01720 
01721 error_out:

01723     return NULL;
