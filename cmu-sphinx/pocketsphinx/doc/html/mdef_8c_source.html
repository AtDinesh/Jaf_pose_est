00157             m->wpos_ci_lclist[wpos][(int) ci] = lcptr;  /* This is what needs to be freed */
00158         }
00159         if ((rcptr = find_ph_rc(lcptr->rclist, rc)) != NULL) {
00160             __BIGSTACKVARIABLE__ char buf[4096];
00161 
00162             mdef_phone_str(m, rcptr->pid, buf);

00164         }

00167         rcptr->rc = rc;

00219 mdef_phone_id(mdef_t * m,
00220               int ci, int lc, int rc, word_posn_t wpos)
00221 {
00222     ph_lc_t *lcptr;
00223     ph_rc_t *rcptr;
00224     int newl, newr;
00225 
00226     assert(m);
00227     assert((ci >= 0) && (ci < m->n_ciphone));
00228     assert((lc >= 0) && (lc < m->n_ciphone));
00229     assert((rc >= 0) && (rc < m->n_ciphone));
00230     assert((wpos >= 0) && (wpos < N_WORD_POSN));
00231 
00232     if (((lcptr =
00233           find_ph_lc(m->wpos_ci_lclist[wpos][(int) ci], lc)) == NULL)
00234         || ((rcptr = find_ph_rc(lcptr->rclist, rc)) == NULL)) {
00235         /* Not found; backoff to silence context if non-silence filler context */
00236         if (m->sil < 0)
00237             return -1;
00238 
00239         newl = m->ciphone[(int) lc].filler ? m->sil : lc;
00240         newr = m->ciphone[(int) rc].filler ? m->sil : rc;
00241         if ((newl == lc) && (newr == rc))
00242             return -1;
00243 
00244         return (mdef_phone_id(m, ci, newl, newr, wpos));
00245     }
00246 
00247     return (rcptr->pid);
00248 }

00271 /* Parse tmat and state->senone mappings for phone p and fill in structure */
00272 static void
00273 parse_tmat_senmap(mdef_t * m, char *line, int32 off, int p)
00274 {
00275     int32 wlen, n, s;
00276     __BIGSTACKVARIABLE__ char word[1024], *lp;
00277 
00278     lp = line + off;
00279 
00280     /* Read transition matrix id */
00281     if ((sscanf(lp, "%d%n", &n, &wlen) != 1) || (n < 0))
00282         E_FATAL("Missing or bad transition matrix id: %s\n", line);
00283     m->phone[p].tmat = n;
00284     if (m->n_tmat <= n)
00285         E_FATAL("tmat-id(%d) > #tmat in header(%d): %s\n", n, m->n_tmat,
00286                 line);
00287     lp += wlen;
00288 
00289     /* Read senone mappings for each emitting state */
00290     for (n = 0; n < m->n_emit_state; n++) {
00291         if ((sscanf(lp, "%d%n", &s, &wlen) != 1) || (s < 0))
00292             E_FATAL("Missing or bad state[%d]->senone mapping: %s\n", n,
00293                     line);
00294 
00295         if ((p < m->n_ciphone) && (m->n_ci_sen <= s))
00296             E_FATAL("CI-senone-id(%d) > #CI-senones(%d): %s\n", s,
00297                     m->n_ci_sen, line);
00298         if (m->n_sen <= s)
00299             E_FATAL("Senone-id(%d) > #senones(%d): %s\n", s, m->n_sen,
00300                     line);
00301 
00302         m->sseq[p][n] = s;
00303         lp += wlen;
00304     }
00305 
00306     /* Check for the last non-emitting state N */
00307     if ((sscanf(lp, "%s%n", word, &wlen) != 1) || (strcmp(word, "N") != 0))
00308         E_FATAL("Missing non-emitting state spec: %s\n", line);
00309     lp += wlen;
00310 
00311     /* Check for end of line */
00312     if (sscanf(lp, "%s%n", word, &wlen) == 1)
00313         E_FATAL("Non-empty beyond non-emitting final state: %s\n", line);
00314 }

00317 static void
00318 parse_base_line(mdef_t * m, char *line, int p)
00319 {
00320     int32 wlen, n;
00321     __BIGSTACKVARIABLE__ char word[1024], *lp;
00322     int ci;
00323 
00324     lp = line;
00325 
00326     /* Read base phone name */
00327     if (sscanf(lp, "%s%n", word, &wlen) != 1)
00328         E_FATAL("Missing base phone name: %s\n", line);
00329     lp += wlen;
00330 
00331     /* Make sure it's not a duplicate */
00332     ci = mdef_ciphone_id(m, word);
00333     if (ci >= 0)
00334         E_FATAL("Duplicate base phone: %s\n", line);
00335 
00336     /* Add ciphone to ciphone table with id p */
00337     ciphone_add(m, word, p);
00338     ci = (int) p;
00339 
00340     /* Read and skip "-" for lc, rc, wpos */
00341     for (n = 0; n < 3; n++) {
00342         if ((sscanf(lp, "%s%n", word, &wlen) != 1)
00343             || (strcmp(word, "-") != 0))
00344             E_FATAL("Bad context info for base phone: %s\n", line);
00345         lp += wlen;
00346     }
00347 
00348     /* Read filler attribute, if present */
00349     if (sscanf(lp, "%s%n", word, &wlen) != 1)
00350         E_FATAL("Missing filler atribute field: %s\n", line);
00351     lp += wlen;
00352     if (strcmp(word, "filler") == 0)
00353         m->ciphone[(int) ci].filler = 1;
00354     else if (strcmp(word, "n/a") == 0)
00355         m->ciphone[(int) ci].filler = 0;
00356     else
00357         E_FATAL("Bad filler attribute field: %s\n", line);
00358 
00359     triphone_add(m, ci, -1, -1, WORD_POSN_UNDEFINED, p);
00360 
00361     /* Parse remainder of line: transition matrix and state->senone mappings */
00362     parse_tmat_senmap(m, line, lp - line, p);
00363 }

00366 static void
00367 parse_tri_line(mdef_t * m, char *line, int p)
00368 {
00369     int32 wlen;
00370     __BIGSTACKVARIABLE__ char word[1024], *lp;
00371     int ci, lc, rc;
00372     word_posn_t wpos = WORD_POSN_BEGIN;
00373 
00374     lp = line;
00375 
00376     /* Read base phone name */
00377     if (sscanf(lp, "%s%n", word, &wlen) != 1)
00378         E_FATAL("Missing base phone name: %s\n", line);
00379     lp += wlen;
00380 
00381     ci = mdef_ciphone_id(m, word);
00382     if (ci < 0)
00383         E_FATAL("Unknown base phone: %s\n", line);
00384 
00385     /* Read lc */
00386     if (sscanf(lp, "%s%n", word, &wlen) != 1)
00387         E_FATAL("Missing left context: %s\n", line);
00388     lp += wlen;
00389     lc = mdef_ciphone_id(m, word);
00390     if (lc < 0)
00391         E_FATAL("Unknown left context: %s\n", line);
00392 
00393     /* Read rc */
00394     if (sscanf(lp, "%s%n", word, &wlen) != 1)
00395         E_FATAL("Missing right context: %s\n", line);
00396     lp += wlen;
00397     rc = mdef_ciphone_id(m, word);
00398     if (rc < 0)
00399         E_FATAL("Unknown right  context: %s\n", line);
00400 
00401     /* Read tripone word-position within word */
00402     if ((sscanf(lp, "%s%n", word, &wlen) != 1) || (word[1] != '\0'))
00403         E_FATAL("Missing or bad word-position spec: %s\n", line);
00404     lp += wlen;
00405     switch (word[0]) {
00406     case 'b':
00407         wpos = WORD_POSN_BEGIN;
00408         break;
00409     case 'e':
00410         wpos = WORD_POSN_END;
00411         break;
00412     case 's':
00413         wpos = WORD_POSN_SINGLE;
00414         break;
00415     case 'i':
00416         wpos = WORD_POSN_INTERNAL;
00417         break;
00418     default:
00419         E_FATAL("Bad word-position spec: %s\n", line);
00420     }
00421 
00422     /* Read filler attribute, if present.  Must match base phone attribute */
00423     if (sscanf(lp, "%s%n", word, &wlen) != 1)
00424         E_FATAL("Missing filler attribute field: %s\n", line);
00425     lp += wlen;
00426     if (((strcmp(word, "filler") == 0) && (m->ciphone[(int) ci].filler)) ||
00427         ((strcmp(word, "n/a") == 0) && (!m->ciphone[(int) ci].filler))) {
00428         /* Everything is fine */
00429     }
00430     else
00431         E_FATAL("Bad filler attribute field: %s\n", line);
00432 
00433     triphone_add(m, ci, lc, rc, wpos, p);
00434 
00435     /* Parse remainder of line: transition matrix and state->senone mappings */
00436     parse_tmat_senmap(m, line, lp - line, p);
00437 }

00440 static void
00441 sseq_compress(mdef_t * m)
00442 {
00443     hash_table_t *h;
00444     uint16 **sseq;
00445     int32 n_sseq;
00446     int32 p, j, k;
00447     glist_t g;
00448     gnode_t *gn;
00449     hash_entry_t *he;
00450 
00451     k = m->n_emit_state * sizeof(int16);
00452 
00453     h = hash_table_new(m->n_phone, HASH_CASE_YES);
00454     n_sseq = 0;
00455 
00456     /* Identify unique senone-sequence IDs.  BUG: tmat-id not being considered!! */
00457     for (p = 0; p < m->n_phone; p++) {
00458         /* Add senone sequence to hash table */
00459         if (n_sseq
00460             == (j = hash_table_enter_bkey_int32(h, (char *)m->sseq[p], k, n_sseq)))
00461             n_sseq++;
00462 
00463         m->phone[p].ssid = j;
00464     }
00465 
00466     /* Generate compacted sseq table */
00467     sseq = ckd_calloc_2d(n_sseq, m->n_emit_state, sizeof(**sseq)); /* freed in mdef_free() */
00468 
00469     g = hash_table_tolist(h, &j);
00470     assert(j == n_sseq);
00471 
00472     for (gn = g; gn; gn = gnode_next(gn)) {
00473         he = (hash_entry_t *) gnode_ptr(gn);
00474         j = (long)hash_entry_val(he);
00475         memcpy(sseq[j], hash_entry_key(he), k);
00476     }
00477     glist_free(g);
00478 
00479     /* Free the old, temporary senone sequence table, replace with compacted one */
00480     ckd_free_2d(m->sseq);
00481     m->sseq = sseq;
00482     m->n_sseq = n_sseq;
00483 
00484     hash_table_free(h);
00485 }

00499 /*
00500  * Initialize phones (ci and triphones) and state->senone mappings from .mdef file.
00501  */
00502 mdef_t *
00503 mdef_init(char *mdeffile, int32 breport)
00504 {
00505     FILE *fp;
00506     int32 n_ci, n_tri, n_map, n;
00507     __BIGSTACKVARIABLE__ char tag[1024], buf[1024];
00508     uint16 **senmap;
00509     int p;
00510     int32 s, ci, cd;
00511     mdef_t *m;
00512 
00513     if (!mdeffile)
00514         E_FATAL("No mdef-file\n");
00515 
00516     if (breport)
00517         E_INFO("Reading model definition: %s\n", mdeffile);
00518 
00519     m = (mdef_t *) ckd_calloc(1, sizeof(mdef_t));       /* freed in mdef_free */
00520 
00521     if ((fp = fopen(mdeffile, "r")) == NULL)
00522         E_FATAL_SYSTEM("Failed to open mdef file '%s' for reading", mdeffile);
00523 
00524     if (noncomment_line(buf, sizeof(buf), fp) < 0)
00525         E_FATAL("Empty file: %s\n", mdeffile);
00526 
00527     if (strncmp(buf, "BMDF", 4) == 0 || strncmp(buf, "FDMB", 4) == 0) {
00528         E_INFO
00529             ("Found byte-order mark %.4s, assuming this is a binary mdef file\n",
00530              buf);
00531         fclose(fp);
00532         ckd_free(m);
00533         return NULL;
00534     }
00535     if (strncmp(buf, MODEL_DEF_VERSION, strlen(MODEL_DEF_VERSION)) != 0)
00536         E_FATAL("Version error: Expecing %s, but read %s\n",
00537                 MODEL_DEF_VERSION, buf);
00538 
00539     /* Read #base phones, #triphones, #senone mappings defined in header */
00540     n_ci = -1;
00541     n_tri = -1;
00542     n_map = -1;
00543     m->n_ci_sen = -1;
00544     m->n_sen = -1;
00545     m->n_tmat = -1;
00546     do {
00547         if (noncomment_line(buf, sizeof(buf), fp) < 0)
00548             E_FATAL("Incomplete header\n");
00549 
00550         if ((sscanf(buf, "%d %s", &n, tag) != 2) || (n < 0))
00551             E_FATAL("Error in header: %s\n", buf);
00552 
00553         if (strcmp(tag, "n_base") == 0)
00554             n_ci = n;
00555         else if (strcmp(tag, "n_tri") == 0)
00556             n_tri = n;
00557         else if (strcmp(tag, "n_state_map") == 0)
00558             n_map = n;
00559         else if (strcmp(tag, "n_tied_ci_state") == 0)
00560             m->n_ci_sen = n;
00561         else if (strcmp(tag, "n_tied_state") == 0)
00562             m->n_sen = n;
00563         else if (strcmp(tag, "n_tied_tmat") == 0)
00564             m->n_tmat = n;
00565         else
00566             E_FATAL("Unknown header line: %s\n", buf);
00567     } while ((n_ci < 0) || (n_tri < 0) || (n_map < 0) ||
00568              (m->n_ci_sen < 0) || (m->n_sen < 0) || (m->n_tmat < 0));
00569 
00570     if ((n_ci == 0) || (m->n_ci_sen == 0) || (m->n_tmat == 0)
00571         || (m->n_ci_sen > m->n_sen))

00608 
00609     /*
00610      * Read base phones and triphones.  They'll simply be assigned a running sequence
00611      * number as their "phone-id".  If the phone-id < n_ci, it's a ciphone.
00612      */

00614     /* Read base phones */
00615     for (p = 0; p < n_ci; p++) {

00617             E_FATAL("Premature EOF reading CIphone %d\n", p);
00618         parse_base_line(m, buf, p);
00619     }
00620     m->sil = mdef_ciphone_id(m, S3_SILENCE_CIPHONE);
00621 
00622     /* Read triphones, if any */
00623     for (; p < m->n_phone; p++) {

00625             E_FATAL("Premature EOF reading phone %d\n", p);
00626         parse_tri_line(m, buf, p);

00630         E_ERROR("Non-empty file beyond expected #phones (%d)\n",
00631                 m->n_phone);
00632 
00633     /* Build CD senones to CI senones map */
00634     if (m->n_ciphone * m->n_emit_state != m->n_ci_sen)
00635         E_FATAL
00636             ("#CI-senones(%d) != #CI-phone(%d) x #emitting-states(%d)\n",
00637              m->n_ci_sen, m->n_ciphone, m->n_emit_state);
00638     m->cd2cisen = (int16 *) ckd_calloc(m->n_sen, sizeof(*m->cd2cisen)); /* freed in mdef_free */
00639 
00640     m->sen2cimap = (int16 *) ckd_calloc(m->n_sen, sizeof(*m->sen2cimap)); /* freed in mdef_free */
00641 
00642     for (s = 0; s < m->n_sen; s++)
00643         m->sen2cimap[s] = -1;
00644     for (s = 0; s < m->n_ci_sen; s++) { /* CI senones */
00645         m->cd2cisen[s] = s;
00646         m->sen2cimap[s] = s / m->n_emit_state;
00647     }
00648     for (p = n_ci; p < m->n_phone; p++) {       /* CD senones */
00649         for (s = 0; s < m->n_emit_state; s++) {
00650             cd = m->sseq[p][s];
00651             ci = m->sseq[m->phone[p].ci][s];
00652             m->cd2cisen[cd] = ci;
00653             m->sen2cimap[cd] = m->phone[p].ci;
00654         }
00655     }
00656 
00657     sseq_compress(m);
00658     fclose(fp);
00659 
00660     return m;
00661 }
