00070     int i, nodes, ci_idx, lc_idx, rc_idx;
00071     int nchars;
00072 
00073     if ((mdef = mdef_init((char *) filename, TRUE)) == NULL)

00075 
00076     /* Enforce some limits.  */
00077     if (mdef->n_sen > BAD_SENID) {

00091         E_ERROR("Number of phones exceeds limit: %d > %d\n",
00092                 mdef->n_ciphone, 255);
00093         mdef_free(mdef);
00094         return NULL;
00095     }
00096 
00097     bmdef = ckd_calloc(1, sizeof(*bmdef));
00098     bmdef->refcnt = 1;
00099 
00100     /* Easy stuff.  The mdef.c code has done the heavy lifting for us. */
00101     bmdef->n_ciphone = mdef->n_ciphone;
00102     bmdef->n_phone = mdef->n_phone;
00103     bmdef->n_emit_state = mdef->n_emit_state;
00104     bmdef->n_ci_sen = mdef->n_ci_sen;
00105     bmdef->n_sen = mdef->n_sen;
00106     bmdef->n_tmat = mdef->n_tmat;
00107     bmdef->n_sseq = mdef->n_sseq;
00108     bmdef->sseq = mdef->sseq;
00109     bmdef->cd2cisen = mdef->cd2cisen;
00110     bmdef->sen2cimap = mdef->sen2cimap;
00111     bmdef->n_ctx = 3;           /* Triphones only. */
00112     bmdef->sil = mdef->sil;
00113     mdef->sseq = NULL;          /* We are taking over this one. */
00114     mdef->cd2cisen = NULL;      /* And this one. */
00115     mdef->sen2cimap = NULL;     /* And this one. */
00116

00118      * ASCII-betically then we are in a world of hurt and
00119      * therefore will simply refuse to continue. */

00124     bmdef->ciname[0] = ckd_calloc(nchars, 1);

00150             bmdef->phone[i].info.cd.ctx[2] = mdef->phone[i].rc;
00151         }
00152     }
00153 
00154     /* Walk the wpos_ci_lclist once to find the total number of
00155      * nodes and the starting locations for each level. */
00156     nodes = lc_idx = ci_idx = rc_idx = 0;
00157     for (i = 0; i < N_WORD_POSN; ++i) {
00158         int j;
00159         for (j = 0; j < mdef->n_ciphone; ++j) {
00160             ph_lc_t *lc;
00161 
00162             for (lc = mdef->wpos_ci_lclist[i][j]; lc; lc = lc->next) {
00163                 ph_rc_t *rc;
00164                 for (rc = lc->rclist; rc; rc = rc->next) {
00165                     ++nodes;    /* RC node */
00166                 }
00167                 ++nodes;        /* LC node */
00168                 ++rc_idx;       /* Start of RC nodes (after LC nodes) */
00169             }
00170             ++nodes;            /* CI node */
00171             ++lc_idx;           /* Start of LC nodes (after CI nodes) */
00172             ++rc_idx;           /* Start of RC nodes (after CI and LC nodes) */
00173         }
00174         ++nodes;                /* wpos node */
00175         ++ci_idx;               /* Start of CI nodes (after wpos nodes) */
00176         ++lc_idx;               /* Start of LC nodes (after CI nodes) */
00177         ++rc_idx;               /* STart of RC nodes (after wpos, CI, and LC nodes) */
00178     }
00179     E_INFO("Allocating %d * %d bytes (%d KiB) for CD tree\n",
00180            nodes, sizeof(*bmdef->cd_tree), 
00181            nodes * sizeof(*bmdef->cd_tree) / 1024);
00182     bmdef->n_cd_tree = nodes;
00183     bmdef->cd_tree = ckd_calloc(nodes, sizeof(*bmdef->cd_tree));
00184     for (i = 0; i < N_WORD_POSN; ++i) {
00185         int j;
00186 
00187         bmdef->cd_tree[i].ctx = i;
00188         bmdef->cd_tree[i].n_down = mdef->n_ciphone;
00189         bmdef->cd_tree[i].c.down = ci_idx;
00190 #if 0
00191         E_INFO("%d => %c (%d@%d)\n",
00192                i, (WPOS_NAME)[i],
00193                bmdef->cd_tree[i].n_down, bmdef->cd_tree[i].c.down);
00194 #endif
00195 
00196         /* Now we can build the rest of the tree. */
00197         for (j = 0; j < mdef->n_ciphone; ++j) {
00198             ph_lc_t *lc;
00199 
00200             bmdef->cd_tree[ci_idx].ctx = j;
00201             bmdef->cd_tree[ci_idx].c.down = lc_idx;
00202             for (lc = mdef->wpos_ci_lclist[i][j]; lc; lc = lc->next) {
00203                 ph_rc_t *rc;
00204 
00205                 bmdef->cd_tree[lc_idx].ctx = lc->lc;
00206                 bmdef->cd_tree[lc_idx].c.down = rc_idx;
00207                 for (rc = lc->rclist; rc; rc = rc->next) {
00208                     bmdef->cd_tree[rc_idx].ctx = rc->rc;
00209                     bmdef->cd_tree[rc_idx].n_down = 0;
00210                     bmdef->cd_tree[rc_idx].c.pid = rc->pid;
00211 #if 0
00212                     E_INFO("%d => %s %s %s %c (%d@%d)\n",
00213                            rc_idx,
00214                            bmdef->ciname[j],
00215                            bmdef->ciname[lc->lc],
00216                            bmdef->ciname[rc->rc],
00217                            (WPOS_NAME)[i],
00218                            bmdef->cd_tree[rc_idx].n_down,
00219                            bmdef->cd_tree[rc_idx].c.down);
00220 #endif
00221 
00222                     ++bmdef->cd_tree[lc_idx].n_down;
00223                     ++rc_idx;
00224                 }
00225                 /* If there are no triphones here,
00226                  * this is considered a leafnode, so
00227                  * set the pid to -1. */
00228                 if (bmdef->cd_tree[lc_idx].n_down == 0)
00229                     bmdef->cd_tree[lc_idx].c.pid = -1;
00230 #if 0
00231                 E_INFO("%d => %s %s %c (%d@%d)\n",
00232                        lc_idx,
00233                        bmdef->ciname[j],
00234                        bmdef->ciname[lc->lc],
00235                        (WPOS_NAME)[i],
00236                        bmdef->cd_tree[lc_idx].n_down,
00237                        bmdef->cd_tree[lc_idx].c.down);
00238 #endif
00239 
00240                 ++bmdef->cd_tree[ci_idx].n_down;
00241                 ++lc_idx;
00242             }
00243 
00244             /* As above, so below. */
00245             if (bmdef->cd_tree[ci_idx].n_down == 0)
00246                 bmdef->cd_tree[ci_idx].c.pid = -1;
00247 #if 0
00248             E_INFO("%d => %d=%s (%d@%d)\n",
00249                    ci_idx, j, bmdef->ciname[j],
00250                    bmdef->cd_tree[ci_idx].n_down,
00251                    bmdef->cd_tree[ci_idx].c.down);
00252 #endif
00253 
00254             ++ci_idx;
00255         }
00256     }
00257 
00258     mdef_free(mdef);
00259 
00260     bmdef->alloc_mode = BIN_MDEF_FROM_TEXT;
00261     return bmdef;
00262 }

00271 int
00272 bin_mdef_free(bin_mdef_t * m)
00273 {
00274     if (m == NULL)
00275         return 0;
00276     if (--m->refcnt > 0)
00277         return m->refcnt;
00278 
00279     switch (m->alloc_mode) {
00280     case BIN_MDEF_FROM_TEXT:
00281         ckd_free(m->ciname[0]);
00282         ckd_free(m->sseq[0]);
00283         ckd_free(m->phone);
00284         ckd_free(m->cd_tree);
00285         break;
00286     case BIN_MDEF_IN_MEMORY:
00287         ckd_free(m->ciname[0]);
00288         break;
00289     case BIN_MDEF_ON_DISK:
00290         break;
00291     }
00292     if (m->filemap)
00293         mmio_file_unmap(m->filemap);
00294     ckd_free(m->cd2cisen);
00295     ckd_free(m->sen2cimap);
00296     ckd_free(m->ciname);
00297     ckd_free(m->sseq);
00298     ckd_free(m);
00299     return 0;
00300 }

00322 bin_mdef_t *
00323 bin_mdef_read(cmd_ln_t *config, const char *filename)
00324 {
00325     bin_mdef_t *m;
00326     FILE *fh;
00327     size_t tree_start;
00328     int32 val, i, swap, pos, end;
00329     int32 *sseq_size;
00330     int do_mmap;
00331 
00332     /* Try to read it as text first. */
00333     if ((m = bin_mdef_read_text(config, filename)) != NULL)
00334         return m;
00335 
00336     E_INFO("Reading binary model definition: %s\n", filename);
00337     if ((fh = fopen(filename, "rb")) == NULL)
00338         return NULL;
00339 
00340     if (fread(&val, 4, 1, fh) != 1) {
00341         fclose(fh);
00342         E_ERROR_SYSTEM("Failed to read byte-order marker from %s\n",
00343                        filename);
00344         return NULL;
00345     }
00346     swap = 0;
00347     if (val == BIN_MDEF_OTHER_ENDIAN) {
00348         swap = 1;
00349         E_INFO("Must byte-swap %s\n", filename);
00350     }
00351     if (fread(&val, 4, 1, fh) != 1) {
00352         fclose(fh);
00353         E_ERROR_SYSTEM("Failed to read version from %s\n", filename);
00354         return NULL;
00355     }
00356     if (swap)
00357         SWAP_INT32(&val);
00358     if (val > BIN_MDEF_FORMAT_VERSION) {
00359         E_ERROR("File format version %d for %s is newer than library\n",
00360                 val, filename);
00361         fclose(fh);
00362         return NULL;
00363     }
00364     if (fread(&val, 4, 1, fh) != 1) {
00365         fclose(fh);
00366         E_ERROR_SYSTEM("Failed to read header length from %s\n", filename);
00367         return NULL;
00368     }
00369     if (swap)
00370         SWAP_INT32(&val);
00371     /* Skip format descriptor. */
00372     fseek(fh, val, SEEK_CUR);
00373 
00374     /* Finally allocate it. */
00375     m = ckd_calloc(1, sizeof(*m));
00376     m->refcnt = 1;
00377 
00378     /* Check these, to make gcc/glibc shut up. */
00379 #define FREAD_SWAP32_CHK(dest)                                          \
00380     if (fread((dest), 4, 1, fh) != 1) {                                 \
00381         fclose(fh);                                                     \
00382         ckd_free(m);                                                    \
00383         E_ERROR_SYSTEM("Failed to read %s from %s\n", #dest, filename); \
00384         return NULL;                                                    \
00385     }                                                                   \
00386     if (swap) SWAP_INT32(dest);
00387     
00388     FREAD_SWAP32_CHK(&m->n_ciphone);
00389     FREAD_SWAP32_CHK(&m->n_phone);
00390     FREAD_SWAP32_CHK(&m->n_emit_state);
00391     FREAD_SWAP32_CHK(&m->n_ci_sen);
00392     FREAD_SWAP32_CHK(&m->n_sen);
00393     FREAD_SWAP32_CHK(&m->n_tmat);
00394     FREAD_SWAP32_CHK(&m->n_sseq);
00395     FREAD_SWAP32_CHK(&m->n_ctx);
00396     FREAD_SWAP32_CHK(&m->n_cd_tree);
00397     FREAD_SWAP32_CHK(&m->sil);
00398 
00399     /* CI names are first in the file. */
00400     m->ciname = ckd_calloc(m->n_ciphone, sizeof(*m->ciname));
00401 
00402     /* Decide whether to read in the whole file or mmap it. */
00403     do_mmap = config ? cmd_ln_boolean_r(config, "-mmap") : TRUE;
00404     if (swap) {
00405         E_WARN("-mmap specified, but mdef is other-endian.  Will not memory-map.\n");
00406         do_mmap = FALSE;
00407     } 
00408     /* Actually try to mmap it. */
00409     if (do_mmap) {
00410         m->filemap = mmio_file_read(filename);
00411         if (m->filemap == NULL)
00412             do_mmap = FALSE;
00413     }
00414     pos = ftell(fh);
00415     if (do_mmap) {
00416         /* Get the base pointer from the memory map. */
00417         m->ciname[0] = (char *)mmio_file_ptr(m->filemap) + pos;
00418         /* Success! */
00419         m->alloc_mode = BIN_MDEF_ON_DISK;
00420     }
00421     else {
00422         /* Read everything into memory. */
00423         m->alloc_mode = BIN_MDEF_IN_MEMORY;
00424         fseek(fh, 0, SEEK_END);
00425         end = ftell(fh);
00426         fseek(fh, pos, SEEK_SET);
00427         m->ciname[0] = ckd_malloc(end - pos);
00428         if (fread(m->ciname[0], 1, end - pos, fh) != end - pos)
00429             E_FATAL("Failed to read %d bytes of data from %s\n", end - pos, filename);
00430     }
00431 
00432     for (i = 1; i < m->n_ciphone; ++i)
00433         m->ciname[i] = m->ciname[i - 1] + strlen(m->ciname[i - 1]) + 1;
00434 
00435     /* Skip past the padding. */
00436     tree_start =
00437         m->ciname[i - 1] + strlen(m->ciname[i - 1]) + 1 - m->ciname[0];
00438     tree_start = (tree_start + 3) & ~3;
00439     m->cd_tree = (cd_tree_t *) (m->ciname[0] + tree_start);
00440     if (swap) {
00441         for (i = 0; i < m->n_cd_tree; ++i) {
00442             SWAP_INT16(&m->cd_tree[i].ctx);
00443             SWAP_INT16(&m->cd_tree[i].n_down);
00444             SWAP_INT32(&m->cd_tree[i].c.down);
00445         }
00446     }
00447     m->phone = (mdef_entry_t *) (m->cd_tree + m->n_cd_tree);
00448     if (swap) {
00449         for (i = 0; i < m->n_phone; ++i) {
00450             SWAP_INT32(&m->phone[i].ssid);
00451             SWAP_INT32(&m->phone[i].tmat);
00452         }
00453     }
00454     sseq_size = (int32 *) (m->phone + m->n_phone);
00455     if (swap)
00456         SWAP_INT32(sseq_size);
00457     m->sseq = ckd_calloc(m->n_sseq, sizeof(*m->sseq));
00458     m->sseq[0] = (uint16 *) (sseq_size + 1);
00459     if (swap) {
00460         for (i = 0; i < *sseq_size; ++i)
00461             SWAP_INT16(m->sseq[0] + i);
00462     }
00463     if (m->n_emit_state) {
00464         for (i = 1; i < m->n_sseq; ++i)
00465             m->sseq[i] = m->sseq[0] + i * m->n_emit_state;
00466     }
00467     else {
00468         m->sseq_len = (uint8 *) (m->sseq[0] + *sseq_size);
00469         for (i = 1; i < m->n_sseq; ++i)
00470             m->sseq[i] = m->sseq[i - 1] + m->sseq_len[i - 1];
00471     }
00472 
00473     /* Now build the CD-to-CI mappings using the senone sequences.
00474      * This is the only really accurate way to do it, though it is
00475      * still inaccurate in the case of heterogeneous topologies or
00476      * cross-state tying. */
00477     m->cd2cisen = (int16 *) ckd_malloc(m->n_sen * sizeof(*m->cd2cisen));
00478     m->sen2cimap = (int16 *) ckd_malloc(m->n_sen * sizeof(*m->sen2cimap));
00479 
00480     /* Default mappings (identity, none) */
00481     for (i = 0; i < m->n_ci_sen; ++i)
00482         m->cd2cisen[i] = i;
00483     for (; i < m->n_sen; ++i)
00484         m->cd2cisen[i] = -1;
00485     for (i = 0; i < m->n_sen; ++i)
00486         m->sen2cimap[i] = -1;
00487     for (i = 0; i < m->n_phone; ++i) {
00488         int32 j, ssid = m->phone[i].ssid;
00489 
00490         for (j = 0; j < bin_mdef_n_emit_state_phone(m, i); ++j) {
00491             int s = bin_mdef_sseq2sen(m, ssid, j);
00492             int ci = bin_mdef_pid2ci(m, i);
00493             /* Take the first one and warn if we have cross-state tying. */
00494             if (m->sen2cimap[s] == -1)
00495                 m->sen2cimap[s] = ci;
00496             if (m->sen2cimap[s] != ci)
00497                 E_WARN
00498                     ("Senone %d is shared between multiple base phones\n",
00499                      s);
00500 
00501             if (j > bin_mdef_n_emit_state_phone(m, ci))
00502                 E_WARN("CD phone %d has fewer states than CI phone %d\n",
00503                        i, ci);
00504             else
00505                 m->cd2cisen[s] =
00506                     bin_mdef_sseq2sen(m, m->phone[ci].ssid, j);
00507         }
00508     }

00814 bin_mdef_phone_id_nearest(bin_mdef_t * m, int32 b, int32 l, int32 r, int32 pos)
00815 {

00817 
00818 
00819 
00820     /* In the future, we might back off when context is not available,
00821      * but for now we'll just return the CI phone. */
00822     if (l < 0 || r < 0)
00823         return b;
00824 
00825     p = bin_mdef_phone_id(m, b, l, r, pos);
00826     if (p >= 0)
00827         return p;
00828 
00829     /* Exact triphone not found; backoff to other word positions */
00830     for (tmppos = 0; tmppos < N_WORD_POSN; tmppos++) {
00831         if (tmppos != pos) {
00832             p = bin_mdef_phone_id(m, b, l, r, tmppos);
00833             if (p >= 0)
00834                 return p;
00835         }
00836     }
00837 
00838     /* Nothing yet; backoff to silence phone if non-silence filler context */
00839     /* In addition, backoff to silence phone on left/right if in beginning/end position */
00840     if (m->sil >= 0) {
00841         int newl = l, newr = r;
00842         if (m->phone[(int)l].info.ci.filler
00843             || pos == WORD_POSN_BEGIN || pos == WORD_POSN_SINGLE)
00844             newl = m->sil;
00845         if (m->phone[(int)r].info.ci.filler
00846             || pos == WORD_POSN_END || pos == WORD_POSN_SINGLE)
00847             newr = m->sil;

00849             p = bin_mdef_phone_id(m, b, newl, newr, pos);
00850             if (p >= 0)
00851                 return p;
00852 
00853             for (tmppos = 0; tmppos < N_WORD_POSN; tmppos++) {
00854                 if (tmppos != pos) {
00855                     p = bin_mdef_phone_id(m, b, newl, newr, tmppos);
00856                     if (p >= 0)
00857                         return p;
00858                 }
00859             }
00860         }
