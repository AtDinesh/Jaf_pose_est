00081 ps_mgau_t *
00082 ms_mgau_init(acmod_t *acmod, logmath_t *lmath, bin_mdef_t *mdef)
00083 {
00084     /* Codebooks */
00085     ms_mgau_model_t *msg;
00086     ps_mgau_t *mg;
00087     gauden_t *g;
00088     senone_t *s;
00089     cmd_ln_t *config;
00090     int i;
00091 
00092     config = acmod->config;
00093 
00094     msg = (ms_mgau_model_t *) ckd_calloc(1, sizeof(ms_mgau_model_t));
00095     msg->config = config;
00096     msg->g = NULL;
00097     msg->s = NULL;
00098     
00099     g = msg->g = gauden_init(cmd_ln_str_r(config, "-mean"),
00100                              cmd_ln_str_r(config, "-var"),
00101                              cmd_ln_float32_r(config, "-varfloor"),
00102                              lmath);
00103 
00104     /* Verify n_feat and veclen, against acmod. */
00105     if (g->n_feat != feat_dimension1(acmod->fcb)) {
00106         E_ERROR("Number of streams does not match: %d != %d\n",
00107                 g->n_feat, feat_dimension1(acmod->fcb));
00108         goto error_out;
00109     }
00110     for (i = 0; i < g->n_feat; ++i) {
00111         if (g->featlen[i] != feat_dimension2(acmod->fcb, i)) {
00112             E_ERROR("Dimension of stream %d does not match: %d != %d\n", i,
00113                     g->featlen[i], feat_dimension2(acmod->fcb, i));
00114             goto error_out;
00115         }
00116     }
00117 
00118     s = msg->s = senone_init(msg->g,
00119                              cmd_ln_str_r(config, "-mixw"),
00120                              cmd_ln_str_r(config, "-senmgau"),
00121                              cmd_ln_float32_r(config, "-mixwfloor"),
00122                              lmath, mdef);
00123 
00124     s->aw = cmd_ln_int32_r(config, "-aw");
00125 
00126     /* Verify senone parameters against gauden parameters */
00127     if (s->n_feat != g->n_feat)
00128         E_FATAL("#Feature mismatch: gauden= %d, senone= %d\n", g->n_feat,
00129                 s->n_feat);
00130     if (s->n_cw != g->n_density)
00131         E_FATAL("#Densities mismatch: gauden= %d, senone= %d\n",
00132                 g->n_density, s->n_cw);
00133     if (s->n_gauden > g->n_mgau)
00134         E_FATAL("Senones need more codebooks (%d) than present (%d)\n",
00135                 s->n_gauden, g->n_mgau);
00136     if (s->n_gauden < g->n_mgau)
00137         E_ERROR("Senones use fewer codebooks (%d) than present (%d)\n",
00138                 s->n_gauden, g->n_mgau);
00139 
00140     msg->topn = cmd_ln_int32_r(config, "-topn");
00141     E_INFO("The value of topn: %d\n", msg->topn);
00142     if (msg->topn == 0 || msg->topn > msg->g->n_density) {
00143         E_WARN
00144             ("-topn argument (%d) invalid or > #density codewords (%d); set to latter\n",
00145              msg->topn, msg->g->n_density);
00146         msg->topn = msg->g->n_density;
00147     }
00148 
00149     msg->dist = (gauden_dist_t ***)
00150         ckd_calloc_3d(g->n_mgau, g->n_feat, msg->topn,
00151                       sizeof(gauden_dist_t));
00152     msg->mgau_active = ckd_calloc(g->n_mgau, sizeof(int8));
00153 
00154     mg = (ps_mgau_t *)msg;
00155     mg->vt = &ms_mgau_funcs;
00156     return mg;
00157 error_out:
00158     ms_mgau_free(ps_mgau_base(msg));
00159     return NULL;    
00160 }

00211

00217             senscr[s] = senone_eval(sen, s, msg->dist[sen->mgau[s]], topn);
00218             if (best > senscr[s]) {
00219                 best = senscr[s];

00222 
00223         /* Normalize senone scores */
00224         for (s = 0; s < sen->n_sen; s++) {
00225             int32 bs = senscr[s] - best;
00226             if (bs > 32767)
00227                 bs = 32767;
00228             if (bs < -32768)
00229                 bs = -32768;
00230             senscr[s] = bs;
00231         }
00232     }
00233     else {
00234         int32 i, n;
00235         /* Flag all active mixture-gaussian codebooks */
00236         for (gid = 0; gid < g->n_mgau; gid++)
00237             msg->mgau_active[gid] = 0;
00238 
00239         n = 0;
00240         for (i = 0; i < n_senone_active; i++) {
00241             /* senone_active consists of deltas. */
00242             int32 s = senone_active[i] + n;
00243             msg->mgau_active[sen->mgau[s]] = 1;
00244             n = s;
00245         }
00246 
00247         /* Compute topn gaussian density values (for active codebooks) */
00248         for (gid = 0; gid < g->n_mgau; gid++) {
00249             if (msg->mgau_active[gid])
00250                 gauden_dist(g, gid, topn, feat, msg->dist[gid]);
00251         }
00252 
00253         best = (int32) 0x7fffffff;
00254         n = 0;
00255         for (i = 0; i < n_senone_active; i++) {
00256             int32 s = senone_active[i] + n;
00257             senscr[s] = senone_eval(sen, s, msg->dist[sen->mgau[s]], topn);
00258             if (best > senscr[s]) {
00259                 best = senscr[s];
00260             }
00261             n = s;
00262         }
00263 
00264         /* Normalize senone scores */
00265         n = 0;
00266         for (i = 0; i < n_senone_active; i++) {
00267             int32 s = senone_active[i] + n;
00268             int32 bs = senscr[s] - best;
