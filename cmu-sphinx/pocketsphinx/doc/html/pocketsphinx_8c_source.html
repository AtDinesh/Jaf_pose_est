00192 int
00193 ps_reinit(ps_decoder_t *ps, cmd_ln_t *config)
00194 {
00195     char const *lmfile, *lmctl = NULL;
00196 
00197     if (config && config != ps->config) {
00198         cmd_ln_free_r(ps->config);
00199         ps->config = cmd_ln_retain(config);
00200     }
00201 
00202     err_set_debug_level(cmd_ln_int32_r(ps->config, "-debug"));
00203     ps->mfclogdir = cmd_ln_str_r(ps->config, "-mfclogdir");
00204     ps->rawlogdir = cmd_ln_str_r(ps->config, "-rawlogdir");
00205     ps->senlogdir = cmd_ln_str_r(ps->config, "-senlogdir");
00206 
00207     /* Fill in some default arguments. */
00208     ps_init_defaults(ps);
00209 
00210     /* Free old searches (do this before other reinit) */
00211     ps_free_searches(ps);
00212 
00213     /* Free old acmod. */
00214     acmod_free(ps->acmod);
00215     ps->acmod = NULL;
00216 
00217     /* Free old dictionary (must be done after the two things above) */
00218     dict_free(ps->dict);
00219     ps->dict = NULL;
00220     
00221     /* Free d2p */
00222     dict2pid_free(ps->d2p);
00223     ps->d2p = NULL;
00224 
00225     /* Logmath computation (used in acmod and search) */
00226     if (ps->lmath == NULL
00227         || (logmath_get_base(ps->lmath) != 
00228             (float64)cmd_ln_float32_r(ps->config, "-logbase"))) {
00229         if (ps->lmath)
00230             logmath_free(ps->lmath);
00231         ps->lmath = logmath_init
00232             ((float64)cmd_ln_float32_r(ps->config, "-logbase"), 0,
00233              cmd_ln_boolean_r(ps->config, "-bestpath"));
00234     }
00235 
00236     /* Acoustic model (this is basically everything that
00237      * uttproc.c, senscr.c, and others used to do) */
00238     if ((ps->acmod = acmod_init(ps->config, ps->lmath, NULL, NULL)) == NULL)
00239         return -1;
00240 
00241     if ((ps->pl_window = cmd_ln_int32_r(ps->config, "-pl_window"))) {
00242         /* Initialize an auxiliary phone loop search, which will run in
00243          * "parallel" with FSG or N-Gram search. */
00244         if ((ps->phone_loop = phone_loop_search_init(ps->config,
00245                                                      ps->acmod, ps->dict)) == NULL)
00246             return -1;
00247         ps->searches = glist_add_ptr(ps->searches, ps->phone_loop);
00248     }
00249 
00250     /* Dictionary and triphone mappings (depends on acmod). */
00251     /* FIXME: pass config, change arguments, implement LTS, etc. */
00252     if ((ps->dict = dict_init(ps->config, ps->acmod->mdef)) == NULL)
00253         return -1;
00254 
00255     /* Determine whether we are starting out in FSG or N-Gram search mode. */
00256     if (cmd_ln_str_r(ps->config, "-fsg") || cmd_ln_str_r(ps->config, "-jsgf")) {
00257         ps_search_t *fsgs;
00258 
00259         if ((ps->d2p = dict2pid_build(ps->acmod->mdef, ps->dict)) == NULL)
00260             return -1;
00261         if ((fsgs = fsg_search_init(ps->config, ps->acmod, ps->dict, ps->d2p)) == NULL)
00262             return -1;
00263         fsgs->pls = ps->phone_loop;
00264         ps->searches = glist_add_ptr(ps->searches, fsgs);
00265         ps->search = fsgs;
00266     }
00267     else if ((lmfile = cmd_ln_str_r(ps->config, "-lm"))
00268              || (lmctl = cmd_ln_str_r(ps->config, "-lmctl"))) {
00269         ps_search_t *ngs;
00270 
00271         /* Make the acmod's feature buffer growable if we are doing two-pass search. */
00272         if (cmd_ln_boolean_r(ps->config, "-fwdflat")
00273             && cmd_ln_boolean_r(ps->config, "-fwdtree"))
00274             acmod_set_grow(ps->acmod, TRUE);
00275 
00276         if ((ps->d2p = dict2pid_build(ps->acmod->mdef, ps->dict)) == NULL)
00277             return -1;
00278         if ((ngs = ngram_search_init(ps->config, ps->acmod, ps->dict, ps->d2p)) == NULL)
00279             return -1;
00280         ngs->pls = ps->phone_loop;
00281         ps->searches = glist_add_ptr(ps->searches, ngs);
00282         ps->search = ngs;
00283     }
00284     /* Otherwise, we will initialize the search whenever the user
00285      * decides to load an FSG or a language model. */
00286     else {
00287         if ((ps->d2p = dict2pid_build(ps->acmod->mdef, ps->dict)) == NULL)
00288             return -1;
00289     }
00290 
00291     /* Initialize performance timer. */
00292     ps->perf.name = "decode";
00293     ptmr_init(&ps->perf);
00294 
00295     return 0;
00296 }

00406         ngs = (ngram_search_t *)search;
00407         /* Free any previous lmset if this is a new one. */
00408         if (ngs->lmset != NULL && ngs->lmset != lmset)
00409             ngram_model_free(ngs->lmset);
00410         ngs->lmset = lmset;
00411         /* Tell N-Gram search to update its view of the world. */
00412         if (ps_search_reinit(search, ps->dict, ps->d2p) < 0)
00413             return NULL;
00414     } else {
00415         /* Just activate the existing search */
00416         ngs = (ngram_search_t *)search;
00417     }
00418     ps->search = search;
00419     return ngs->lmset;

00456 int
00457 ps_load_dict(ps_decoder_t *ps, char const *dictfile,
00458              char const *fdictfile, char const *format)
00459 {
00460     cmd_ln_t *newconfig;
00461     dict2pid_t *d2p;
00462     dict_t *dict;
00463     gnode_t *gn;
00464     int rv;
00465 
00466     /* Create a new scratch config to load this dict (so existing one
00467      * won't be affected if it fails) */
00468     newconfig = cmd_ln_init(NULL, ps_args(), TRUE, NULL);
00469     cmd_ln_set_boolean_r(newconfig, "-dictcase",
00470                          cmd_ln_boolean_r(ps->config, "-dictcase"));
00471     cmd_ln_set_str_r(newconfig, "-dict", dictfile);
00472     if (fdictfile)
00473         cmd_ln_set_str_r(newconfig, "-fdict", fdictfile);
00474     else
00475         cmd_ln_set_str_r(newconfig, "-fdict",
00476                          cmd_ln_str_r(ps->config, "-fdict"));
00477 
00478     /* Try to load it. */
00479     if ((dict = dict_init(newconfig, ps->acmod->mdef)) == NULL) {
00480         cmd_ln_free_r(newconfig);
00481         return -1;
00482     }
00483 
00484     /* Reinit the dict2pid. */
00485     if ((d2p = dict2pid_build(ps->acmod->mdef, dict)) == NULL) {
00486         cmd_ln_free_r(newconfig);
00487         return -1;
00488     }
00489 
00490     /* Success!  Update the existing config to reflect new dicts and
00491      * drop everything into place. */
00492     cmd_ln_free_r(newconfig);
00493     cmd_ln_set_str_r(ps->config, "-dict", dictfile);
00494     if (fdictfile)
00495         cmd_ln_set_str_r(ps->config, "-fdict", fdictfile);
00496     dict_free(ps->dict);
00497     ps->dict = dict;
00498     dict2pid_free(ps->d2p);
00499     ps->d2p = d2p;
00500 
00501     /* And tell all searches to reconfigure themselves. */
00502     for (gn = ps->searches; gn; gn = gnode_next(gn)) {
00503         ps_search_t *search = gnode_ptr(gn);
00504         if ((rv = ps_search_reinit(search, dict, d2p)) < 0)
00505             return rv;
00506     }
00507 
00508     return 0;
00509 }

00518 int
00519 ps_add_word(ps_decoder_t *ps,
00520             char const *word,
00521             char const *phones,
00522             int update)
00523 {
00524     int32 wid, lmwid;
00525     ngram_model_t *lmset;
00526     s3cipid_t *pron;
00527     char **phonestr, *tmp;
00528     int np, i, rv;
00529 
00530     /* Parse phones into an array of phone IDs. */
00531     tmp = ckd_salloc(phones);
00532     np = str2words(tmp, NULL, 0);
00533     phonestr = ckd_calloc(np, sizeof(*phonestr));
00534     str2words(tmp, phonestr, np);
00535     pron = ckd_calloc(np, sizeof(*pron));
00536     for (i = 0; i < np; ++i) {
00537         pron[i] = bin_mdef_ciphone_id(ps->acmod->mdef, phonestr[i]);
00538         if (pron[i] == -1) {
00539             E_ERROR("Unknown phone %s in phone string %s\n",
00540                     phonestr[i], tmp);
00541             ckd_free(phonestr);
00542             ckd_free(tmp);
00543             ckd_free(pron);
00544             return -1;
00545         }
00546     }
00547     /* No longer needed. */
00548     ckd_free(phonestr);
00549     ckd_free(tmp);
00550 
00551     /* Add it to the dictionary. */
00552     if ((wid = dict_add_word(ps->dict, word, pron, np)) == -1) {
00553         ckd_free(pron);
00554         return -1;
00555     }
00556     /* No longer needed. */
00557     ckd_free(pron);
00558 
00559     /* Now we also have to add it to dict2pid. */
00560     dict2pid_add_word(ps->d2p, wid);
00561 
00562     if ((lmset = ps_get_lmset(ps)) != NULL) {
00563         /* Add it to the LM set (meaning, the current LM).  In a perfect
00564          * world, this would result in the same WID, but because of the
00565          * weird way that word IDs are handled, it doesn't. */
00566         if ((lmwid = ngram_model_add_word(lmset, word, 1.0))
00567             == NGRAM_INVALID_WID)
00568             return -1;
00569     }
00570  
00571     /* Rebuild the widmap and search tree if requested. */
00572     if (update) {
00573         if ((rv = ps_search_reinit(ps->search, ps->dict, ps->d2p) < 0))
00574             return rv;
00575     }
00576     return wid;
00577 }

00586     /* If this file is seekable or maxsamps is specified, then decode
00587      * the whole thing at once. */
00588     if (maxsamps != -1 || (pos = ftell(rawfh)) >= 0) {
00589         int16 *data;
00590 
00591         if (maxsamps == -1) {
00592             long endpos;
00593             fseek(rawfh, 0, SEEK_END);
00594             endpos = ftell(rawfh);
00595             fseek(rawfh, pos, SEEK_SET);
00596             maxsamps = endpos - pos;

00600         ps_process_raw(ps, data, total, FALSE, TRUE);
00601         ckd_free(data);
00602     }
00603     else {
00604         /* Otherwise decode it in a stream. */
00605         total = 0;
00606         while (!feof(rawfh)) {
00607             int16 data[256];
00608             size_t nread;
00609 
00610             nread = fread(data, sizeof(*data), sizeof(data)/sizeof(*data), rawfh);
00611             ps_process_raw(ps, data, nread, FALSE, FALSE);
00612             total += nread;
00613         }
00614     }
00615     ps_end_utt(ps);

00703 static int
00704 ps_search_forward(ps_decoder_t *ps)
00705 {
00706     int nfr;
00707 
00708     nfr = 0;
00709     while (ps->acmod->n_feat_frame > 0) {
00710         int k;
00711         if (ps->phone_loop)
00712             if ((k = ps_search_step(ps->phone_loop, ps->acmod->output_frame)) < 0)
00713                 return k;
00714         if (ps->acmod->output_frame >= ps->pl_window)
00715             if ((k = ps_search_step(ps->search,
00716                                     ps->acmod->output_frame - ps->pl_window)) < 0)
00717                 return k;
00718         acmod_advance(ps->acmod);
00719         ++ps->n_frame;
00720         ++nfr;
00721     }
00722     return nfr;
00723 }

00814 int
00815 ps_end_utt(ps_decoder_t *ps)
00816 {
00817     int rv, i;
00818 
00819     acmod_end_utt(ps->acmod);
00820 
00821     /* Search any remaining frames. */
00822     if ((rv = ps_search_forward(ps)) < 0) {
00823         ptmr_stop(&ps->perf);
00824         return rv;
00825     }
00826     /* Finish phone loop search. */
00827     if (ps->phone_loop) {
00828         if ((rv = ps_search_finish(ps->phone_loop)) < 0) {
00829             ptmr_stop(&ps->perf);
00830             return rv;
00831         }
00832     }
00833     /* Search any frames remaining in the lookahead window. */
00834     for (i = ps->acmod->output_frame - ps->pl_window;
00835          i < ps->acmod->output_frame; ++i)
00836         ps_search_step(ps->search, i);
00837     /* Finish main search. */
00838     if ((rv = ps_search_finish(ps->search)) < 0) {
00839         ptmr_stop(&ps->perf);
00840         return rv;
00841     }
00842     ptmr_stop(&ps->perf);
00843 
00844     /* Log a backtrace if requested. */
00845     if (cmd_ln_boolean_r(ps->config, "-backtrace")) {
00846         char const *uttid, *hyp;
00847         ps_seg_t *seg;
00848         int32 score;
00849 
00850         hyp = ps_get_hyp(ps, &score, &uttid);
00851         E_INFO("%s: %s (%d)\n", uttid, hyp, score);
00852         E_INFO_NOFN("%-20s %-5s %-5s %-5s %-10s %-10s %-3s\n",
00853                     "word", "start", "end", "pprob", "ascr", "lscr", "lback");
00854         for (seg = ps_seg_iter(ps, &score); seg;
00855              seg = ps_seg_next(seg)) {
00856             char const *word;

00858             int32 post, lscr, ascr, lback;
00859 
00860             word = ps_seg_word(seg);
00861             ps_seg_frames(seg, &sf, &ef);

00863             E_INFO_NOFN("%-20s %-5d %-5d %-1.3f %-10d %-10d %-3d\n",
00864                         word, sf, ef, logmath_exp(ps_get_logmath(ps), post), ascr, lscr, lback);
00865         }
00866     }
00867     return rv;
