00069 
00070     /* Look for an existing link between "from" and "to" nodes */

00072         if (fwdlink->link->to == to)
00073             break;

00076         latlink_list_t *revlink;

00082         revlink = listelem_malloc(dag->latlink_list_alloc);

00085         link->to = to;

00087         link->ef = ef;
00088         link->best_prev = NULL;

00091         fwdlink->next = from->exits;

00093         revlink->next = to->entries;
00094         to->entries = revlink;
00095     }
00096     else {
00097         /* Link already exists; just retain the best ascr */
00098         if (score BETTER_THAN fwdlink->link->ascr) {

00603             continue;

00634          * not exist in the language model.  FIXME: This is
00635          * potentially buggy, as we already do this before outputing
00636          * lattices. */

00638         silpen = pip + logmath_log(dag->lmath,
00639                                    cmd_ln_float32_r(ps->config, "-silprob"));

00641                                     cmd_ln_float32_r(ps->config, "-fillprob"));

00643     }

00649     lineiter_free(line);
00650     if (fp) fclose_comp(fp, ispipe);
00651     ckd_free(darray);
00652     return NULL;

00909 static void
00910 ps_lattice_compute_lscr(ps_seg_t *seg, ps_latlink_t *link, int to)
00911 {
00912     ngram_model_t *lmset;
00913 
00914     /* Language model score is included in the link score for FSG
00915      * search.  FIXME: Of course, this is sort of a hack :( */
00916     if (0 != strcmp(ps_search_name(seg->search), "ngram")) {
00917         seg->lback = 1; /* Unigram... */
00918         seg->lscr = 0;
00919         return;
00920     }
00921         
00922     lmset = ((ngram_search_t *)seg->search)->lmset;
00923 
00924     if (link->best_prev == NULL) {
00925         if (to) /* Sentence has only two words. */
00926             seg->lscr = ngram_bg_score(lmset, link->to->basewid,
00927                                        link->from->basewid, &seg->lback)
00928                 >> SENSCR_SHIFT;
00929         else {/* This is the start symbol, its lscr is always 0. */
00930             seg->lscr = 0;
00931             seg->lback = 1;
00932         }
00933     }
00934     else {
00935         /* Find the two predecessor words. */
00936         if (to) {
00937             seg->lscr = ngram_tg_score(lmset, link->to->basewid,
00938                                        link->from->basewid,
00939                                        link->best_prev->from->basewid,
00940                                        &seg->lback) >> SENSCR_SHIFT;
00941         }
00942         else {
00943             if (link->best_prev->best_prev)
00944                 seg->lscr = ngram_tg_score(lmset, link->from->basewid,
00945                                            link->best_prev->from->basewid,
00946                                            link->best_prev->best_prev->from->basewid,
00947                                            &seg->lback) >> SENSCR_SHIFT;
00948             else
00949                 seg->lscr = ngram_bg_score(lmset, link->from->basewid,
00950                                            link->best_prev->from->basewid,
00951                                            &seg->lback) >> SENSCR_SHIFT;
00952         }
00953     }

00956 static void
00957 ps_lattice_link2itor(ps_seg_t *seg, ps_latlink_t *link, int to)
00958 {
00959     dag_seg_t *itor = (dag_seg_t *)seg;
00960     ps_latnode_t *node;
00961 
00962     if (to) {
00963         node = link->to;
00964         seg->ef = node->lef;
00965         seg->prob = 0; /* norm + beta - norm */
00966     }
00967     else {
00968         latlink_list_t *x;
00969         ps_latnode_t *n;
00970         logmath_t *lmath = ps_search_acmod(seg->search)->lmath;
00971 
00972         node = link->from;
00973         seg->ef = link->ef;
00974         seg->prob = link->alpha + link->beta - itor->norm;
00975         /* Sum over all exits for this word and any alternate
00976            pronunciations at the same frame. */
00977         for (n = node; n; n = n->alt) {
00978             for (x = n->exits; x; x = x->next) {
00979                 if (x->link == link)
00980                     continue;
00981                 seg->prob = logmath_add(lmath, seg->prob,
00982                                         x->link->alpha + x->link->beta - itor->norm);
00983             }
00984         }
00985     }
00986     seg->word = dict_wordstr(ps_search_dict(seg->search), node->wid);
00987     seg->sf = node->sf;
00988     seg->ascr = link->ascr << SENSCR_SHIFT;
00989     /* Compute language model score from best predecessors. */
00990     ps_lattice_compute_lscr(seg, link, to);

01233  * calculation for free that way too.  (interesting research topic: is
01234  * there a reliable Viterbi analogue to word-level Forward-Backward
01235  * like there is for state-level?  Or, is it just lattice density?)
01236  */
01237 ps_latlink_t *
01238 ps_lattice_bestpath(ps_lattice_t *dag, ngram_model_t *lmset,
01239                     float32 lwf, float32 ascale)
01240 {
01241     ps_search_t *search;
01242     ps_latnode_t *node;
01243     ps_latlink_t *link;
01244     ps_latlink_t *bestend;
01245     latlink_list_t *x;
01246     logmath_t *lmath;
01247     int32 bestescr;
01248 
01249     search = dag->search;
01250     lmath = dag->lmath;
01251 
01252     /* Initialize path scores for all links exiting dag->start, and
01253      * set all other scores to the minimum.  Also initialize alphas to
01254      * log-zero. */
01255     for (node = dag->nodes; node; node = node->next) {
01256         for (x = node->exits; x; x = x->next) {
01257             x->link->path_scr = MAX_NEG_INT32;
01258             x->link->alpha = logmath_get_zero(lmath);
01259         }
01260     }
01261     for (x = dag->start->exits; x; x = x->next) {
01262         int32 n_used;
01263 
01264         /* Ignore filler words. */
01265         if (dict_filler_word(ps_search_dict(search), x->link->to->basewid)
01266             && x->link->to != dag->end)
01267             continue;
01268 
01269         /* Best path points to dag->start, obviously. */
01270         if (lmset)
01271             x->link->path_scr = x->link->ascr +
01272                 (ngram_bg_score(lmset, x->link->to->basewid,
01273                                 ps_search_start_wid(search), &n_used) 
01274                  >> SENSCR_SHIFT)
01275                  * lwf;
01276         else
01277             x->link->path_scr = x->link->ascr;
01278         x->link->best_prev = NULL;
01279         /* No predecessors for start links. */
01280         x->link->alpha = 0;
01281     }
01282 
01283     /* Traverse the edges in the graph, updating path scores. */
01284     for (link = ps_lattice_traverse_edges(dag, NULL, NULL);
01285          link; link = ps_lattice_traverse_next(dag, NULL)) {
01286         int32 bprob, n_used;
01287 
01288         /* Skip filler nodes in traversal. */
01289         if (dict_filler_word(ps_search_dict(search), link->from->basewid) && link->from != dag->start)
01290             continue;
01291         if (dict_filler_word(ps_search_dict(search), link->to->basewid) && link->to != dag->end)
01292             continue;
01293 
01294         /* Sanity check, we should not be traversing edges that
01295          * weren't previously updated, otherwise nasty overflows will result. */
01296         assert(link->path_scr != MAX_NEG_INT32);
01297 
01298         /* Calculate common bigram probability for all alphas. */
01299         if (lmset)
01300             bprob = ngram_ng_prob(lmset,
01301                                   link->to->basewid,
01302                                   &link->from->basewid, 1, &n_used);
01303         else
01304             bprob = 0;
01305         /* Add in this link's acoustic score, which was a constant
01306            factor in previous computations (if any). */
01307         link->alpha += (link->ascr << SENSCR_SHIFT) * ascale;
01308 
01309         /* Update scores for all paths exiting link->to. */
01310         for (x = link->to->exits; x; x = x->next) {
01311             int32 tscore, score;
01312 
01313             /* Skip links to filler words in update. */
01314             if (dict_filler_word(ps_search_dict(search), x->link->to->basewid)
01315                 && x->link->to != dag->end)
01316                 continue;
01317 
01318             /* Update alpha with sum of previous alphas. */
01319             x->link->alpha = logmath_add(lmath, x->link->alpha, link->alpha + bprob);
01320             /* Calculate trigram score for bestpath. */
01321             if (lmset)
01322                 tscore = (ngram_tg_score(lmset, x->link->to->basewid,
01323                                         link->to->basewid,
01324                                         link->from->basewid, &n_used) >> SENSCR_SHIFT)
01325                     * lwf;
01326             else
01327                 tscore = 0;
01328             /* Update link score with maximum link score. */
01329             score = link->path_scr + tscore + x->link->ascr;
01330             if (score BETTER_THAN x->link->path_scr) {
01331                 x->link->path_scr = score;
01332                 x->link->best_prev = link;
01333             }
01334         }
01335     }
01336 
01337     /* Find best link entering final node, and calculate normalizer
01338      * for posterior probabilities. */
01339     bestend = NULL;
01340     bestescr = MAX_NEG_INT32;
01341 
01342     /* Normalizer is the alpha for the imaginary link exiting the
01343        final node. */
01344     dag->norm = logmath_get_zero(lmath);
01345     for (x = dag->end->entries; x; x = x->next) {
01346         int32 bprob, n_used;
01347 
01348         if (dict_filler_word(ps_search_dict(search), x->link->from->basewid))
01349             continue;
01350         if (lmset)
01351             bprob = ngram_ng_prob(lmset,
01352                                   x->link->to->basewid,
01353                                   &x->link->from->basewid, 1, &n_used);
01354         else
01355             bprob = 0;
01356         dag->norm = logmath_add(lmath, dag->norm, x->link->alpha + bprob);
01357         if (x->link->path_scr BETTER_THAN bestescr) {
01358             bestescr = x->link->path_scr;
01359             bestend = x->link;
01360         }
01361     }
01362     /* FIXME: floating point... */
01363     dag->norm += (int32)(dag->final_node_ascr << SENSCR_SHIFT) * ascale;
01364 
01365     E_INFO("Normalizer P(O) = alpha(%s:%d:%d) = %d\n",
01366            dict_wordstr(dag->search->dict, dag->end->wid),
01367            dag->end->sf, dag->end->lef,
01368            dag->norm);
01369     return bestend;
01370 }

01372 static int32
01373 ps_lattice_joint(ps_lattice_t *dag, ps_latlink_t *link, float32 ascale)
01374 {
01375     ngram_model_t *lmset;
01376     int32 jprob;
01377 
01378     /* Sort of a hack... */
01379     if (dag->search && 0 == strcmp(ps_search_name(dag->search), "ngram"))
01380         lmset = ((ngram_search_t *)dag->search)->lmset;
01381     else
01382         lmset = NULL;
01383 
01384     jprob = (dag->final_node_ascr << SENSCR_SHIFT) * ascale;
01385     while (link) {
01386         if (lmset) {
01387             int lback;
01388             /* Compute unscaled language model probability.  Note that
01389                this is actually not the language model probability
01390                that corresponds to this link, but that is okay,
01391                because we are just taking the sum over all links in
01392                the best path. */
01393             jprob += ngram_ng_prob(lmset, link->to->basewid,
01394                                    &link->from->basewid, 1, &lback);
01395         }
01396         /* If there is no language model, we assume that the language
01397            model probability (such as it is) has been included in the
01398            link score. */
01399         jprob += (link->ascr << SENSCR_SHIFT) * ascale;
01400         link = link->best_prev;
01401     }
01402 
01403     E_INFO("Joint P(O,S) = %d P(S|O) = %d\n", jprob, jprob - dag->norm);
01404     return jprob;
01405 }

01407 int32
01408 ps_lattice_posterior(ps_lattice_t *dag, ngram_model_t *lmset,
01409                      float32 ascale)
01410 {
01411     ps_search_t *search;
01412     logmath_t *lmath;
01413     ps_latnode_t *node;
01414     ps_latlink_t *link;
01415     latlink_list_t *x;
01416     ps_latlink_t *bestend;
01417     int32 bestescr;
01418 
01419     search = dag->search;
01420     lmath = dag->lmath;
01421 
01422     /* Reset all betas to zero. */
01423     for (node = dag->nodes; node; node = node->next) {
01424         for (x = node->exits; x; x = x->next) {
01425             x->link->beta = logmath_get_zero(lmath);
01426         }
01427     }
01428 
01429     bestend = NULL;
01430     bestescr = MAX_NEG_INT32;
01431     /* Accumulate backward probabilities for all links. */
01432     for (link = ps_lattice_reverse_edges(dag, NULL, NULL);
01433          link; link = ps_lattice_reverse_next(dag, NULL)) {
01434         int32 bprob, n_used;
01435 
01436         /* Skip filler nodes in traversal. */
01437         if (dict_filler_word(ps_search_dict(search), link->from->basewid) && link->from != dag->start)
01438             continue;
01439         if (dict_filler_word(ps_search_dict(search), link->to->basewid) && link->to != dag->end)
01440             continue;
01441 
01442         /* Calculate LM probability. */
01443         if (lmset)
01444             bprob = ngram_ng_prob(lmset, link->to->basewid,
01445                                   &link->from->basewid, 1, &n_used);
01446         else
01447             bprob = 0;
01448 
01449         if (link->to == dag->end) {
01450             /* Track the best path - we will backtrace in order to
01451                calculate the unscaled joint probability for sentence
01452                posterior. */
01453             if (link->path_scr BETTER_THAN bestescr) {
01454                 bestescr = link->path_scr;
01455                 bestend = link;
01456             }
01457             /* Imaginary exit link from final node has beta = 1.0 */
01458             link->beta = bprob + (dag->final_node_ascr << SENSCR_SHIFT) * ascale;
01459         }
01460         else {
01461             /* Update beta from all outgoing betas. */
01462             for (x = link->to->exits; x; x = x->next) {
01463                 if (dict_filler_word(ps_search_dict(search), x->link->to->basewid) && x->link->to != dag->end)
01464                     continue;
01465                 link->beta = logmath_add(lmath, link->beta,
01466                                          x->link->beta + bprob
01467                                          + (x->link->ascr << SENSCR_SHIFT) * ascale);
01468             }
01469         }
01470     }
01471 
01472     /* Return P(S|O) = P(O,S)/P(O) */
01473     return ps_lattice_joint(dag, bestend, ascale) - dag->norm;
01474 }

01521 /* Parameters to prune n-best alternatives search */
01522 #define MAX_PATHS       500     /* Max allowed active paths at any time */
01523 #define MAX_HYP_TRIES   10000
01524 
01525 /*
01526  * For each node in any path between from and end of utt, find the
01527  * best score from "from".sf to end of utt.  (NOTE: Uses bigram probs;
01528  * this is an estimate of the best score from "from".)  (NOTE #2: yes,
01529  * this is the "heuristic score" used in A* search)
01530  */
01531 static int32
01532 best_rem_score(ps_astar_t *nbest, ps_latnode_t * from)
01533 {
01534     latlink_list_t *x;
01535     int32 bestscore, score;
01536 
01537     if (from->info.rem_score <= 0)
01538         return (from->info.rem_score);
01539 
01540     /* Best score from "from" to end of utt not known; compute from successors */
01541     bestscore = WORST_SCORE;
01542     for (x = from->exits; x; x = x->next) {
01543         int32 n_used;
01544 
01545         score = best_rem_score(nbest, x->link->to);
01546         score += x->link->ascr;
01547         if (nbest->lmset)
01548             score += (ngram_bg_score(nbest->lmset, x->link->to->basewid,
01549                                      from->basewid, &n_used) >> SENSCR_SHIFT)
01550                       * nbest->lwf;
01551         if (score BETTER_THAN bestscore)
01552             bestscore = score;
01553     }
01554     from->info.rem_score = bestscore;
01555 
01556     return bestscore;
01557 }
01558 
01559 /*
01560  * Insert newpath in sorted (by path score) list of paths.  But if newpath is
01561  * too far down the list, drop it (FIXME: necessary?)
01562  * total_score = path score (newpath) + rem_score to end of utt.
01563  */
01564 static void
01565 path_insert(ps_astar_t *nbest, ps_latpath_t *newpath, int32 total_score)
01566 {
01567     ps_latpath_t *prev, *p;
01568     int32 i;
01569 
01570     prev = NULL;
01571     for (i = 0, p = nbest->path_list; (i < MAX_PATHS) && p; p = p->next, i++) {
01572         if ((p->score + p->node->info.rem_score) < total_score)
01573             break;
01574         prev = p;
01575     }
01576 
01577     /* newpath should be inserted between prev and p */
01578     if (i < MAX_PATHS) {
01579         /* Insert new partial hyp */
01580         newpath->next = p;
01581         if (!prev)
01582             nbest->path_list = newpath;
01583         else
01584             prev->next = newpath;
01585         if (!p)
01586             nbest->path_tail = newpath;
01587 
01588         nbest->n_path++;
01589         nbest->n_hyp_insert++;
01590         nbest->insert_depth += i;
01591     }
01592     else {
01593         /* newpath score too low; reject it and also prune paths beyond MAX_PATHS */
01594         nbest->path_tail = prev;
01595         prev->next = NULL;
01596         nbest->n_path = MAX_PATHS;
01597         listelem_free(nbest->latpath_alloc, newpath);
01598 
01599         nbest->n_hyp_reject++;
01600         for (; p; p = newpath) {
01601             newpath = p->next;
01602             listelem_free(nbest->latpath_alloc, p);
01603             nbest->n_hyp_reject++;
01604         }
01605     }
01606 }

01608 /* Find all possible extensions to given partial path */
01609 static void
01610 path_extend(ps_astar_t *nbest, ps_latpath_t * path)
01611 {
01612     latlink_list_t *x;
01613     ps_latpath_t *newpath;
01614     int32 total_score, tail_score;
01615 
01616     /* Consider all successors of path->node */
01617     for (x = path->node->exits; x; x = x->next) {
01618         int32 n_used;
01619 
01620         /* Skip successor if no path from it reaches the final node */
01621         if (x->link->to->info.rem_score <= WORST_SCORE)
01622             continue;
01623 
01624         /* Create path extension and compute exact score for this extension */
01625         newpath = listelem_malloc(nbest->latpath_alloc);
01626         newpath->node = x->link->to;
01627         newpath->parent = path;
01628         newpath->score = path->score + x->link->ascr;
01629         if (nbest->lmset) {
01630             if (path->parent) {
01631                 newpath->score += nbest->lwf
01632                     * (ngram_tg_score(nbest->lmset, newpath->node->basewid,
01633                                       path->node->basewid,
01634                                       path->parent->node->basewid, &n_used)
01635                        >> SENSCR_SHIFT);
01636             }
01637             else 
01638                 newpath->score += nbest->lwf
01639                     * (ngram_bg_score(nbest->lmset, newpath->node->basewid,
01640                                       path->node->basewid, &n_used)
01641                        >> SENSCR_SHIFT);
01642         }
01643 
01644         /* Insert new partial path hypothesis into sorted path_list */
01645         nbest->n_hyp_tried++;
01646         total_score = newpath->score + newpath->node->info.rem_score;
01647 
01648         /* First see if hyp would be worse than the worst */
01649         if (nbest->n_path >= MAX_PATHS) {
01650             tail_score =
01651                 nbest->path_tail->score
01652                 + nbest->path_tail->node->info.rem_score;
01653             if (total_score < tail_score) {
01654                 listelem_free(nbest->latpath_alloc, newpath);
01655                 nbest->n_hyp_reject++;
01656                 continue;
01657             }
01658         }
01659 
01660         path_insert(nbest, newpath, total_score);
01661     }
01662 }

01664 ps_astar_t *
01665 ps_astar_start(ps_lattice_t *dag,
01666                   ngram_model_t *lmset,
01667                   float32 lwf,
01668                   int sf, int ef,
01669                   int w1, int w2)
01670 {
01671     ps_astar_t *nbest;
01672     ps_latnode_t *node;
01673 
01674     nbest = ckd_calloc(1, sizeof(*nbest));
01675     nbest->dag = dag;
01676     nbest->lmset = lmset;
01677     nbest->lwf = lwf;
01678     nbest->sf = sf;
01679     if (ef < 0)
01680         nbest->ef = dag->n_frames + 1;
01681     else
01682         nbest->ef = ef;
01683     nbest->w1 = w1;
01684     nbest->w2 = w2;
01685     nbest->latpath_alloc = listelem_alloc_init(sizeof(ps_latpath_t));
01686 
01687     /* Initialize rem_score (A* heuristic) to default values */
01688     for (node = dag->nodes; node; node = node->next) {
01689         if (node == dag->end)
01690             node->info.rem_score = 0;
01691         else if (node->exits == NULL)
01692             node->info.rem_score = WORST_SCORE;
01693         else
01694             node->info.rem_score = 1;   /* +ve => unknown value */
01695     }
01696 
01697     /* Create initial partial hypotheses list consisting of nodes starting at sf */
01698     nbest->path_list = nbest->path_tail = NULL;
01699     for (node = dag->nodes; node; node = node->next) {
01700         if (node->sf == sf) {
01701             ps_latpath_t *path;
01702             int32 n_used;
01703 
01704             best_rem_score(nbest, node);
01705             path = listelem_malloc(nbest->latpath_alloc);
01706             path->node = node;
01707             path->parent = NULL;
01708             if (nbest->lmset)
01709                 path->score = nbest->lwf *
01710                     ((w1 < 0)
01711                     ? ngram_bg_score(nbest->lmset, node->basewid, w2, &n_used)
01712                     : ngram_tg_score(nbest->lmset, node->basewid, w2, w1, &n_used));
01713             else
01714                 path->score = 0;
01715             path->score >>= SENSCR_SHIFT;
01716             path_insert(nbest, path, path->score + node->info.rem_score);
01717         }
01718     }
01719 
01720     return nbest;
01721 }

01723 ps_latpath_t *
01724 ps_astar_next(ps_astar_t *nbest)
01725 {
01726     ps_lattice_t *dag;
01727 
01728     dag = nbest->dag;
01729 
01730     /* Pop the top (best) partial hypothesis */
01731     while ((nbest->top = nbest->path_list) != NULL) {
01732         nbest->path_list = nbest->path_list->next;
01733         if (nbest->top == nbest->path_tail)
01734             nbest->path_tail = NULL;
01735         nbest->n_path--;
01736 
01737         /* Complete hypothesis? */
01738         if ((nbest->top->node->sf >= nbest->ef)
01739             || ((nbest->top->node == dag->end) &&
01740                 (nbest->ef > dag->end->sf))) {
01741             /* FIXME: Verify that it is non-empty.  Also we may want
01742              * to verify that it is actually distinct from other
01743              * paths, since often this is not the case*/
01744             return nbest->top;
01745         }
01746         else {
01747             if (nbest->top->node->fef < nbest->ef)
01748                 path_extend(nbest, nbest->top);
01749         }
01750     }
01751 
01752     /* Did not find any more paths to extend. */
01753     return NULL;
01754 }

01758 {
01759     ps_search_t *search;
01760     ps_latpath_t *p;
01761     size_t len;
01762     char *c;
01763     char *hyp;
01764 
01765     search = nbest->dag->search;
01766 
01767     /* Backtrace once to get hypothesis length. */
01768     len = 0;

01770         if (dict_real_word(ps_search_dict(search), p->node->basewid)) {
01771             char *wstr = dict_wordstr(ps_search_dict(search), p->node->basewid);
01772             if (wstr != NULL)
01773                 len += strlen(wstr) + 1;
01774         }
01775     }
01776 
01777     if (len == 0) {
01778         return NULL;
01779     }
01780 
01781     /* Backtrace again to construct hypothesis string. */
01782     hyp = ckd_calloc(1, len);
01783     c = hyp + len - 1;
01784     for (p = path; p; p = p->parent) {
01785         if (dict_real_word(ps_search_dict(search), p->node->basewid)) {
01786             char *wstr = dict_wordstr(ps_search_dict(search), p->node->basewid);
01787             if (wstr != NULL) {
01788                 len = strlen(wstr);
01789                 c -= len;
01790                 memcpy(c, wstr, len);
01791                 if (c > hyp) {
01792                     --c;
01793                     *c = ' ';
01794                 }
01795             }
01796         }
01797     }
01798 
01799     nbest->hyps = glist_add_ptr(nbest->hyps, hyp);
01800     return hyp;
01801 }
