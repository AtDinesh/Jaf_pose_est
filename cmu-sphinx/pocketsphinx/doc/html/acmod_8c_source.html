00567         acmod->mfc_buf = ckd_calloc_2d(nfr + 1, fe_get_output_size(acmod->fe),
00568                                        sizeof(**acmod->mfc_buf));
00569         acmod->n_mfc_alloc = nfr + 1;
00570     }
00571     acmod->n_mfc_frame = 0;
00572     acmod->mfc_outidx = 0;
00573     fe_start_utt(acmod->fe);
00574     if (fe_process_frames(acmod->fe, inout_raw, inout_n_samps,
00575                           acmod->mfc_buf, &nfr) < 0)
00576         return -1;
00577     fe_end_utt(acmod->fe, acmod->mfc_buf[nfr], &ntail);
00578     nfr += ntail;
00579 
00580     cepptr = acmod->mfc_buf;
00581     nfr = acmod_process_full_cep(acmod, &cepptr, &nfr);
00582     acmod->n_mfc_frame = 0;
00583     return nfr;
00584 }

00589 static int32
00590 acmod_process_mfcbuf(acmod_t *acmod)
00591 {
00592     mfcc_t **mfcptr;
00593     int32 ncep;
00594 
00595     ncep = acmod->n_mfc_frame;
00596     /* Also do this in two parts because of the circular mfc_buf. */
00597     if (acmod->mfc_outidx + ncep > acmod->n_mfc_alloc) {
00598         int32 ncep1 = acmod->n_mfc_alloc - acmod->mfc_outidx;
00599         int saved_state = acmod->state;
00600 
00601         /* Make sure we don't end the utterance here. */
00602         if (acmod->state == ACMOD_ENDED)
00603             acmod->state = ACMOD_PROCESSING;
00604         mfcptr = acmod->mfc_buf + acmod->mfc_outidx;
00605         ncep1 = acmod_process_cep(acmod, &mfcptr, &ncep1, FALSE);
00606         /* It's possible that not all available frames were filled. */
00607         ncep -= ncep1;
00608         acmod->n_mfc_frame -= ncep1;
00609         acmod->mfc_outidx += ncep1;
00610         acmod->mfc_outidx %= acmod->n_mfc_alloc;
00611         /* Restore original state (could this really be the end) */
00612         acmod->state = saved_state;
00613     }
00614     mfcptr = acmod->mfc_buf + acmod->mfc_outidx;
00615     ncep = acmod_process_cep(acmod, &mfcptr, &ncep, FALSE);
00616     acmod->n_mfc_frame -= ncep;
00617     acmod->mfc_outidx += ncep;
00618     acmod->mfc_outidx %= acmod->n_mfc_alloc;
00619     return ncep;
00620 }

00622 int
00623 acmod_process_raw(acmod_t *acmod,
00624                   int16 const **inout_raw,
00625                   size_t *inout_n_samps,
00626                   int full_utt)
00627 {
00628     int32 ncep;
00629 
00630     /* If this is a full utterance, process it all at once. */
00631     if (full_utt)
00632         return acmod_process_full_raw(acmod, inout_raw, inout_n_samps);
00633 
00634     /* Append MFCCs to the end of any that are previously in there
00635      * (in practice, there will probably be none) */
00636     if (inout_n_samps && *inout_n_samps) {
00637         int16 const *prev_audio_inptr = *inout_raw;
00638         int inptr;
00639 
00640         /* Total number of frames available. */
00641         ncep = acmod->n_mfc_alloc - acmod->n_mfc_frame;
00642         /* Where to start writing them (circular buffer) */
00643         inptr = (acmod->mfc_outidx + acmod->n_mfc_frame) % acmod->n_mfc_alloc;
00644 
00645         /* Write them in two (or more) parts if there is wraparound. */
00646         while (inptr + ncep > acmod->n_mfc_alloc) {
00647             int32 ncep1 = acmod->n_mfc_alloc - inptr;
00648             if (fe_process_frames(acmod->fe, inout_raw, inout_n_samps,
00649                                   acmod->mfc_buf + inptr, &ncep1) < 0)
00650                 return -1;
00651             /* Write to logging file if any. */
00652             if (acmod->rawfh) {
00653                 fwrite(prev_audio_inptr, 2,
00654                        *inout_raw - prev_audio_inptr,
00655                        acmod->rawfh);
00656                 prev_audio_inptr = *inout_raw;
00657             }
00658             /* ncep1 now contains the number of frames actually
00659              * processed.  This is a good thing, but it means we
00660              * actually still might have some room left at the end of
00661              * the buffer, hence the while loop.  Unfortunately it
00662              * also means that in the case where we are really
00663              * actually done, we need to get out totally, hence the
00664              * goto. */
00665             acmod->n_mfc_frame += ncep1;
00666             ncep -= ncep1;
00667             inptr += ncep1;
00668             inptr %= acmod->n_mfc_alloc;
00669             if (ncep1 == 0)
00670                 goto alldone;
00671         }
00672         assert(inptr + ncep <= acmod->n_mfc_alloc);
00673         if (fe_process_frames(acmod->fe, inout_raw, inout_n_samps,
00674                               acmod->mfc_buf + inptr, &ncep) < 0)
00675             return -1;
00676         /* Write to logging file if any. */
00677         if (acmod->rawfh) {
00678             fwrite(prev_audio_inptr, 2,
00679                    *inout_raw - prev_audio_inptr, acmod->rawfh);
00680             prev_audio_inptr = *inout_raw;
00681         }
00682         acmod->n_mfc_frame += ncep;
00683     alldone:
00684         ;
00685     }
00686 
00687     /* Hand things off to acmod_process_cep. */
00688     return acmod_process_mfcbuf(acmod);
00689 }

00691 int
00692 acmod_process_cep(acmod_t *acmod,
00693                   mfcc_t ***inout_cep,
00694                   int *inout_n_frames,
00695                   int full_utt)
00696 {
00697     int32 nfeat, ncep, inptr;
00698     int orig_n_frames;
00699 
00700     /* If this is a full utterance, process it all at once. */
00701     if (full_utt)
00702         return acmod_process_full_cep(acmod, inout_cep, inout_n_frames);
00703 
00704     /* Write to log file. */
00705     if (acmod->mfcfh)
00706         acmod_log_mfc(acmod, *inout_cep, *inout_n_frames);
00707 
00708     /* Maximum number of frames we're going to generate. */
00709     orig_n_frames = ncep = nfeat = *inout_n_frames;
00710 
00711     /* FIXME: This behaviour isn't guaranteed... */
00712     if (acmod->state == ACMOD_ENDED)
00713         nfeat += feat_window_size(acmod->fcb);
00714     else if (acmod->state == ACMOD_STARTED)
00715         nfeat -= feat_window_size(acmod->fcb);
00716 
00717     /* Clamp number of features to fit available space. */
00718     if (nfeat > acmod->n_feat_alloc - acmod->n_feat_frame) {
00719         /* Grow it as needed - we have to grow it at the end of an
00720          * utterance because we can't return a short read there. */
00721         if (acmod->grow_feat || acmod->state == ACMOD_ENDED)
00722             acmod_grow_feat_buf(acmod, acmod->n_feat_alloc + nfeat);
00723         else
00724             ncep -= (nfeat - (acmod->n_feat_alloc - acmod->n_feat_frame));
00725     }
00726 
00727     /* Where to start writing in the feature buffer. */
00728     if (acmod->grow_feat) {
00729         /* Grow to avoid wraparound if grow_feat == TRUE. */
00730         inptr = acmod->feat_outidx + acmod->n_feat_frame;
00731         while (inptr + nfeat >= acmod->n_feat_alloc)
00732             acmod_grow_feat_buf(acmod, acmod->n_feat_alloc * 2);
00733     }
00734     else {
00735         inptr = (acmod->feat_outidx + acmod->n_feat_frame) % acmod->n_feat_alloc;
00736     }
00737 
00738 
00739     /* FIXME: we can't split the last frame drop properly to be on the bounary, so just return */
00740     if (inptr + nfeat > acmod->n_feat_alloc && acmod->state == ACMOD_ENDED) {
00741         *inout_n_frames -= ncep;
00742         *inout_cep += ncep;
00743         return 0;
00744     }
00745 
00746     /* Write them in two parts if there is wraparound. */
00747     if (inptr + nfeat > acmod->n_feat_alloc) {
00748         int32 ncep1 = acmod->n_feat_alloc - inptr;
00749 
00750         /* Make sure we don't end the utterance here. */
00751         nfeat = feat_s2mfc2feat_live(acmod->fcb, *inout_cep,
00752                                      &ncep1,
00753                                      (acmod->state == ACMOD_STARTED),
00754                                      FALSE,
00755                                      acmod->feat_buf + inptr);
00756         if (nfeat < 0)
00757             return -1;
00758         /* Move the output feature pointer forward. */
00759         acmod->n_feat_frame += nfeat;
00760         assert(acmod->n_feat_frame <= acmod->n_feat_alloc);
00761         inptr += nfeat;
00762         inptr %= acmod->n_feat_alloc;
00763         /* Move the input feature pointers forward. */
00764         *inout_n_frames -= ncep1;
00765         *inout_cep += ncep1;
00766         ncep -= ncep1;
00767     }
00768 
00769     nfeat = feat_s2mfc2feat_live(acmod->fcb, *inout_cep,
00770                                  &ncep,
00771                                  (acmod->state == ACMOD_STARTED),
00772                                  (acmod->state == ACMOD_ENDED),
00773                                  acmod->feat_buf + inptr);
00774     if (nfeat < 0)
00775         return -1;
00776     acmod->n_feat_frame += nfeat;
00777     assert(acmod->n_feat_frame <= acmod->n_feat_alloc);
00778     /* Move the input feature pointers forward. */
00779     *inout_n_frames -= ncep;
00780     *inout_cep += ncep;
00781     if (acmod->state == ACMOD_STARTED)
00782         acmod->state = ACMOD_PROCESSING;
00783     return orig_n_frames - *inout_n_frames;
00784 }

00786 int
00787 acmod_process_feat(acmod_t *acmod,
00788                    mfcc_t **feat)
00789 {
00790     int i, inptr;
00791 
00792     if (acmod->n_feat_frame == acmod->n_feat_alloc) {
00793         if (acmod->grow_feat)
00794             acmod_grow_feat_buf(acmod, acmod->n_feat_alloc * 2);
00795         else
00796             return 0;
00797     }
00798 
00799     if (acmod->grow_feat) {
00800         /* Grow to avoid wraparound if grow_feat == TRUE. */
00801         inptr = acmod->feat_outidx + acmod->n_feat_frame;
00802         while (inptr + 1 >= acmod->n_feat_alloc)
00803             acmod_grow_feat_buf(acmod, acmod->n_feat_alloc * 2);
00804     }
00805     else {
00806         inptr = (acmod->feat_outidx + acmod->n_feat_frame) % acmod->n_feat_alloc;
00807     }
00808     for (i = 0; i < feat_dimension1(acmod->fcb); ++i)
00809         memcpy(acmod->feat_buf[inptr][i],
00810                feat[i], feat_dimension2(acmod->fcb, i) * sizeof(**feat));
00811     ++acmod->n_feat_frame;
00812     assert(acmod->n_feat_frame <= acmod->n_feat_alloc);
00813 
00814     return 1;
00815 }

00897 int
00898 acmod_write_scores(acmod_t *acmod, int n_active, uint8 const *active,
00899                    int16 const *senscr, FILE *senfh)
00900 {
00901     int16 n_active2;
00902 
00903     /* Uncompressed frame format:
00904      *
00905      * (2 bytes) n_active: Number of active senones
00906      * If all senones active:
00907      * (n_active * 2 bytes) scores of active senones
00908      *
00909      * Otherwise:
00910      * (2 bytes) n_active: Number of active senones
00911      * (n_active bytes) deltas to active senones
00912      * (n_active * 2 bytes) scores of active senones
00913      */
00914     n_active2 = n_active;
00915     if (fwrite(&n_active2, 2, 1, senfh) != 1)
00916         goto error_out;
00917     if (n_active == bin_mdef_n_sen(acmod->mdef)) {
00918         if (fwrite(senscr, 2, n_active, senfh) != n_active)
00919             goto error_out;
00920     }
00921     else {
00922         int i, n;
00923         if (fwrite(active, 1, n_active, senfh) != n_active)
00924             goto error_out;
00925         for (i = n = 0; i < n_active; ++i) {
00926             n += active[i];
00927             if (fwrite(senscr + n, 2, 1, senfh) != 1)
00928                 goto error_out;
00929         }
00930     }
00931     return 0;
00932 error_out:
00933     E_ERROR_SYSTEM("Failed to write frame to senone file");
00934     return -1;
00935 }

00940 static int
00941 acmod_read_scores_internal(acmod_t *acmod)
00942 {
00943     FILE *senfh = acmod->insenfh;
00944     int16 n_active;
00945     int rv;
00946 
00947     if (acmod->n_feat_frame == acmod->n_feat_alloc) {
00948         if (acmod->grow_feat)
00949             acmod_grow_feat_buf(acmod, acmod->n_feat_alloc * 2);
00950         else
00951             return 0;
00952     }
00953 
00954     if (senfh == NULL)
00955         return -1;
00956     if ((rv = fread(&n_active, 2, 1, senfh)) < 0)
00957         goto error_out;
00958     else if (rv == 0)
00959         return 0;
00960 
00961     acmod->n_senone_active = n_active;
00962     if (acmod->n_senone_active == bin_mdef_n_sen(acmod->mdef)) {
00963         if ((rv = fread(acmod->senone_scores, 2,
00964                         acmod->n_senone_active, senfh)) < 0)
00965             goto error_out;
00966         else if (rv != acmod->n_senone_active)
00967             return 0;
00968     }
00969     else {
00970         int i, n;
00971         if ((rv = fread(acmod->senone_active, 1,
00972                         acmod->n_senone_active, senfh)) < 0)
00973             goto error_out;
00974         else if (rv != acmod->n_senone_active)
00975             return 0;
00976         for (i = 0, n = 0; i < acmod->n_senone_active; ++i) {
00977             int j, sen = n + acmod->senone_active[i];
00978             for (j = n + 1; j < sen; ++j)
00979                 acmod->senone_scores[j] = SENSCR_DUMMY;
00980             if ((rv = fread(acmod->senone_scores + sen, 2, 1, senfh)) < 0)
00981                 goto error_out;
00982             else if (rv == 0)
00983                 return 0;
00984             n = sen;
00985         }
00986         ++n;
00987         while (n < bin_mdef_n_sen(acmod->mdef))
00988             acmod->senone_scores[n++] = SENSCR_DUMMY;
00989     }
00990     return 1;
00991 error_out:
00992     E_ERROR_SYSTEM("Failed to read frame from senone file");
00993     return -1;
00994 }

00996 int
00997 acmod_read_scores(acmod_t *acmod)
00998 {
00999     int inptr, rv;
01000 
01001     if (acmod->grow_feat) {
01002         /* Grow to avoid wraparound if grow_feat == TRUE. */
01003         inptr = acmod->feat_outidx + acmod->n_feat_frame;
01004         /* Has to be +1, otherwise, next time acmod_advance() is
01005          * called, this will wrap around. */
01006         while (inptr + 1 >= acmod->n_feat_alloc)
01007             acmod_grow_feat_buf(acmod, acmod->n_feat_alloc * 2);
01008     }
01009     else {
01010         inptr = (acmod->feat_outidx + acmod->n_feat_frame) % acmod->n_feat_alloc;
01011     }
01012 
01013     if ((rv = acmod_read_scores_internal(acmod)) != 1)
01014         return rv;
01015 
01016     /* Set acmod->senscr_frame appropriately so that these scores
01017        get reused below in acmod_score(). */
01018     acmod->senscr_frame = acmod->output_frame + acmod->n_feat_frame;
01019 
01020     E_DEBUG(1,("Frame %d has %d active states\n",
01021                acmod->senscr_frame, acmod->n_senone_active));
01022 
01023     /* Increment the "feature frame counter" and record the file
01024      * position for the relevant frame in the (possibly circular)
01025      * buffer. */
01026     ++acmod->n_feat_frame;
01027     acmod->framepos[inptr] = ftell(acmod->insenfh);
01028 
01029     return 1;
01030 }

01032 static int
01033 calc_frame_idx(acmod_t *acmod, int *inout_frame_idx)
01034 {
01035     int frame_idx;
01036 
01037     /* Calculate the absolute frame index to be scored. */
01038     if (inout_frame_idx == NULL)
01039         frame_idx = acmod->output_frame;
01040     else if (*inout_frame_idx < 0)
01041         frame_idx = acmod->output_frame + 1 + *inout_frame_idx;
01042     else
01043         frame_idx = *inout_frame_idx;
01044

01048 static int
01049 calc_feat_idx(acmod_t *acmod, int frame_idx)
01050 {
01051     int n_backfr, feat_idx;
01052 
01053     n_backfr = acmod->n_feat_alloc - acmod->n_feat_frame;
01054     if (frame_idx < 0 || acmod->output_frame - frame_idx > n_backfr) {
01055         E_ERROR("Frame %d outside queue of %d frames, %d alloc (%d > %d), cannot score\n",
01056                 frame_idx, acmod->n_feat_frame, acmod->n_feat_alloc,
01057                 acmod->output_frame - frame_idx, n_backfr);
01058         return -1;
01059     }
01060 
01061     /* Get the index in feat_buf/framepos of the frame to be scored. */
01062     feat_idx = ((acmod->feat_outidx + frame_idx - acmod->output_frame)
01063                 % acmod->n_feat_alloc);
01064     if (feat_idx < 0) feat_idx += acmod->n_feat_alloc;
01065 
01066     return feat_idx;
01067 }

01070 acmod_get_frame(acmod_t *acmod, int *inout_frame_idx)

01075     frame_idx = calc_frame_idx(acmod, inout_frame_idx);

01087 int16 const *
01088 acmod_score(acmod_t *acmod, int *inout_frame_idx)
01089 {
01090     int frame_idx, feat_idx;
01091 
01092     /* Calculate the absolute frame index to be scored. */
01093     frame_idx = calc_frame_idx(acmod, inout_frame_idx);
01094 
01095     /* If all senones are being computed, or we are using a senone file,
01096        then we can reuse existing scores. */
01097     if ((acmod->compallsen || acmod->insenfh)
01098         && frame_idx == acmod->senscr_frame) {
01099         if (inout_frame_idx)
01100             *inout_frame_idx = frame_idx;
01101         return acmod->senone_scores;
01102     }
01103 
01104     /* Calculate position of requested frame in circular buffer. */
01105     if ((feat_idx = calc_feat_idx(acmod, frame_idx)) < 0)
01106         return NULL;
01107 
01108     /* If there is an input senone file locate the appropriate frame and read it. */
01109     if (acmod->insenfh) {
01110         fseek(acmod->insenfh, acmod->framepos[feat_idx], SEEK_SET);
01111         if (acmod_read_scores_internal(acmod) < 0)
01112             return NULL;
01113     }
01114     else {
01115         /* Build active senone list. */
01116         acmod_flags2list(acmod);
01117 
01118         /* Generate scores for the next available frame */
01119         ps_mgau_frame_eval(acmod->mgau,
01120                            acmod->senone_scores,
01121                            acmod->senone_active,
01122                            acmod->n_senone_active,
01123                            acmod->feat_buf[feat_idx],
01124                            frame_idx,
01125                            acmod->compallsen);
01126     }
01127 
01128     if (inout_frame_idx)
01129         *inout_frame_idx = frame_idx;
01130     acmod->senscr_frame = frame_idx;
01131 
01132     /* Dump scores to the senone dump file if one exists. */
01133     if (acmod->senfh) {
01134         if (acmod_write_scores(acmod, acmod->n_senone_active,
01135                                acmod->senone_active,

01231 int32
01232 acmod_flags2list(acmod_t *acmod)
01233 {
01234     int32 w, l, n, b, total_dists, total_words, extra_bits;
01235     bitvec_t *flagptr;
01236 
01237     total_dists = bin_mdef_n_sen(acmod->mdef);
01238     if (acmod->compallsen) {
01239         acmod->n_senone_active = total_dists;
01240         return total_dists;
01241     }
01242     total_words = total_dists / BITVEC_BITS;
01243     extra_bits = total_dists % BITVEC_BITS;
01244     w = n = l = 0;
01245     for (flagptr = acmod->senone_active_vec; w < total_words; ++w, ++flagptr) {
01246         if (*flagptr == 0)
01247             continue;
01248         for (b = 0; b < BITVEC_BITS; ++b) {
01249             if (*flagptr & (1UL << b)) {
01250                 int32 sen = w * BITVEC_BITS + b;
01251                 int32 delta = sen - l;
01252                 /* Handle excessive deltas "lossily" by adding a few
01253                    extra senones to bridge the gap. */
01254                 while (delta > 255) {
01255                     acmod->senone_active[n++] = 255;
01256                     delta -= 255;
01257                 }
01258                 acmod->senone_active[n++] = delta;
01259                 l = sen;
01260             }
01261         }
01262     }
01263 
01264     for (b = 0; b < extra_bits; ++b) {
01265         if (*flagptr & (1UL << b)) {
01266             int32 sen = w * BITVEC_BITS + b;
01267             int32 delta = sen - l;
01268             /* Handle excessive deltas "lossily" by adding a few
01269                extra senones to bridge the gap. */
01270             while (delta > 255) {
01271                 acmod->senone_active[n++] = 255;
01272                 delta -= 255;
01273             }
01274             acmod->senone_active[n++] = delta;
01275             l = sen;
01276         }
01277     }
01278 
01279     acmod->n_senone_active = n;
01280     E_DEBUG(1, ("acmod_flags2list: %d active in frame %d\n",
01281                 acmod->n_senone_active, acmod->output_frame));
01282     return n;
