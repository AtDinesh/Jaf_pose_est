00075 static void
00076 eval_topn(s2_semi_mgau_t *s, int32 feat, mfcc_t *z)
00077 {
00078     int i, ceplen;
00079     vqFeature_t *topn;
00080 
00081     topn = s->f[feat];
00082     ceplen = s->veclen[feat];
00083 
00084     for (i = 0; i < s->max_topn; i++) {
00085         mfcc_t *mean, diff, sqdiff, compl; /* diff, diff^2, component likelihood */
00086         vqFeature_t vtmp;
00087         mfcc_t *var, d;
00088         mfcc_t *obs;
00089         int32 cw, j;
00090 
00091         cw = topn[i].codeword;
00092         mean = s->means[feat][0] + cw * ceplen;
00093         var = s->vars[feat][0] + cw * ceplen;
00094         d = s->dets[feat][cw];
00095         obs = z;
00096         for (j = 0; j < ceplen; j++) {
00097             diff = *obs++ - *mean++;
00098             sqdiff = MFCCMUL(diff, diff);
00099             compl = MFCCMUL(sqdiff, *var);
00100             d = GMMSUB(d, compl);
00101             ++var;
00102         }
00103         topn[i].score = (int32)d;
00104         if (i == 0)
00105             continue;
00106         vtmp = topn[i];
00107         for (j = i - 1; j >= 0 && (int32)d > topn[j].score; j--) {
00108             topn[j + 1] = topn[j];
00109         }
00110         topn[j + 1] = vtmp;
00111     }
00112 }

00114 static void
00115 eval_cb(s2_semi_mgau_t *s, int32 feat, mfcc_t *z)
00116 {
00117     vqFeature_t *worst, *best, *topn;
00118     mfcc_t *mean;
00119     mfcc_t *var, *det, *detP, *detE;
00120     int32 i, ceplen;
00121 
00122     best = topn = s->f[feat];
00123     worst = topn + (s->max_topn - 1);
00124     mean = s->means[feat][0];
00125     var = s->vars[feat][0];
00126     det = s->dets[feat];
00127     detE = det + s->n_density;
00128     ceplen = s->veclen[feat];
00129 
00130     for (detP = det; detP < detE; ++detP) {
00131         mfcc_t diff, sqdiff, compl; /* diff, diff^2, component likelihood */
00132         mfcc_t d;
00133         mfcc_t *obs;
00134         vqFeature_t *cur;
00135         int32 cw, j;
00136 
00137         d = *detP;
00138         obs = z;
00139         cw = detP - det;
00140         for (j = 0; (j < ceplen) && (d >= worst->score); ++j) {
00141             diff = *obs++ - *mean++;
00142             sqdiff = MFCCMUL(diff, diff);
00143             compl = MFCCMUL(sqdiff, *var);
00144             d = GMMSUB(d, compl);
00145             ++var;
00146         }
00147         if (j < ceplen) {
00148             /* terminated early, so not in topn */
00149             mean += (ceplen - j);
00150             var += (ceplen - j);
00151             continue;
00152         }
00153         if ((int32)d < worst->score)
00154             continue;
00155         for (i = 0; i < s->max_topn; i++) {
00156             /* already there, so don't need to insert */
00157             if (topn[i].codeword == cw)
00158                 break;
00159         }
00160         if (i < s->max_topn)
00161             continue;       /* already there.  Don't insert */
00162         /* remaining code inserts codeword and dist in correct spot */
00163         for (cur = worst - 1; cur >= best && (int32)d >= cur->score; --cur)
00164             memcpy(cur + 1, cur, sizeof(vqFeature_t));
00165         ++cur;
00166         cur->codeword = cw;
00167         cur->score = (int32)d;
00168     }
00169 }

00184 static int
00185 mgau_norm(s2_semi_mgau_t *s, int feat)
00186 {
00187     int32 norm;
00188     int j;
00189 
00190     /* Compute quantized normalizing constant. */
00191     norm = s->f[feat][0].score >> SENSCR_SHIFT;
00192 
00193     /* Normalize the scores, negate them, and clamp their dynamic range. */
00194     for (j = 0; j < s->max_topn; ++j) {
00195         s->f[feat][j].score = -((s->f[feat][j].score >> SENSCR_SHIFT) - norm);
00196         if (s->f[feat][j].score > MAX_NEG_ASCR)
00197             s->f[feat][j].score = MAX_NEG_ASCR;
00198         if (s->topn_beam[feat] && s->f[feat][j].score > s->topn_beam[feat])
00199             break;
00200     }
00201     return j;
00202 }

01005                 n_sen, s->n_sen);
01006         goto error_out;
01007     }
01008 
01009     if (!((n_clust == 0) || (n_clust == 15) || (n_clust == 16))) {
01010         E_ERROR("Cluster count must be 0, 15, or 16\n");
01011         goto error_out;
01012     }
01013     if (n_clust == 15)
01014         ++n_clust;
01015 
01016     if (!((n_bits == 8) || (n_bits == 4))) {
01017         E_ERROR("Cluster count must be 4 or 8\n");
01018         goto error_out;
01019     }
01020 
01021     if (do_mmap) {
01022             E_INFO("Using memory-mapped I/O for senones\n");
01023     }

01025 
01026     /* Allocate memory for pdfs (or memory map them) */
01027     if (do_mmap) {
01028         s->sendump_mmap = mmio_file_read(file);
01029         /* Get cluster codebook if any. */
01030         if (n_clust) {
01031             s->mixw_cb = ((uint8 *) mmio_file_ptr(s->sendump_mmap)) + offset;
01032             offset += n_clust;
01033         }
01034     }
01035     else {
01036         /* Get cluster codebook if any. */
01037         if (n_clust) {
01038             s->mixw_cb = ckd_calloc(1, n_clust);
01039             if (fread(s->mixw_cb, 1, n_clust, fp) != (size_t) n_clust) {
01040                 E_ERROR("Failed to read %d bytes from sendump\n", n_clust);
01041                 goto error_out;
01042             }
01043         }
01044     }
01045 
01046     /* Set up pointers, or read, or whatever */
01047     if (s->sendump_mmap) {
01048         s->mixw = ckd_calloc_2d(s->n_feat, n_density, sizeof(*s->mixw));
01049         for (n = 0; n < n_feat; n++) {
01050             int step = c;
01051             if (n_bits == 4)
01052                 step = (step + 1) / 2;
01053             for (i = 0; i < r; i++) {
01054                 s->mixw[n][i] = ((uint8 *) mmio_file_ptr(s->sendump_mmap)) + offset;
01055                 offset += step;
01056             }
01057         }
01058     }
01059     else {
01060         s->mixw = ckd_calloc_3d(n_feat, n_density, n_sen, sizeof(***s->mixw));
01061         /* Read pdf values and ids */
01062         for (n = 0; n < n_feat; n++) {
01063             int step = c;
01064             if (n_bits == 4)
01065                 step = (step + 1) / 2;
01066             for (i = 0; i < r; i++) {
01067                 if (fread(s->mixw[n][i], sizeof(***s->mixw), step, fp)
01068                     != (size_t) step) {
01069                     E_ERROR("Failed to read %d bytes from sendump\n", step);
01070                     goto error_out;
01071                 }
01072             }
01073         }
01074     }
01075 
01076     fclose(fp);
01077     return 0;
01078 error_out:
01079     fclose(fp);
01080     return -1;

01083 static int32
01084 read_mixw(s2_semi_mgau_t * s, char const *file_name, double SmoothMin)
01085 {
01086     char **argname, **argval;
01087     char eofchk;
01088     FILE *fp;
01089     int32 byteswap, chksum_present;
01090     uint32 chksum;
01091     float32 *pdf;
01092     int32 i, f, c, n;
01093     int32 n_sen;
01094     int32 n_feat;
01095     int32 n_comp;
01096     int32 n_err;
01097 
01098     E_INFO("Reading mixture weights file '%s'\n", file_name);
01099 
01100     if ((fp = fopen(file_name, "rb")) == NULL)
01101         E_FATAL_SYSTEM("Failed to open mixture weights file '%s' for reading", file_name);
01102 
01103     /* Read header, including argument-value info and 32-bit byteorder magic */
01104     if (bio_readhdr(fp, &argname, &argval, &byteswap) < 0)
01105         E_FATAL("Failed to read header from file '%s'\n", file_name);
01106 
01107     /* Parse argument-value list */
01108     chksum_present = 0;
01109     for (i = 0; argname[i]; i++) {
01110         if (strcmp(argname[i], "version") == 0) {
01111             if (strcmp(argval[i], MGAU_MIXW_VERSION) != 0)
01112                 E_WARN("Version mismatch(%s): %s, expecting %s\n",
01113                        file_name, argval[i], MGAU_MIXW_VERSION);
01114         }
01115         else if (strcmp(argname[i], "chksum0") == 0) {
01116             chksum_present = 1; /* Ignore the associated value */
01117         }
01118     }
01119     bio_hdrarg_free(argname, argval);
01120     argname = argval = NULL;
01121 
01122     chksum = 0;
01123 
01124     /* Read #senones, #features, #codewords, arraysize */
01125     if ((bio_fread(&n_sen, sizeof(int32), 1, fp, byteswap, &chksum) != 1)
01126         || (bio_fread(&n_feat, sizeof(int32), 1, fp, byteswap, &chksum) !=
01127             1)
01128         || (bio_fread(&n_comp, sizeof(int32), 1, fp, byteswap, &chksum) !=
01129             1)
01130         || (bio_fread(&n, sizeof(int32), 1, fp, byteswap, &chksum) != 1)) {
01131         E_FATAL("bio_fread(%s) (arraysize) failed\n", file_name);
01132     }
01133     if (n_feat != s->n_feat)
01134         E_FATAL("#Features streams(%d) != %d\n", n_feat, s->n_feat);
01135     if (n != n_sen * n_feat * n_comp) {
01136         E_FATAL
01137             ("%s: #float32s(%d) doesn't match header dimensions: %d x %d x %d\n",
01138              file_name, i, n_sen, n_feat, n_comp);
01139     }
01140 
01141     /* n_sen = number of mixture weights per codeword, which is
01142      * fixed at the number of senones since we have only one codebook.
01143      */
01144     s->n_sen = n_sen;
01145 
01146     /* Quantized mixture weight arrays. */
01147     s->mixw = ckd_calloc_3d(s->n_feat, s->n_density, n_sen, sizeof(***s->mixw));
01148 
01149     /* Temporary structure to read in floats before conversion to (int32) logs3 */
01150     pdf = (float32 *) ckd_calloc(n_comp, sizeof(float32));
01151 
01152     /* Read senone probs data, normalize, floor, convert to logs3, truncate to 8 bits */
01153     n_err = 0;
01154     for (i = 0; i < n_sen; i++) {
01155         for (f = 0; f < n_feat; f++) {
01156             if (bio_fread((void *) pdf, sizeof(float32),
01157                           n_comp, fp, byteswap, &chksum) != n_comp) {
01158                 E_FATAL("bio_fread(%s) (arraydata) failed\n", file_name);
01159             }
01160 
01161             /* Normalize and floor */
01162             if (vector_sum_norm(pdf, n_comp) <= 0.0)
01163                 n_err++;
01164             vector_floor(pdf, n_comp, SmoothMin);
01165             vector_sum_norm(pdf, n_comp);
01166 
01167             /* Convert to LOG, quantize, and transpose */
01168             for (c = 0; c < n_comp; c++) {
01169                 int32 qscr;
01170 
01171                 qscr = -logmath_log(s->lmath_8b, pdf[c]);
01172                 if ((qscr > MAX_NEG_MIXW) || (qscr < 0))
01173                     qscr = MAX_NEG_MIXW;
01174                 s->mixw[f][c][i] = qscr;
01175             }
01176         }
01177     }
01178     if (n_err > 0)
01179         E_WARN("Weight normalization failed for %d mixture weights components\n", n_err);
01180 
01181     ckd_free(pdf);
01182 
01183     if (chksum_present)
01184         bio_verify_chksum(fp, byteswap, chksum);
01185 
01186     if (fread(&eofchk, 1, 1, fp) == 1)
01187         E_FATAL("More data than expected in %s\n", file_name);
01188 
01189     fclose(fp);
01190 
01191     E_INFO("Read %d x %d x %d mixture weights\n", n_sen, n_feat, n_comp);
01192     return n_sen;
01193 }
