00072     cmd_ln_t *config = ps_search_config(search);
00073     acmod_t *acmod = ps_search_acmod(search);

00075 
00076     /* Free old dict2pid, dict, if necessary. */
00077     ps_search_base_reinit(search, dict, d2p);

00079     /* Initialize HMM context. */
00080     if (pls->hmmctx)
00081         hmm_context_free(pls->hmmctx);

00083                                    acmod->tmat->tp, NULL, acmod->mdef->sseq);
00084     if (pls->hmmctx == NULL)
00085         return -1;
00086 
00087     /* Initialize phone HMMs. */
00088     if (pls->phones) {
00089         for (i = 0; i < pls->n_phones; ++i)
00090             hmm_deinit((hmm_t *)&pls->phones[i]);
00091         ckd_free(pls->phones);
00092     }
00093     pls->n_phones = bin_mdef_n_ciphone(acmod->mdef);
00094     pls->phones = ckd_calloc(pls->n_phones, sizeof(*pls->phones));
00095     for (i = 0; i < pls->n_phones; ++i) {
00096         pls->phones[i].ciphone = i;
00097         hmm_init(pls->hmmctx, (hmm_t *)&pls->phones[i],
00098                  FALSE,
00099                  bin_mdef_pid2ssid(acmod->mdef, i),
00100                  bin_mdef_pid2tmatid(acmod->mdef, i));
00101     }
00102     pls->beam = logmath_log(acmod->lmath, cmd_ln_float64_r(config, "-pl_beam"));
00103     pls->pbeam = logmath_log(acmod->lmath, cmd_ln_float64_r(config, "-pl_pbeam"));
00104     pls->pip = logmath_log(acmod->lmath, cmd_ln_float64_r(config, "-pip"));
00105     E_INFO("State beam %d Phone exit beam %d Insertion penalty %d\n",
00106            pls->beam, pls->pbeam, pls->pip);

00186 static int32
00187 evaluate_hmms(phone_loop_search_t *pls, int16 const *senscr, int frame_idx)
00188 {
00189     int32 bs = WORST_SCORE;
00190     int i;
00191 
00192     hmm_context_set_senscore(pls->hmmctx, senscr);
00193 
00194     for (i = 0; i < pls->n_phones; ++i) {
00195         hmm_t *hmm = (hmm_t *)&pls->phones[i];
00196         int32 score;
00197 
00198         if (hmm_frame(hmm) < frame_idx)
00199             continue;
00200         score = hmm_vit_eval(hmm);
00201         if (score BETTER_THAN bs) {
00202             bs = score;
00203         }
00204     }
00205     pls->best_score = bs;
00206     return bs;
00207 }

00209 static void
00210 prune_hmms(phone_loop_search_t *pls, int frame_idx)
00211 {
00212     int32 thresh = pls->best_score + pls->beam;
00213     int nf = frame_idx + 1;
00214     int i;
00215 
00216     /* Check all phones to see if they remain active in the next frame. */
00217     for (i = 0; i < pls->n_phones; ++i) {
00218         hmm_t *hmm = (hmm_t *)&pls->phones[i];
00219 
00220         if (hmm_frame(hmm) < frame_idx)
00221             continue;
00222         /* Retain if score better than threshold. */
00223         if (hmm_bestscore(hmm) BETTER_THAN thresh) {
00224             hmm_frame(hmm) = nf;
00225         }
00226         else
00227             hmm_clear_scores(hmm);
00228     }
00229 }

00231 static void
00232 phone_transition(phone_loop_search_t *pls, int frame_idx)
00233 {
00234     int32 thresh = pls->best_score + pls->pbeam;
00235     int nf = frame_idx + 1;
00236     int i;
00237 
00238     /* Now transition out of phones whose last states are inside the
00239      * phone transition beam. */
00240     for (i = 0; i < pls->n_phones; ++i) {
00241         hmm_t *hmm = (hmm_t *)&pls->phones[i];
00242         int32 newphone_score;
00243         int j;
00244 
00245         if (hmm_frame(hmm) != nf)
00246             continue;
00247 
00248         newphone_score = hmm_out_score(hmm) + pls->pip;
00249         if (newphone_score BETTER_THAN thresh) {
00250             /* Transition into all phones using the usual Viterbi rule. */
00251             for (j = 0; j < pls->n_phones; ++j) {
00252                 hmm_t *nhmm = (hmm_t *)&pls->phones[j];
00253 
00254                 if (hmm_frame(nhmm) < frame_idx
00255                     || newphone_score BETTER_THAN hmm_in_score(nhmm)) {
00256                     hmm_enter(nhmm, newphone_score, hmm_out_history(hmm), nf);
00257                 }
00258             }
00259         }
00260     }
00261 }

00263 static int
00264 phone_loop_search_step(ps_search_t *search, int frame_idx)
00265 {
00266     phone_loop_search_t *pls = (phone_loop_search_t *)search;
00267     acmod_t *acmod = ps_search_acmod(search);
00268     int16 const *senscr;
00269     int i;
00270 
00271     /* All CI senones are active all the time. */
00272     if (!ps_search_acmod(pls)->compallsen)
00273         for (i = 0; i < pls->n_phones; ++i)
00274             acmod_activate_hmm(acmod, (hmm_t *)&pls->phones[i]);
00275 
00276     /* Calculate senone scores for current frame. */
00277     senscr = acmod_score(acmod, &frame_idx);
00278 
00279     /* Renormalize, if necessary. */
00280     if (pls->best_score + (2 * pls->beam) WORSE_THAN WORST_SCORE) {
00281         E_INFO("Renormalizing Scores at frame %d, best score %d\n",
00282                frame_idx, pls->best_score);
00283         renormalize_hmms(pls, frame_idx, pls->best_score);
00284     }
00285 
00286     /* Evaluate phone HMMs for current frame. */
00287     pls->best_score = evaluate_hmms(pls, senscr, frame_idx);
00288 
00289     /* Prune phone HMMs. */
00290     prune_hmms(pls, frame_idx);
00291 
00292     /* Do phone transitions. */
00293     phone_transition(pls, frame_idx);
00294 
00295     return 0;
00296 }
